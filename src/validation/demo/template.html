当前位置： Validation 如何引用
1.引入KISSY 
<script src="kissy.js"></script>
2.配置环境 
	KISSY.config({
	    packages:[{
	            name:"Validation",
	            path:"../"		//Auth存放的路径
	        }
	    ]
	});
3.使用Auth 
	KISSY.use("validation",function(S,Validation){

		var form = new Validation('#J_form',{
			//..config
		});

	});
