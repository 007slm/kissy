1 /**
  2  * @fileOverview This class specifies the definition for a cell editor of a grid.
  3  * @author dxq613@gmail.com
  4  */
  5 KISSY.add('grid/editor', function (S, Component) {
  6 
  7     var CLS_EDITOR = 'ks-grid-editor',
  8         CLS_EDITOR_ERROR = CLS_EDITOR + '-error',
  9         CLS_EDITOR_CONTROL = CLS_EDITOR + '-control';
 10 
 11     /**
 12      * the render of grid's editor
 13      * @private
 14      */
 15     var GridEditorRender = Component.Render.extend({
 16         /**
 17          * @private
 18          */
 19         createDom:function () {
 20             var _self = this,
 21                 el = _self.get('el'),
 22                 tpl = _self.get('tpl');
 23             new S.Node(tpl).appendTo(el);
 24         }
 25     });
 26 
 27     /**
 28      * This is a base class of grid's editor,which can be used in column's configuration.
 29      * @name Grid.Editor
 30      * @constructor
 31      * @extends Component.Controller
 32      */
 33     var GridEditor = Component.Controller.extend({
 34 
 35         /**
 36          * @private
 37          */
 38         bindUI:function () {
 39             var _self = this,
 40                 binder = _self.get('binder'),
 41                 control = _self.getEditControl(),
 42                 events = _self.get('events'),
 43                 triggerEvent = _self.get('triggerEvent');
 44             if (binder) {
 45                 binder.call(this);
 46             }
 47 
 48             S.each(events, function (event) {
 49                 _self.publish(event, {
 50                     bubbles:1
 51                 });
 52             });
 53 
 54             control.on(triggerEvent, function () {
 55                 var text = _self._getControlText(),
 56                     record = _self.get('record'),
 57                     valid = _self.validEditor(text, record);
 58                 if (valid) {
 59                     _self.fire('changed', {target:_self, text:text, value:_self.getValue(text)});
 60                 }
 61             });
 62         },
 63         /**
 64          * @protect
 65          * True to indicate that this editor contains the element
 66          */
 67         containsElement:function (element) {
 68             var _self = this,
 69                 el = _self.get('el');
 70             return el.contains(element) || el[0] == element;
 71         },
 72         /**
 73          * reset the editor
 74          */
 75         clearValue:function () {
 76             this.setValue('');
 77         },
 78         /**
 79          * make this editor's control focused
 80          */
 81         focus:function () {
 82             var _self = this,
 83                 control = _self.getEditControl();
 84             if (control && control[0] && control[0].focus) {
 85                 control[0].focus();
 86             }
 87         },
 88         /**
 89          * Get the element that the value in it.so as input,select,textarea and so on.
 90          * This element must have the method of 'val',so as Node,which can get or set value.
 91          * When you Inheritance this class,you can ovrride this method.
 92          */
 93         getEditControl:function () {
 94             var _self = this,
 95                 el = _self.get('el');
 96             return el.one('.' + CLS_EDITOR_CONTROL);
 97         },
 98         /**
 99          * Get the user's input,and format it.
100          * When you Inheritance this class,you can ovrride this method.
101          */
102         getValue:function (text) {
103             var _self = this,
104                 formatter = _self.get('formatter');
105             text = text || _self._getControlText();
106             return formatter(text);
107         },
108         /**
109          * Verify user input is correct
110          * @return {Boolean}
111          */
112         hasError:function () {
113             return this.get('el').hasClass(CLS_EDITOR_ERROR);
114         },
115         /**
116          * Set the cell's value.
117          * When you Inheritance this class,you can override this method.
118          */
119         setValue:function (v) {
120             var _self = this,
121                 control = _self.getEditControl();
122             control.val(v);
123         },
124         //the text of user's input
125         _getControlText:function () {
126             var _self = this,
127                 control = _self.getEditControl();
128             return control.val();
129         },
130         //get the error of user's input
131         _getError:function (text, record) {
132             var _self = this,
133             //
134                 basicValidator = _self.get('basicValidator'),
135                 validator = _self.get('validator');
136             record = record || _self.get('record');
137             return basicValidator(text) || validator(text, record);
138         },
139         /**
140          * Verify user input is correct,and show error if there are any error.
141          * @return {Boolean}
142          */
143         validEditor:function (text, record) {
144             var _self = this, errorMsg;
145             text = text || _self._getControlText();
146             errorMsg = _self._getError(text, record);
147             if (errorMsg) {
148                 _self.fire('error', {target:_self, msg:errorMsg, text:text});
149                 _self.get('el').addClass(CLS_EDITOR_ERROR);
150             } else {
151                 _self.get('el').removeClass(CLS_EDITOR_ERROR);
152             }
153             return !errorMsg;
154         },
155         //set the value of this component
156         _uiSetValue:function (v) {
157             var _self = this;
158             v = v ? v.toString() : '';
159             _self.setValue(v);
160             _self.validEditor();
161         },
162         _uiSetRecord:function (v) {
163             if (!v) {
164                 return;
165             }
166             var _self = this,
167                 field = _self.get('field');
168             if (field) {
169                 _self.set('value', v[field]);
170             }
171         }
172     }, {
173         ATTRS:/** * @lends Grid.Editor.prototype*/
174         {
175             /**
176              * This is a default function which is used to verify user's input.
177              * For example when the editor's type is 'number',You don't have to test the user input type is 'number'.
178              * If there is any error,the return value is a message which is not an empty string.
179              * This property is usually overridden by its subclass.
180              * @type {Function}
181              */
182             basicValidator:{
183                 value:function (v) {
184                     return '';
185                 }
186             },
187             /**
188              * When you use custom editor,you can config the tpl and binder that bind event to the custom dom.
189              * @type {Function}
190              * @default null;
191              */
192             binder:{
193 
194             },
195             /**
196              * The initial set of data to apply to the tpl to update the content area of the Component.
197              * @type {Object}
198              * @default {}
199              */
200             data:{
201                 value:{}
202             },
203             /**
204              * Verify if users can edit the cell, if can't edit, do not show.
205              * @param {Object} v The data value of the underlying field.
206              * @param {Object} record The record that user is editing.
207              * @return {Boolean}
208              */
209             editFunction:{
210                 value:function (v, record) {
211                     return false;
212                 }
213             },
214             /**
215              * the collection of editor's events
216              * @type {Array}
217              */
218             events:{
219                 value:[
220                 /**
221                  * @event changed
222                  * Fires when this editor's value changed
223                  * @param {event} e the event object
224                  * @param {Grid.Editor} target
225                  * @param {String} text the user's input text
226                  * @param {Object} value format the user's input text to value
227                  */
228                     'changed',
229                 /**
230                  * @event changed
231                  * Fires when this editor's value changed
232                  * @param {event} e the event object
233                  * @param {Grid.Editor} target
234                  * @param {String} msg the error msg of the user's input
235                  * @param {String} text the user's input text
236                  */
237                     'error'
238                 ]
239             },
240             /**
241              * The field of the record,which is being editing.
242              * @type {String}
243              */
244             field:{
245 
246             },
247             /**
248              * @private
249              * @override
250              */
251             focusable:{
252                 value:false
253             },
254             /**
255              * User input is usually text, so need to convert format.This is a function you can format user's input.
256              * @type {Function}
257              * @default function(v){return v;}
258              */
259             formatter:{
260                 value:function (v) {
261                     return v;
262                 }
263             },
264             /**
265              * The record which user is editing
266              * @type {Object}
267              * @default null
268              */
269             record:{
270                 value:null
271             },
272             /**
273              * The built-in validation rules,followed by kissy 's validation framework.
274              * @type {Object}
275              * @default null
276              */
277             rules:{
278 
279             },
280             /**
281              * The event which triggers editing,this event default is binded to the input of this editor. Maybe :
282              * <ul>
283              *     <li>change</li>
284              *     <li>blur</li>
285              * </ul>
286              */
287             triggerEvent:{
288                 value:'change'
289             },
290             /**
291              * An template used to create the internal structure inside this Component's encapsulating Element.
292              * User can use the syntax of KISSY's template component.
293              * Only in the configuration of the editor.
294              * @type {String}
295              * <pre>
296              *    '<input type="text" class="ks-grid-editor-control" />'
297              * </pre>
298              */
299             tpl:{
300                 view:true,
301                 value:'<input type="text" class="' + CLS_EDITOR_CONTROL + '"/>'
302             },
303             /**
304              * This is a function which is used to verify user's input.
305              * @type {Function}
306              * @default
307              *    <pre>function(v){ return '';}</pre>
308              */
309             validator:{
310                 value:function (v) {
311                     return '';
312                 }
313             },
314             /**
315              * The data value of the underlying field.
316              * When the editor showed,the value of the cell was be set in this component.
317              * After user's editing,the value of this component was be set to the cell.
318              * @type {Object}
319              */
320             value:{
321                 //get the value of this component
322                 getter:function () {
323                     return this.getValue();
324                 }
325             },
326             /**
327              * @private
328              */
329             xrender:{
330                 value:GridEditorRender
331             }
332         }
333     }, {
334         xclass:'grid-editor',
335         priority:1
336     });
337 
338     return GridEditor;
339 }, {
340     requires:['component']
341 });
