1 KISSY.add("editor/plugin/link/utils", function (S, Editor) {
  2 
  3     var Node = S.Node,
  4         KEStyle = Editor.Style,
  5         _ke_saved_href = "_ke_saved_href",
  6         link_Style = {
  7             element:'a',
  8             attributes:{
  9                 href:"#(href)",
 10                 title:"#(title)",
 11                 // ie < 8 会把锚点地址修改，以及相对地址改为绝对地址
 12                 // 1. 编辑器位于 http://x.com/edit.htm
 13                 // 2. 用户输入 ./a.htm
 14                 // 3. 生成为 <a href='http://x.com/a.htm'>
 15                 // 另一个问题 refer: http://stackoverflow.com/questions/687552/prevent-tinymce-internet-explorer-from-converting-urls-to-links
 16                 _ke_saved_href:"#(_ke_saved_href)",
 17                 target:"#(target)"
 18             }
 19         };
 20 
 21     function getAttributes(el) {
 22         var attributes = el.attributes,
 23             re = {};
 24         for (var i = 0; i < attributes.length; i++) {
 25             var a = attributes[i];
 26             if (a.specified) {
 27                 re[a.name] = a.value;
 28             }
 29         }
 30         if (el.style.cssText) {
 31             re.style = el.style.cssText;
 32         }
 33         return re;
 34     }
 35 
 36 
 37     function removeLink(editor, a) {
 38         editor.execCommand("save");
 39         var sel = editor.getSelection(),
 40             range = sel.getRanges()[0];
 41         if (range && range.collapsed) {
 42             var bs = sel.createBookmarks();
 43             // 不使用核心 styles ，直接清除元素标记即可。
 44             a._4e_remove(true);
 45             sel.selectBookmarks(bs);
 46         } else if (range) {
 47             var attrs = getAttributes(a[0]);
 48             new KEStyle(link_Style, attrs).remove(editor.get("document")[0]);
 49         }
 50         editor.execCommand("save");
 51         editor.notifySelectionChange();
 52     }
 53 
 54     function applyLink(editor, attr, _selectedEl) {
 55         // 注意同步，取的话要从 _ke_saved_href 取原始值的
 56         attr[_ke_saved_href] = attr.href;
 57         // 是修改行为
 58         if (_selectedEl) {
 59             editor.execCommand("save");
 60             _selectedEl.attr(attr);
 61         } else {
 62             var sel = editor.getSelection(),
 63                 range = sel && sel.getRanges()[0];
 64             //编辑器没有焦点或没有选择区域时直接插入链接地址
 65             if (!range || range.collapsed) {
 66                 var a = new Node("<a>" + attr.href + "</a>",
 67                     attr, editor.get("document")[0]);
 68                 editor.insertElement(a);
 69             } else {
 70                 editor.execCommand("save");
 71                 var linkStyle = new KEStyle(link_Style, attr);
 72                 linkStyle.apply(editor.get("document")[0]);
 73             }
 74         }
 75         editor.execCommand("save");
 76         editor.notifySelectionChange();
 77     }
 78 
 79 
 80     return {
 81         removeLink:removeLink,
 82         applyLink:applyLink,
 83         _ke_saved_href:_ke_saved_href
 84     }
 85 }, {
 86     requires:['editor']
 87 });
