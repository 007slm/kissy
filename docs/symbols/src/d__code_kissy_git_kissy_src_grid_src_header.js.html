1 /**
  2  * @fileOverview This class specifies the definition for a header of a grid.
  3  * @author dxq613@gmail.com, yiminghe@gmail.com
  4  */
  5 KISSY.add('grid/header', function (S, Component, Column) {
  6 
  7     var CLS_SCROLL_WITH = 17;
  8 
  9     var headerRender = Component.Render.extend({
 10 
 11         renderUI:function () {
 12             var _self = this,
 13                 el = _self.get('el'),
 14                 tableCls = _self.get('tableCls'),
 15                 temp = '<table cellspacing="0" class="ks-grid-table" cellpadding="0">' +
 16                     '<thead><tr></tr></thead>' +
 17                     '</table>',
 18                 tableEl = new S.Node(temp).appendTo(el);
 19             tableEl.addClass(tableCls);
 20             _self.set('tableEl', tableEl);
 21         },
 22         /**
 23          * @see {Component.Render#getContentElement}
 24          */
 25         getContentElement:function () {
 26             return this.get('el').one('tr');
 27         },
 28         scrollTo:function (obj) {
 29             var _self = this,
 30                 el = _self.get('el');
 31             if (obj.top !== undefined) {
 32                 el.scrollTop(obj.top);
 33             }
 34             if (obj.left !== undefined) {
 35                 el.scrollLeft(obj.left);
 36             }
 37         }
 38     }, {
 39         ATTRS:{
 40             emptyCellEl:{},
 41             tableEl:{}
 42         }
 43     });
 44     /**
 45      * Container which holds headers and is docked at the top or bottom of a Grid.
 46      * The HeaderContainer drives resizing/moving/hiding of columns within the GridView.
 47      * As headers are hidden, moved or resized,
 48      * the header container is responsible for triggering changes within the view.
 49      * @name Grid.Header
 50      * @constructor
 51      * @extends Component.Controller
 52      */
 53     var header = Component.Controller.extend(
 54         /**
 55          * @lends Grid.Header.prototype
 56          */
 57         {
 58             /**
 59              * add a columns to header
 60              * @param {Object|Grid.Column} c The column object or column config.
 61              * @index {Number} index The position of the column in a header,0 based.
 62              */
 63             addColumn:function (c, index) {
 64                 var _self = this,
 65                     insertIndex = 0,
 66                     columns = _self.get('columns');
 67                 c = _self._createColumn(c);
 68                 if (index === undefined) {
 69                     index = columns.length;
 70                     insertIndex = _self.get('children').length - 1;
 71                 }
 72                 columns.splice(index, 0, c);
 73                 _self.addChild(c, insertIndex);
 74                 _self.fire('add', {column:c, index:index});
 75                 return c;
 76             },
 77             /**
 78              * remove a columns from header
 79              * @param {Grid.Column|Number} c is The column object or The position of the column in a header,0 based.
 80              */
 81             removeColumn:function (c) {
 82                 var _self = this,
 83                     columns = _self.get('columns'),
 84                     index;
 85                 c = S.isNumber(c) ? columns[c] : c;
 86                 index = S.indexOf(c, columns);
 87                 columns.splice(index, 1);
 88                 _self.fire('remove', {column:c, index:index});
 89                 return _self.removeChild(c, true);
 90             },
 91             /**
 92              * For overridden.
 93              * @see Component.Controller#bindUI
 94              */
 95             bindUI:function () {
 96                 var _self = this;
 97                 _self._bindColumnsEvent();
 98             },
 99             /*
100              * For overridden.
101              * @protected
102              * @override
103              */
104             initializer:function () {
105                 var _self = this,
106                     children = _self.get('children'),
107                     columns = _self.get('columns');
108                 S.each(columns, function (item, index) {
109                     var columnControl = _self._createColumn(item);
110                     children[index] = columnControl;
111                     columns[index] = columnControl;
112                 });
113                 children.push(_self._createEmptyColumn());
114             },
115             /**
116              * get the columns of this header,the result equals the 'children' property .
117              * @return {Array} columns
118              * @example var columns = header.getColumns();
119              *    <br>or</br>
120              * var columns = header.get('children');
121              */
122             getColumns:function () {
123                 return this.get('columns');
124             },
125             /**
126              * Obtain the sum of the width of all columns
127              * @return {Number}
128              */
129             getColumnsWidth:function () {
130                 var _self = this,
131                     columns = _self.getColumns(),
132                     totalWidth = 0;
133 
134                 S.each(columns, function (column) {
135                     if (!column.get('hide')) {
136                         totalWidth += column.get("el").outerWidth();
137                     }
138                 });
139                 return totalWidth;
140             },
141             /**
142              * get {@link Grid.Column} instance by index,when column moved ,the index changed.
143              * @param {Number} index The index of columns
144              * @return {Grid.Column} the column in the header,if the index outof the range,the result is null
145              */
146             getColumnByIndex:function (index) {
147                 var _self = this,
148                     columns = _self.getColumns(),
149                     result = columns[index];
150                 return result;
151             },
152             /**
153              * get {@link Grid.Column} instance by id,when column rendered ,this id can't to be changed
154              * @param {String|Number}id The id of columns
155              * @return {Grid.Column} the column in the header,if the index out of the range,the result is null
156              */
157             getColumnById:function (id) {
158                 var _self = this,
159                     columns = _self.getColumns(),
160                     result = null;
161                 S.each(columns, function (column) {
162                     if (column.get('id') === id) {
163                         result = column;
164                         return false;
165                     }
166                 });
167                 return result;
168             },
169             /**
170              * get {@link Grid.Column} instance's index,when column moved ,the index changed.
171              * @param {Grid.Column} column The instance of column
172              * @return {Number} the index of column in the header,if the column not in the header,the index is -1
173              */
174             getColumnIndex:function (column) {
175                 var _self = this,
176                     columns = _self.getColumns();
177                 return S.indexOf(column, columns);
178             },
179             /**
180              * move the header followed by body's or document's scrolling
181              * @param {Object} obj the scroll object which has two value:top(scrollTop),left(scrollLeft)
182              */
183             scrollTo:function (obj) {
184                 this.get('view').scrollTo(obj);
185             },
186             //when column's event fire ,this header must handle them.
187             _bindColumnsEvent:function () {
188                 var _self = this;
189 
190                 _self.on('afterWidthChange', function (e) {
191                     var sender = e.target;
192                     if (sender !== _self) {
193                         _self.setTableWidth();
194                     }
195                 });
196                 _self.on('afterVisibleChange', function (e) {
197                     var sender = e.target;
198                     if (sender !== _self) {
199                         _self.setTableWidth();
200                     }
201                 });
202                 _self.on('afterSortStateChange', function (e) {
203                     var sender = e.target,
204                         columns = _self.getColumns(),
205                         val = e.newVal;
206                     if (val) {
207                         S.each(columns, function (column) {
208                             if (column !== sender) {
209                                 column.set('sortState', '');
210                             }
211                         });
212                     }
213                 });
214             },
215             //create the column control
216             _createColumn:function (cfg) {
217                 if (cfg instanceof Column) {
218                     return cfg;
219                 }
220                 if (!cfg.id) {
221                     cfg.id = S.guid('col');
222                 }
223                 return new Column(cfg);
224             },
225             _createEmptyColumn:function () {
226                 return new Column.Empty();
227             },
228             //when set grid's height, scroll bar emerged.
229             _isAllowScrollLeft:function () {
230                 var _self = this,
231                     parent = _self.get('parent');
232 
233                 return parent && !!parent.get('height');
234             },
235             /**
236              * force every column fit the table's width
237              */
238             forceFitColumns:function () {
239                 var _self = this,
240                     columns = _self.getColumns(),
241                     width = _self.get('width'),
242                     allowScroll = _self._isAllowScrollLeft();
243 
244                 //if there is not a width config of grid ,The forceFit action can't work
245                 if (width) {
246 
247                     if (allowScroll) {
248                         width -= CLS_SCROLL_WITH;
249                     }
250 
251                     var adjustCount = 0;
252 
253                     S.each(columns, function (column) {
254                         if (!column.get('hide') && column.get('resizable')) {
255                             adjustCount++;
256                         }
257                         if (!column.get('hide') && !column.get('resizable')) {
258                             width -= column.get("el").outerWidth();
259                         }
260                     });
261 
262                     var colWidth = width / adjustCount;
263 
264                     S.each(columns, function (column) {
265                         if (!column.get('hide') && column.get('resizable')) {
266                             var columnEl = column.get("el");
267                             var borderWidth =
268                                 parseInt(columnEl.css("border-left-width")) || 0 +
269                                     parseInt(columnEl.css("border-right-width")) || 0;
270                             // ！ note
271                             //
272                             // 会再调用 setTableWidth， 循环调用
273                             column.set("width", colWidth - borderWidth, {
274                                 silent:1
275                             });
276                             columnEl.width(colWidth - borderWidth);
277                         }
278                     });
279 
280                     _self.fire('forceFitWidth');
281                 }
282             },
283             /**
284              * set the header's inner table's width
285              */
286             setTableWidth:function () {
287                 var _self = this;
288                 if (_self.get('forceFit')) {
289                     _self.forceFitColumns();
290                 }
291             },
292             //when header's width changed, it also effects its columns.
293             _uiSetWidth:function () {
294                 var _self = this;
295                 _self.setTableWidth();
296             },
297             _uiSetForceFit:function (v) {
298                 var _self = this;
299                 if (v) {
300                     _self.setTableWidth();
301                 }
302             }
303 
304         }, {
305             ATTRS:/** * @lends Grid.Header.prototype*/
306             {
307 
308                 /**
309                  *  A string column id or the numeric index of the column
310                  * that should be initially activated within the container's layout on render.
311                  * @type String|Number
312                  * @default null
313                  */
314                 activeItem:{
315                     value:null
316                 },
317                 /**
318                  * the collection of columns
319                  */
320                 columns:{
321                     value:[]
322                 },
323                 /**
324                  * @private
325                  */
326                 emptyColumn:{
327 
328                 },
329                 /**
330                  * @protected
331                  */
332                 focusable:{
333                     value:false
334                 },
335                 /**
336                  * true to force the columns to fit into the available width. Headers are first sized according to configuration, whether that be a specific width, or flex.
337                  * Then they are all proportionally changed in width so that the entire content width is used.
338                  * @type {Boolean}
339                  * @default 'false'
340                  */
341                 forceFit:{
342                     view:true,
343                     value:false
344                 },
345                 /**
346                  * The CSS class to apply to this header's table element.
347                  */
348                 tableCls:{
349                     view:true,
350                     value:'ks-grid-table'
351                 },
352                 /**
353                  * @private
354                  */
355                 xrender:{
356                     value:headerRender
357                 },
358                 /**
359                  * the collection of header's events
360                  * @type Array
361                  */
362                 events:{
363                     value:[
364                     /**
365                      * @event Grid.Header#add
366                      * Fires when this column's width changed
367                      * @param {event} e the event object
368                      * @param {Grid.Column} e.column which column added
369                      * @param {Number} index the add column's index in this header
370                      *
371                      */
372                         'add',
373                     /**
374                      * @event Grid.Header#add
375                      * Fires when this column's width changed
376                      * @param {event} e the event object
377                      * @param {Grid.Column} e.column which column removed
378                      * @param {Number} index the removed column's index in this header
379                      */
380                         'remove'
381                     ]
382                 }
383             }
384         }, {
385             xclass:'grid-header',
386             priority:1
387         });
388 
389     return header;
390 }, {
391     requires:['component', './column']
392 });
