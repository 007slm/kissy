1 /**
  2  * source editor for kissy editor
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/sourceArea/index", function (S, Editor, B) {
  6 
  7     var SOURCE_MODE = Editor.SOURCE_MODE ,
  8         WYSIWYG_MODE = Editor.WYSIWYG_MODE;
  9     return {
 10         init:function (editor) {
 11             editor.addButton({
 12                 title:"源码",
 13                 contentCls:"ks-editor-toolbar-source"
 14             }, {
 15                 init:function () {
 16                     var self = this;
 17                     editor.on("wysiwygMode", self.boff, self);
 18                     editor.on("sourceMode", self.bon, self);
 19                 },
 20                 offClick:function () {
 21                     editor.set("mode", SOURCE_MODE);
 22                 },
 23                 onClick:function () {
 24                     editor.set("mode", WYSIWYG_MODE);
 25                 }
 26             });
 27         }
 28     };
 29 }, {
 30     requires:['editor', '../button/']
 31 });
