1 /**
  2  * Set up editor constructor
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/core/base", function (S, HtmlParser, Component, UIBase) {
  6     var PREFIX = "editor/plugin/", SUFFIX = "/";
  7 
  8     /**
  9      * KISSY Editor
 10      * @class
 11      * @extends Component.Controller
 12      * @extends UIBase.Box
 13      * @name Editor
 14      */
 15     var Editor = UIBase.create(Component.Controller, [UIBase.Box],
 16         /**
 17          * @lends Editor#
 18          */
 19         {
 20             initializer:function () {
 21                 var self = this, textarea;
 22                 self.__commands = {};
 23                 self.__dialogs = {};
 24                 if (textarea = self.get("textarea")) {
 25                     if (!self.get("render") && !self.get("elBefore")) {
 26                         var next = textarea.next();
 27                         if (next) {
 28                             self.__set("elBefore", next);
 29                         } else {
 30                             self.__set("render", textarea.parent());
 31                         }
 32                     }
 33                     if (!self.get("width")) {
 34                         self.__set("width", textarea.style("width") || textarea.css("width"));
 35                     }
 36                     if (!self.get("height")) {
 37                         self.__set("height", textarea.style("height") || textarea.css("height"));
 38                     }
 39                 } else {
 40                     self.__editor_created_new = 1;
 41                 }
 42             },
 43             use:function (mods, callback) {
 44                 var self = this,
 45                     BASIC = self.__CORE_PLUGINS || [
 46                         "htmlDataProcessor",
 47                         "enterKey",
 48                         "clipboard",
 49                         "selection"
 50                     ];
 51 
 52                 mods = mods.split(",");
 53 
 54                 for (var l = mods.length - 1; l >= 0; l--) {
 55                     if (!mods[l]) {
 56                         mods.splice(l, 1);
 57                     }
 58                 }
 59 
 60                 for (var i = 0; i < BASIC.length; i++) {
 61                     var b = BASIC[i];
 62                     if (!S.inArray(b, mods)) {
 63                         mods.unshift(b);
 64                     }
 65                 }
 66 
 67                 S.each(mods, function (m, i) {
 68                     if (mods[i]) {
 69                         mods[i] = PREFIX + m + SUFFIX;
 70                     }
 71                 });
 72 
 73                 function useMods(modFns) {
 74                     // 载入了插件的attach功能，现在按照顺序一个个attach
 75                     for (var i = 0; i < modFns.length; i++) {
 76                         modFns[i].init(self);
 77                     }
 78                     callback && callback.call(self);
 79                 }
 80 
 81                 //编辑器实例 use 时会进行编辑器 ui 操作而不单单是功能定义，必须 ready
 82                 S.use(mods.join(","), function () {
 83                     var args = S.makeArray(arguments);
 84                     args.shift();
 85                     useMods(args);
 86                 });
 87 
 88                 self.__CORE_PLUGINS = [];
 89 
 90                 return self;
 91             }
 92         },
 93 
 94         {
 95             Config:{
 96                 base:S.Config.base + "editor/"
 97             },
 98             XHTML_DTD:HtmlParser['DTD'],
 99             ATTRS:/**
100              * @lends Editor#
101              */
102             {
103                 /**
104                  * textarea 元素
105                  * @type Node
106                  */
107                 textarea:{},
108                 /**
109                  * iframe 元素
110                  * @type Node
111                  */
112                 iframe:{},
113                 /**
114                  * iframe 中的 contentWindow
115                  * @type Node
116                  */
117                 window:{},
118                 /**
119                  * iframe 中的 document
120                  * @type Node
121                  */
122                 document:{},
123                 /**
124                  * iframe 元素的 父节点
125                  * @type Node
126                  */
127                 iframeWrapEl:{},
128                 /**
129                  * 工具栏节点
130                  * @type Node
131                  */
132                 toolBarEl:{},
133                 /**
134                  * 状态栏节点
135                  * @type Node
136                  */
137                 statusBarEl:{},
138                 handleMouseEvents:{
139                     value:false
140                 },
141                 focusable:{
142                     value:false
143                 },
144                 /**
145                  * 编辑器当前模式：源码模式或可视化模式
146                  * @default 可视化模式
147                  */
148                 mode:{
149                     value:1
150                 },
151                 /**
152                  * 编辑器当前内容
153                  * @type String
154                  */
155                 data:{
156                     getter:function () {
157                         return this._getData();
158                     },
159                     setter:function (v) {
160                         return this._setData(v);
161                     }
162                 },
163                 /**
164                  * 编辑器经过格式化的当前内容
165                  * @type String
166                  */
167                 formatData:{
168                     getter:function () {
169                         return this._getData(1);
170                     },
171                     setter:function (v) {
172                         return this._setData(v);
173                     }
174                 },
175                 prefixCls:{
176                     value:"ke-"
177                 }
178             }
179         }, "Editor");
180 
181     Editor.DefaultRender = UIBase.create(Component.Render, [UIBase.Box.Render], {
182         /**
183          * 高度不在 el 上设置，设置 iframeWrap 以及 textarea（for ie）
184          * width 依然在 el 上设置
185          */
186         _uiSetHeight:function () {
187         }
188     });
189 
190     S.mix(Editor, S.EventTarget);
191 
192     KISSY.Editor = Editor;
193 
194     return Editor;
195 }, {
196     requires:['htmlparser', 'component', 'uibase', 'core']
197 });
