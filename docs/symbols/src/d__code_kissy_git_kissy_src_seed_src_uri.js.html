 13         URI_SPLIT_REG = new RegExp(
 14             '^' +
 15                 /*
 16                  Scheme names consist of a sequence of characters beginning with a
 17                  letter and followed by any combination of letters, digits, plus
 18                  ("+"), period ("."), or hyphen ("-").
 19                  */
 20                 '(?:([\\w\\d+.-]+):)?' + // scheme
 21 
 22                 '(?://' +
 23                 /*
 24                  The authority component is preceded by a double slash ("//") and is
 25                  terminated by the next slash ("/"), question mark ("?"), or number
 26                  sign ("#") character, or by the end of the URI.
 27                  */
 28                 '(?:([^/?#@]*)@)?' + // userInfo
 29 
 30                 '(' +
 31                 '[\\w\\d\\-\\u0100-\\uffff.+%]*' +
 32                 '|' +
 33                 // ipv6
 34                 '\\[[^\\]]+\\]' +
 35                 ')' + // hostname - restrict to letters,
 36                 // digits, dashes, dots, percent
 37                 // escapes, and unicode characters.
 38                 '(?::([0-9]+))?' + // port
 39                 ')?' +
 40                 /*
 41                  The path is terminated
 42                  by the first question mark ("?") or number sign ("#") character, or
 43                  by the end of the URI.
 44                  */
 45                 '([^?#]+)?' + // path. hierarchical part
 46                 /*
 47                  The query component is indicated by the first question
 48                  mark ("?") character and terminated by a number sign ("#") character
 49                  or by the end of the URI.
 50                  */
 51                 '(?:\\?([^#]*))?' + // query. non-hierarchical data
 52                 /*
 53                  The fragment identifier component of a URI allows indirect
 54                  identification of a secondary resource by reference to a primary
 55                  resource and additional identifying information.
 56 
 57                  A
 58                  fragment identifier component is indicated by the presence of a
 59                  number sign ("#") character and terminated by the end of the URI.
 60                  */
 61                 '(?:#(.*))?' + // fragment
 62                 '$'),

 76     function parseQuery(self) {
 77         if (!self._queryMap) {
 78             self._queryMap = S.unparam(self._query);
 79         }

208             var self = this,
209                 _queryMap,
210                 currentValue;
211             if (S.isObject(key)) {
212                 if (key instanceof Query) {
213                     key = key.get();
214                 }
215                 S.each(key, function (v, k) {
216                     self.add(k, v);
217                 });
218             } else {
219                 parseQuery(self);
220                 _queryMap = self._queryMap;
221                 currentValue = _queryMap[key];
222                 if (currentValue === undefined) {
223                     currentValue = value;
224                 } else {
225                     currentValue = [].concat(currentValue).concat(value);
226                 }
227                 _queryMap[key] = currentValue;
228             }
229             return self;

244     function padding2(str) {
245         return str.length == 1 ? "0" + str : str;
246     }

248     function equalsIgnoreCase(str1, str2) {
249         return str1.toLowerCase() == str2.toLowerCase();

252     // www.ta#bao.com // => www.ta.com/#bao.com
253     // www.ta%23bao.com
254     // Percent-Encoding
255     function encodeSpecialChars(str, specialCharsReg) {
256         // encodeURI( ) is intended to encode complete URIs,
257         // the following ASCII punctuation characters,
258         // which have special meaning in URIs, are not escaped either:
259         // ; / ? : @ & = + $ , #
260         return encodeURI(str).replace(specialCharsReg, function (m) {
261             return "%" + padding2(m.charCodeAt(0).toString(16));
262         });
263     }

266     /**
267      * @class
268      * Uri class for KISSY.
269      * Most of its interfaces are same with window.location.
270      * @param {String|KISSY.Uri} [uriStr] Encoded uri string.
271      * @memberOf KISSY
272      */
273     function Uri(uriStr) {
274 
275         if (uriStr instanceof  Uri) {
276             return uriStr.clone();
277         }
278 
279         var m, self = this;
280 
281         S.mix(self,
282             /**
283              * @lends KISSY.Uri#
284              */
285             {
286                 /**
287                  * scheme such as "http:". aka protocol without colon
288                  * @type String
289                  */
290                 scheme:"",
291                 /**
292                  * User credentials such as "yiminghe:gmail"
293                  * @type {String}
294                  */
295                 userInfo:"",
296                 /**
297                  * hostname such as "docs.kissyui.com". aka domain
298                  * @type {String}
299                  */
300                 hostname:"",
301                 /**
302                  * Port such as "8080"
303                  * @type {String}
304                  */
305                 port:"",
306                 /**
307                  * path such as "/index.htm". aka pathname
308                  * @type {String}
309                  */
310                 path:"",
311                 /**
312                  * Query object for search string. aka search
313                  * @type {KISSY.Uri.Query}
314                  */
315                 query:"",
316                 /**
317                  * fragment such as "#!/test/2". aka hash
318                  */
319                 fragment:""
320             });
321 
322         uriStr = uriStr || "";
323         m = uriStr.match(URI_SPLIT_REG) || [];
324 
325         S.each(REG_INFO, function (index, key) {
326             var match = m[index] || "";
327             if (key == "query") {
328                 // need encoded content
329                 self.query = new Query(match);
330             } else {
331                 // need to decode to get data structure in memory
332                 self[key] = decodeURIComponent(match);
333             }
334         });

360         /**
361          * The reference resolution algorithm.rfc 5.2
362          * return a resolved uri corresponding to current uri
363          * @param {KISSY.Uri|String} relativeUri
364          * @example
365          * <code>
366          *   this: "http://y/yy/z.com?t=1#v=2"
367          *   "https:/y/" => "https:/y/"
368          *   "//foo" => "http://foo"
369          *   "foo" => "http://y/yy/foo"
370          *   "/foo" => "http://y/foo"
371          *   "?foo" => "http://y/yy/z.com?foo"
372          *   "#foo" => http://y/yy/z.com?t=1#foo"
373          * </code>
374          * @return {KISSY.Uri}
375          */
376         resolve:function (relativeUri) {
377 
378             if (S.isString(relativeUri)) {
379                 relativeUri = new Uri(relativeUri);
380             }
381 
382             var self = this,
383                 override = 0,
384                 lastSlashIndex,
385                 order = ["scheme", "userInfo", "hostname", "port", "path", "query", "fragment"],
386                 target = self.clone();
387 
388             S.each(order, function (o) {
389                 if (o == "path") {
390                     // relativeUri does not set for scheme/userInfo/hostname/port
391                     if (override) {
392                         target[o] = relativeUri[o];
393                     } else {
394                         var path = relativeUri['path'];
395                         if (path) {
396                             // force to override target 's query with relative
397                             override = 1;
398                             if (!S.startsWith(path, "/")) {
399                                 if (target.hostname && !target.path) {
400                                     // RFC 3986, section 5.2.3, case 1
401                                     path = "/" + path;
402                                 } else if (target.path) {
403                                     // RFC 3986, section 5.2.3, case 2
404                                     lastSlashIndex = target.path.lastIndexOf('/');
405                                     if (lastSlashIndex != -1) {
406                                         path = target.path.slice(0, lastSlashIndex + 1) + path;
407                                     }
408                                 }
409                             }
410                             // remove .. / .  as part of the resolution process
411                             target.path = Path.normalize(path);
412                         }
413                     }
414                 } else if (o == "query") {
415                     if (override || relativeUri['query'].toString()) {
416                         target.query = relativeUri['query'].clone();
417                         override = 1;
418                     }
419                 } else if (override || relativeUri[o]) {
420                     target[o] = relativeUri[o];
421                     override = 1;
422                 }
423             });
424 
425             return target;
426 
427         },

576         /**
577          * serialize to string.
578          * rfc 5.3 Component Recomposition.
579          * but kissy does not differentiate between undefined and empty.
580          * @param {boolean} [serializeArray=true]
581          * whether append [] to key name when value 's type is array
582          * @return {String}
583          */
584         toString:function (serializeArray) {
585 
586             var out = [], self = this,
587                 scheme,
588                 hostname,
589                 path,
590                 port,
591                 fragment,
592                 query,
593                 userInfo;
594 
595             if (scheme = self.scheme) {
596                 out.push(encodeSpecialChars(scheme, reDisallowedInSchemeOrUserInfo));
597                 out.push(":");
598             }
599 
600             if (hostname = self.hostname) {
601                 out.push("//");
602                 if (userInfo = self.userInfo) {
603                     out.push(encodeSpecialChars(userInfo, reDisallowedInSchemeOrUserInfo));
604                     out.push("@");
605                 }
606 
607                 out.push(encodeURIComponent(hostname));
608 
609                 if (port = self.port) {
610                     out.push(":");
611                     out.push(port);
612                 }
613             }
614 
615             if (path = self.path) {
616                 if (hostname && !S.startsWith(path, "/")) {
617                     path = "/" + path;
618                 }
619                 path = Path.normalize(path);
620                 out.push(encodeSpecialChars(path, reDisallowedInPathName));
621             }
622 
623             if (query = ( self.query.toString(serializeArray))) {
624                 out.push("?");
625                 out.push(query);
626             }
627 
628             if (fragment = self.fragment) {
629                 out.push("#");
630                 out.push(encodeSpecialChars(fragment, reDisallowedInFragment))
631             }
632 
633             return out.join("");
634         }
