1 /**
  2  * flash dialog
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/flash/dialog", function (S, Editor, flashUtils, Overlay4E,Select) {
  6     var CLS_FLASH = 'ke_flash',
  7         TYPE_FLASH = 'flash',
  8         Dialog = Overlay4E.Dialog,
  9         TIP = "请输入如 http://www.xxx.com/xxx.swf",
 10         MIDDLE = "vertical-align:middle",
 11         bodyHtml = "<div style='padding:20px 20px 0 20px'>" +
 12             "<p>" +
 13             "<label>网址： " +
 14             "<input " +
 15             " data-verify='^https?://[^\\s]+$' " +
 16             " data-warning='网址格式为：http://' " +
 17             "class='ke-flash-url ke-input' style='width:300px;" +
 18             MIDDLE + "' />" +
 19             "</label>" +
 20             "</p>" +
 21             "<table style='margin:10px 0 5px  40px;width:300px;'>" +
 22             "<tr>" +
 23             "<td>" +
 24             "<label>宽度： " +
 25             "<input " +
 26             " data-verify='^(?!0$)\\d+$' " +
 27             " data-warning='宽度请输入正整数' " +
 28             "class='ke-flash-width ke-input' style='width:60px;" +
 29             "margin-left:2px;" +
 30             MIDDLE + "' /> 像素 </label>" +
 31             "</td>" +
 32             "<td>" +
 33             "<label>高度：<input " +
 34             " data-verify='^(?!0$)\\d+$' " +
 35             " data-warning='高度请输入正整数' " +
 36             "class='ke-flash-height ke-input' " +
 37             "style='width:60px;" +
 38             MIDDLE + "' /> 像素 </label></td>" +
 39             "</tr>" +
 40             "<tr>" +
 41             "<td>" +
 42             "<label>对齐： " +
 43             "<select class='ke-flash-align' title='对齐'>" +
 44             "<option value='none'>无</option>" +
 45             "<option value='left'>左对齐</option>" +
 46             "<option value='right'>右对齐</option>" +
 47             "</select>" +
 48             "</td>" +
 49             "<td>" +
 50             "<label>间距：" +
 51             "<input " +
 52             " data-verify='^\\d+$' " +
 53             " data-warning='间距请输入非负整数' "
 54             + "class='ke-flash-margin ke-input' " +
 55             "style='width:60px;" +
 56             MIDDLE + "' value='"
 57             + 5 + "'/> 像素" +
 58             "</label>" +
 59             "</td></tr>" +
 60             "</table>" +
 61             "</div>",
 62         footHtml = "<div style='padding:5px 20px 20px;'><a " +
 63             "class='ke-flash-ok ke-button' " +
 64             "style='margin-left:40px;margin-right:20px;'>确定</a> " +
 65             "<a class='ke-flash-cancel ke-button'>取消</a></div>";
 66 
 67 
 68     function FlashDialog(editor) {
 69         var self = this;
 70         self.editor = editor;
 71         Editor.Utils.lazyRun(self, "_prepareShow", "_realShow");
 72         self._config();
 73     }
 74 
 75     S.augment(FlashDialog, {
 76         addRes:Editor.Utils.addRes,
 77         destroyRes:Editor.Utils.destroyRes,
 78         _config:function () {
 79             var self = this;
 80             self._urlTip = TIP;
 81             self._type = TYPE_FLASH;
 82             self._cls = CLS_FLASH;
 83             self._config_dwidth = "400px";
 84             self._title = "Flash";//属性";
 85             self._bodyHtml = bodyHtml;
 86             self._footHtml = footHtml;
 87         },
 88         //建立弹出窗口
 89         _prepareShow:function () {
 90             var self = this;
 91             self.dialog = new Dialog({
 92                 autoRender:true,
 93                 headerContent:self._title,
 94                 bodyContent:self._bodyHtml,
 95                 footerContent:self._footHtml,
 96                 width:self._config_dwidth || "500px",
 97                 mask:true
 98             });
 99             self.addRes(self.dialog);
100             self._initD();
101         },
102         _realShow:function () {
103             //显示前就要内容搞好
104             this._updateD();
105             this.dialog.show();
106         },
107         /**
108          * 子类覆盖，如何从flash url得到合适的应用表示地址
109          * @override
110          * @param r flash 元素
111          */
112         _getFlashUrl:function (r) {
113             return flashUtils.getUrl(r);
114         },
115         /**
116          * 触发前初始化窗口 field，子类覆盖
117          * @override
118          */
119         _updateD:function () {
120             var self = this,
121                 editor = self.editor,
122                 cfg = editor.get("pluginConfig")["flash"] || {},
123                 f = self.selectedFlash;
124             if (f) {
125                 var r = editor.restoreRealElement(f);
126                 if (!r) {
127                     return;
128                 }
129                 if (f.css("width")) {
130                     self.dWidth.val(parseInt(f.css("width")));
131                 }
132                 if (f.css("height")) {
133                     self.dHeight.val(parseInt(f.css("height")));
134                 }
135                 self.dAlign.val(f.css("float"));
136                 Editor.Utils.valInput(self.dUrl, self._getFlashUrl(r));
137                 self.dMargin.val(parseInt(r.style("margin")) || 0);
138             } else {
139                 Editor.Utils.resetInput(self.dUrl);
140                 self.dWidth.val(cfg['defaultWidth'] || "");
141                 self.dHeight.val(cfg['defaultHeight'] || "");
142                 self.dAlign.val("none");
143                 self.dMargin.val("5");
144             }
145         },
146         show:function (_selectedEl) {
147             var self = this;
148             self.selectedFlash = _selectedEl;
149             self._prepareShow();
150         },
151 
152 
153         /**
154          * 映射窗口field，子类覆盖
155          * @override
156          */
157         _initD:function () {
158             var self = this,
159                 d = self.dialog,
160                 el = d.get("el");
161             self.dHeight = el.one(".ke-flash-height");
162             self.dWidth = el.one(".ke-flash-width");
163             self.dUrl = el.one(".ke-flash-url");
164             self.dAlign = Select.decorate(el.one(".ke-flash-align"));
165             self.dMargin = el.one(".ke-flash-margin");
166             var action = el.one(".ke-flash-ok"),
167                 cancel = el.one(".ke-flash-cancel");
168             action.on("click", self._gen, self);
169             cancel.on("click", function (ev) {
170                 d.hide();
171                 ev && ev.halt();
172             });
173 
174             Editor.Utils.placeholder(self.dUrl, self._urlTip);
175             self.addRes(action, cancel, self.dUrl);
176         },
177 
178         /**
179          * 应用子类覆盖，提供 flash 元素的相关信息
180          * @override
181          */
182         _getDInfo:function () {
183             var self = this;
184             return {
185                 url:self.dUrl.val(),
186                 attrs:{
187                     width:self.dWidth.val(),
188                     height:self.dHeight.val(),
189                     style:"margin:" +
190                         (parseInt(self.dMargin.val()) || 0) +
191                         "px;" +
192                         "float:" + self.dAlign.val() + ";"
193                 }
194             };
195         },
196 
197         /**
198          * 真正产生 flash 元素
199          */
200         _gen:function (ev) {
201             ev && ev.halt();
202             var self = this,
203                 editor = self.editor,
204                 dinfo = self._getDInfo(),
205                 url = dinfo && S.trim(dinfo.url),
206                 attrs = dinfo && dinfo.attrs;
207             if (!dinfo) {
208                 return;
209             }
210             var re = Editor.Utils.verifyInputs(self.dialog.get("el").all("input"));
211             if (!re) {
212                 return;
213             }
214             self.dialog.hide();
215             flashUtils.insertFlash(editor, url, attrs, self._cls,
216                 self._type, function (substitute) {
217                     //如果是修改，就再选中
218                     if (self.selectedFlash) {
219                         editor.getSelection()
220                             .selectElement(substitute);
221                     }
222                     editor.notifySelectionChange();
223                 });
224         },
225 
226         destroy:function () {
227             this.destroyRes();
228         }
229     });
230 
231     return FlashDialog;
232 }, {
233     requires:['editor', '../flashCommon/utils', '../overlay/','../select/']
234 });
