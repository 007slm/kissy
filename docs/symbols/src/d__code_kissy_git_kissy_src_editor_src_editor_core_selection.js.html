355                             // visual impact. ( #3231 )

358                                     startOffset = range.startOffset;

360                                 if (startOffset == ( startContainer[0].nodeType === DOM.ELEMENT_NODE ?
361                                     startContainer[0].childNodes.length : startContainer[0].nodeValue.length )

465             }

475         selectElement:function (element) {
476             var range,
477                 self = this,
478                 doc = self.document;
479             if (OLD_IE) {
480                 //do not use empty()，编辑器内滚动条重置了
481                 //选择的 img 内容前后莫名被清除
482                 //self.getNative().empty();
483                 try {
484                     // Try to select the node as a control.
485                     range = doc.body['createControlRange']();
486                     range['addElement'](element[0]);
487                     range.select();
488                 } catch (e) {
489                     // If failed, select it as a text range.
490                     range = doc.body.createTextRange();
491                     range.moveToElementText(element[0]);
492                     range.select();
493                 } finally {
494                     //this.document.fire('selectionchange');
495                 }
496                 self.reset();
497             } else {
498                 // Create the range for the element.
499                 range = doc.createRange();
500                 range.selectNode(element[0]);
501                 // Select the range.
502                 var sel = self.getNative();
503                 sel.removeAllRanges();
504                 sel.addRange(range);
505                 self.reset();
506             }

509         selectRanges:function (ranges) {
510             var self = this;
511             if (OLD_IE) {
512                 if (ranges.length > 1) {
513                     // IE doesn't accept multiple ranges selection, so we join all into one.
514                     var last = ranges[ ranges.length - 1 ];
515                     ranges[ 0 ].setEnd(last.endContainer, last.endOffset);
516                     ranges.length = 1;
517                 }
518 
519                 // IE doesn't accept multiple ranges selection, so we just
520                 // select the first one.
521                 if (ranges[ 0 ])
522                     ranges[ 0 ].select();
523 
524                 self.reset();
525             }
526             else {
527                 var sel = self.getNative();
528                 if (!sel) {
529                     return;
530                 }
531                 sel.removeAllRanges();
532                 for (var i = 0; i < ranges.length; i++) {
533                     var range = ranges[ i ],
534                         nativeRange = self.document.createRange(),
535                         startContainer = range.startContainer;
536 
537                     // In FF2, if we have a collapsed range, inside an empty
538                     // element, we must add something to it otherwise the caret
539                     // will not be visible.
540                     // opera move out of this element
541                     if (range.collapsed &&
542                         (( UA.gecko && UA.gecko < 1.0900 ) || UA.opera || UA['webkit']) &&
543                         startContainer[0].nodeType == DOM.ELEMENT_NODE &&
544                         !startContainer[0].childNodes.length) {
545                         // webkit 光标停留不到在空元素内，要fill char，之后范围定在 fillchar 之后
546                         startContainer[0].appendChild(self.document.createTextNode(UA['webkit'] ? "\u200b" : ""));
547                         range.startOffset++;
548                         range.endOffset++;
549                     }
550 
551                     nativeRange.setStart(startContainer[0], range.startOffset);
552                     nativeRange.setEnd(range.endContainer[0], range.endOffset);
553                     // Select the range.
554                     sel.addRange(nativeRange);
555                 }
556                 self.reset();
557             }

712                         var ieRangeEnd = self.document.body.createTextRange();
713                         // Position the tool range at the end.
714                         ieRangeEnd.moveToElementText(endNode[0]);
715                         // Move the end boundary of the main range to match the tool range.
716                         ieRange.setEndPoint('EndToEnd', ieRangeEnd);
717                         ieRange.moveEnd('character', -1);
718                     }
719                     else {
720                         // The isStartMarkerAlone logic comes from V2. It guarantees that the lines
721                         // will expand and that the cursor will be blinking on the right place.
722                         // Actually, we are using this flag just to avoid using this hack in all
723                         // situations, but just on those needed.
724                         var next = startNode[0].nextSibling;
725                         while (next && !notWhitespaces(next)) {
726                             next = next.nextSibling;
727                         }
728                         isStartMarkerAlone =
729                             (
730                                 !( next && next.nodeValue && next.nodeValue.match(fillerTextRegex) )     // already a filler there?
731                                     && ( forceExpand
732                                     ||
733                                     !startNode[0].previousSibling
734                                     ||
735                                     (
736                                         startNode[0].previousSibling &&
737                                             DOM.nodeName(startNode[0].previousSibling) == 'br'
738                                         )
739                                     )
740                                 );
741 
742                         // Append a temporary <span>﻿</span> before the selection.
743                         // This is needed to avoid IE destroying selections inside empty
744                         // inline elements, like <b></b> (#253).
745                         // It is also needed when placing the selection right after an inline
746                         // element to avoid the selection moving inside of it.
747                         dummySpan = new Node(self.document.createElement('span'));
748                         dummySpan.html('﻿');	// Zero Width No-Break Space (U+FEFF). See #1359.
749                         dummySpan.insertBefore(startNode);
750                         if (isStartMarkerAlone) {
751                             // To expand empty blocks or line spaces after <br>, we need
752                             // instead to have any char, which will be later deleted using the
753                             // selection.
754                             // \ufeff = Zero Width No-Break Space (U+FEFF). (#1359)
755                             DOM.insertBefore(self.document.createTextNode('\ufeff'), startNode[0] || startNode);
756                         }
757                     }
758 
759                     // Remove the markers (reset the position, because of the changes in the DOM tree).
760                     self.setStartBefore(startNode);
761                     startNode._4e_remove();
762 
763                     if (collapsed) {
764                         if (isStartMarkerAlone) {
765                             // Move the selection start to include the temporary \ufeff.
766                             ieRange.moveStart('character', -1);
767                             ieRange.select();
768                             // Remove our temporary stuff.
769                             self.document.selection.clear();
770                         } else
771                             ieRange.select();
772                         if (dummySpan) {
773                             self.moveToPosition(dummySpan, KER.POSITION_BEFORE_START);
774                             dummySpan._4e_remove();
775                         }
776                     }
777                     else {
778                         self.setEndBefore(endNode);
779                         endNode._4e_remove();
780                         ieRange.select();
781                     }
782                     // this.document.fire('selectionchange');
783                 };
