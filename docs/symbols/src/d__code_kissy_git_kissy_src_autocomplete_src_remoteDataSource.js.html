1 /**
  2  * remote datasource
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("autocomplete/remoteDataSource", function (S, IO) {
  6 
  7     /**
  8      * dataSource which wrap {@link IO} utility.
  9      * @class
 10      * @memberOf AutoComplete
 11      * @param {Object} xhrCfg IO configuration.same as {@link} IO
 12      * @param {Object} dataSourceCfg dataSource configs
 13      * @param {String} dataSourceCfg.paramName
 14      * Used as parameter name to send autoS=Complete input's value to server
 15      * @param {String} dataSourceCfg.cache Whether server response data is cached
 16      * @param {Function} dataSourceCfg.parse Serve as a parse function to parse server
 17      * response to return a valid array of data for autoComplete.
 18      */
 19     function RemoteDataSource(xhrCfg, dataSourceCfg) {
 20         var self = this;
 21         xhrCfg.data = xhrCfg.data || {};
 22         self.xhrCfg = xhrCfg;
 23         self.dataSourceCfg = dataSourceCfg || {};
 24         self.io = null;
 25         self.caches = {};
 26     }
 27 
 28 
 29     /**
 30      * Datasource interface. How to get data for autoComplete
 31      * @function
 32      * @name AutoComplete.RemoteDataSource#fetchData
 33      * @param {String} inputVal current active input's value
 34      * @param {Function} callback callback to notify autoComplete when data is ready
 35      * @param {Object} context callback's execution context
 36      */
 37     RemoteDataSource.prototype.fetchData = function (inputVal, callback, context) {
 38         var self = this,
 39             v,
 40             dataSourceCfg = self.dataSourceCfg;
 41         if (self.io) {
 42             // abort previous request
 43             self.io.abort();
 44             self.io = null;
 45         }
 46         if (!inputVal) {
 47             return callback.call(context, []);
 48         }
 49         if (dataSourceCfg.cache) {
 50             if (v = self.caches[inputVal]) {
 51                 return callback.call(context, v);
 52             }
 53         }
 54         var xhrCfg = S.clone(self.xhrCfg);
 55         xhrCfg.data[dataSourceCfg['paramName']] = inputVal;
 56         xhrCfg.success = function (data) {
 57             if (dataSourceCfg.parse) {
 58                 data = dataSourceCfg.parse(data);
 59             }
 60             if (dataSourceCfg.cache) {
 61                 self.caches[inputVal] = data;
 62             }
 63             callback.call(context, data);
 64         };
 65         self.io = IO(xhrCfg);
 66     };
 67 
 68     return RemoteDataSource;
 69 }, {
 70     requires:['ajax']
 71 });
