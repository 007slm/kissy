1 /**
  2  * @fileOverview This class specifies the definition for whow grid
  3  * @author dxq613@gmail.com, yiminghe@gmail.com
  4  */
  5 KISSY.add('grid/base', function (S, Component, Header, GridBody, Util) {
  6 
  7 
  8     var CLS_GRID_WITH = 'ks-grid-width',
  9         CLS_GRID_HEIGHT = 'ks-grid-height',
 10         CLS_GRID_TBAR = 'ks-grid-tbar',
 11         CLS_GRID_BBAR = 'ks-grid-bbar',
 12         HEIGHT_BAR_PADDING = 10;
 13 
 14     /**
 15      * This class specifies the definition for the grid which contains {@link Grid.Header},{@link Grid.GridBody}
 16      * @name Grid
 17      * @constructor
 18      * @extends Component.Controller
 19      * @extends Grid.Bindable
 20      */
 21     var grid = Component.Controller.extend({
 22 
 23         createDom:function () {
 24             var _self = this;
 25 
 26             // 提前,中途设置宽度时会失败！！
 27             if (_self.get("width")) {
 28                 _self.get("el").addClass(CLS_GRID_WITH);
 29             }
 30 
 31             if (_self.get("height")) {
 32                 _self.get("el").addClass(CLS_GRID_HEIGHT);
 33             }
 34 
 35             _self._initHeader();
 36             _self._initBody();
 37             _self._initBars();
 38             _self._initLoadMask();
 39         },
 40 
 41         /*
 42          * For overridden.
 43          * @protected
 44          * @override
 45          */
 46         renderUI:function () {
 47             var _self = this;
 48             S.each(['tbar','header', 'body','bbar'], function (c) {
 49                 if (c = _self.get(c)) {
 50                     c.render();
 51                 }
 52             });
 53         },
 54         /**
 55          * add a column to grid
 56          * @param {Object|Grid.Column} column a column config or an instance of {@link Grid.Column}
 57          * @return {Grid.Column}
 58          */
 59         addColumn:function (column, index) {
 60             var _self = this;
 61             column = _self.get('header').addColumn(column, index);
 62             return column;
 63         },
 64         /**
 65          * @private
 66          */
 67         bindUI:function () {
 68             var _self = this;
 69             _self._bindHeaderEvent();
 70             _self._bindBodyEvent();
 71         },
 72         /**
 73          * clear the selected status of all the rows
 74          */
 75         clearSelection:function () {
 76             var _self = this;
 77             _self.get('body').clearSelection();
 78         },
 79         /**
 80          * remove one column from this grid
 81          * @param {Grid.Column} column the removed column
 82          */
 83         removeColumn:function (column) {
 84             var _self = this;
 85             _self.get('header').removeColumn(column);
 86             //_self.get('body').resetColumns();
 87         },
 88         /**
 89          * set the rows' selected status
 90          * @param {Array|Object} records the records which will be selected
 91          */
 92         setSelection:function (records) {
 93             this.get('body').setSelection(records);
 94         },
 95         /**
 96          * set all rows selected
 97          */
 98         setAllSelection:function () {
 99             this.get('body').setAllSelection();
100         },
101         /**
102          * show data in this grid
103          * @param {Array} data show the given data in table
104          */
105         showData:function (data) {
106             this.get('body').showData(data);
107         },
108         //init header,if there is not a header property in config,instance it
109         _initHeader:function () {
110             var _self = this,
111                 header = _self.get('header');
112             if (!header) {
113                 header = new Header({
114                     columns:_self.get('columns'),
115                     tableCls:_self.get('tableCls'),
116                     forceFit:_self.get('forceFit'),
117                     width:_self.get('width'),
118                     render:_self.get("el"),
119                     parent:_self
120                 }).create();
121                 _self.set('header', header);
122             }
123         },
124         //init grid body
125         _initBody:function () {
126             var _self = this,
127                 body = _self.get('body');
128             if (!body) {
129                 var attrs = _self.getAttrs(),
130                     toBody = {},
131                     bodyConfig;
132 
133                 for (var name in attrs) {
134                     if (attrs.hasOwnProperty(name) && attrs[name].toBody) {
135                         toBody[name] = _self.get(name);
136                     }
137                 }
138                 bodyConfig = S.merge(toBody, _self.get('bodyConfig'));
139                 bodyConfig.render = _self.get("el");
140                 bodyConfig.parent=_self;
141                 body = new GridBody(bodyConfig).create();
142                 _self.set('body', body);
143             }
144         },
145         _initBars:function () {
146             var _self = this,
147                 bbar = _self.get('bbar'),
148                 tbar = _self.get('tbar');
149             _self._initBar(bbar, CLS_GRID_BBAR, 'bbar');
150             _self._initBar(tbar, CLS_GRID_TBAR, 'tbar');
151         },
152         //set bar's elCls to identify top bar or bottom bar
153         _initBar:function (bar, cls, name) {
154             var _self = this;
155             if (bar) {
156                 if (bar.xclass) {
157                     bar.render = _self.get("el");
158                     bar.parent=_self;
159                     bar = Component.create(bar).create();
160                 }
161                 bar.set('elCls', cls);
162                 _self.set(name, bar);
163             }
164             return bar;
165         },
166         //when set 'loadMask = true' ,create a loadMask instance
167         _initLoadMask:function () {
168             var _self = this,
169                 loadMask = _self.get('loadMask');
170             if (loadMask && !loadMask.show) {
171                 loadMask = new Util.LoadMask(_self.get('el'));
172                 _self.set('loadMask', loadMask);
173             }
174         },
175         //bind header event,when column changed,followed this component
176         _bindHeaderEvent:function () {
177             var _self = this,
178                 header = _self.get('header'),
179                 body = _self.get('body'),
180                 store = _self.get('store');
181             header.on('afterWidthChange', function (e) {
182                 var sender = e.target;
183                 if (sender !== header) {
184                     body.resetColumnsWidth(sender);
185                 }
186             });
187 
188             header.on('afterSortStateChange', function (e) {
189                 var column = e.target,
190                     val = e.newVal;
191                 if (val && store) {
192                     store.sort(column.get('dataIndex'), column.get('sortState'));
193                 }
194             });
195 
196 
197             header.on('afterVisibleChange', function (e) {
198                 var sender = e.target;
199                 if (sender !== header) {
200                     body.setColumnVisible(sender);
201                 }
202             });
203 
204             header.on('forceFitWidth', function () {
205                 if (_self.get('rendered')) {
206                     body.resetColumns();
207                 }
208             });
209 
210             header.on('add', function () {
211                 if (_self.get('rendered')) {
212                     body.resetColumns();
213                 }
214             });
215 
216             header.on('remove', function () {
217                 if (_self.get('rendered')) {
218                     body.resetColumns();
219                 }
220             });
221 
222         },
223         //when body scrolled, header can followed
224         _bindBodyEvent:function () {
225             var _self = this,
226                 body = _self.get('body'),
227                 header = _self.get('header');
228             body.on('scroll', function (event) {
229                 header.scrollTo({left:event.scrollLeft, top:event.scrollTop});
230             });
231         },
232         //when set grid's width, the width of its children also changed
233         _uiSetWidth:function (w) {
234             var _self = this;
235             _self.get('header').set('width', w);
236             _self.get('body').set('width', w);
237             _self.get("el").addClass(CLS_GRID_WITH);
238         },
239         //when set grid's height,the scroll can effect the width of its body and header
240         _uiSetHeight:function (h) {
241             var _self = this,
242                 bodyHeight = h,
243                 header = _self.get('header'),
244                 tbar = _self.get('tbar'),
245                 bbar = _self.get('bbar');
246             bodyHeight -= header.get('el').height();
247             if (tbar) {
248                 bodyHeight -= tbar.get('el').height() + HEIGHT_BAR_PADDING;
249             }
250             if (bbar) {
251                 bodyHeight -= bbar.get('el').height() + HEIGHT_BAR_PADDING;
252             }
253             /**/
254             _self.get('body').set('height', bodyHeight);
255             if (_self.get('rendered')) {
256                 if (_self.get('forceFit')) {
257                     header.forceFitColumns();
258                 }
259                 header.setTableWidth();
260             }
261             _self.get("el").addClass(CLS_GRID_HEIGHT);
262         },
263         _uiSetForceFit:function (v) {
264             var _self = this;
265             _self.get('header').set('forceFit', v);
266             _self.get('body').set('forceFit', v);
267         },
268         _uiSetMultiSelect:function (v) {
269             this.get('body').set('multiSelect', v);
270         }
271     }, {
272         ATTRS:{
273             /**
274              * the header of this grid
275              * @private
276              * @type {Grid.Header}
277              */
278             header:{
279 
280 
281             },
282             /**
283              * The table show data
284              * @private
285              * @type {Grid.GridBody}
286              */
287             body:{
288 
289             },
290             /**
291              * the config of the body of this component
292              */
293             bodyConfig:{
294                 value:{}
295             },
296             /**
297              *@private
298              */
299             checkable:{
300                 value:false
301             },
302             /**
303              * columns of this grid,use to initial header and body
304              * @see Grid.Column
305              * @private
306              */
307             columns:{
308                 toBody:true,
309                 value:[]
310             },
311             /**
312              * true to force the columns to fit into the available width.
313              * Headers are first sized according to configuration, whether that be a specific width, or flex.
314              * Then they are all proportionally changed in width so that the entire content width is used.
315              * @type {Boolean}
316              * @default 'false'
317              */
318             forceFit:{
319                 toBody:true,
320                 value:false
321             },
322             height:{
323             },
324             /**
325              * The CSS class to apply to this header's table and body's table elements.
326              * @type {String}
327              * @default 'ks-grid-table' this css cannot be overridden
328              */
329             tableCls:{
330                 toBody:true,
331                 value:''
332             },
333             /**
334              * Does it allow select multiple  rows
335              * @type {Boolean}
336              * @default false
337              */
338             multiSelect:{
339                 toBody:true,
340                 value:false
341             },
342             /**
343              * True to stripe the rows.
344              * @type {Boolean}
345              * @default true
346              */
347             stripeRows:{
348                 toBody:true,
349                 value:true
350             },
351             store:{
352                 toBody:true
353             },
354             loadMask:{
355                 toBody:true
356             },
357             /**
358              * @override
359              * when set this grid's width ,the header and body changed
360              */
361             width:{
362                 toBody:true
363             },
364             /**
365              * the collection of body's events
366              * @type {Array}
367              */
368             events:{
369                 value:[
370                 /**
371                  * after show a collection data in this component
372                  * @name Grid#aftershow
373                  * @event
374                  */
375                     'aftershow'    ,
376                 /**
377                  * fired when click one cell of row
378                  * @name Grid#cellclick
379                  * @event
380                  * @param {event} e  event object
381                  * @param {Object} e.record the record showed by this row
382                  * @param {String} e.field the dataIndex of the column which this cell belong to
383                  * @param {HTMLElement} e.row the dom element of this row
384                  * @param {HTMLElement} e.cell the dom element of this cell
385                  * @param {HTMLElement} e.domTarget the dom element of the click target
386                  */
387                     'cellclick',
388                 /**
389                  * fired when click one row
390                  * @name Grid#rowclick
391                  * @event
392                  * @param {event} e  event object
393                  * @param {Object} e.record the record showed by this row
394                  * @param {HTMLElement} e.row the dom element of this row
395                  */
396                     'rowclick',
397                 /**
398                  * add a row in this component.in general,this event fired after adding a record to the store
399                  * @name Grid#rowcreated
400                  * @event
401                  * @param {event} e  event object
402                  * @param {Object} e.record the record adding to the store
403                  * @param {HTMLElement} e.row the dom element of this row
404                  */
405                     'rowcreated',
406                 /**
407                  * remove a row from this component.in general,this event fired after delete a record from the store
408                  * @name Grid#rowremoved
409                  * @event
410                  * @param {event} e  event object
411                  * @param {Object} e.record the record removed from the store
412                  * @param {HTMLElement} e.row the dom element of this row
413                  */
414                     'rowremoved',
415                 /**
416                  * when click the row,in multiple select model the selected status toggled
417                  * @name Grid#rowselected
418                  * @event
419                  * @param {event} e  event object
420                  * @param {Object} e.record the record showed by this row
421                  * @param {HTMLElement} e.row the dom element of this row
422                  */
423                     'rowselected',
424                 /**
425                  * fire after cancel selected status
426                  * @name Grid#rowunselected
427                  * @event
428                  * @param {event} e  event object
429                  * @param {Object} e.record the record showed by this row
430                  * @param {HTMLElement} e.row the dom element of this row
431                  */
432                     'rowunselected',
433                 /**
434                  * remove a row from this component.in general,this event fired after delete a record from the store
435                  * @name Grid#scroll
436                  * @event
437                  * @param {event} e  event object
438                  * @param {Number} e.scrollLeft the horizontal value that the body scroll to
439                  * @param {Number} e.scrollTop the vertical value that the body scroll to
440                  */
441                     'scroll'
442                 ]
443             }
444         }
445     }, {
446         xclass:'grid',
447         priority:1
448     });
449     return grid;
450 }, {
451     requires:['component', './header', './gridbody', './util']
452 });
