167     function findAWithImg(img) {
168         var ret = img.parent();
169         while (ret) {
170             var name = ret.nodeName();
171             if (name == "a") {
172                 return ret;
173             }
174             if (dtd.$block[name] || dtd.$blockLimit[name]) {
175                 return null;
176             }
177             ret = ret.parent();
178         }
179         return null;
180     }

414         _insert:function () {
415             var self = this,
416                 url = valInput(self.imgUrl),
417                 img,
418                 height = parseInt(valInput(self.imgHeight)),

506         _update:function (_selectedEl) {
507             var self = this,
508                 active = 0,
509                 resetInput = Editor.Utils.resetInput;

511             if (self.selectedEl && self.imageCfg['remote'] !== false) {
512                 valInput(self.imgUrl, self.selectedEl.attr("src"));
513                 var w = self.selectedEl.width(),

515                 valInput(self.imgHeight, h);
516                 valInput(self.imgWidth, w);
517                 self.imgAlign.val(self.selectedEl.style("float") || "none");
518                 var margin = parseInt(self.selectedEl.style("margin"))
519                     || 0;
520                 self.imgMargin.val(margin);
521                 self.imgRatio[0].disabled = false;
522                 self.imgRatioValue = w / h;
523                 var link = findAWithImg(self.selectedEl);
524                 if (link) {
525                     valInput(self.imgLink, link.attr("_ke_saved_href") || link.attr("href"));
526                     self.imgLinkBlank.attr("checked", link.attr("target") == "_blank");
527                 } else {
528                     resetInput(self.imgLink);
529                     self.imgLinkBlank.attr("checked", true);
530                 }
531             } else {
532                 var defaultMargin = self.imageCfg['defaultMargin'];
533                 if (defaultMargin == undefined) {
534                     defaultMargin = MARGIN_DEFAULT;
535                 }
536                 if (self.tab['panels'].length == 2) {
537                     active = 1;
538                 }
539                 self.imgLinkBlank.attr("checked", true);
540                 resetInput(self.imgUrl);
541                 resetInput(self.imgLink);
542                 resetInput(self.imgHeight);
543                 resetInput(self.imgWidth);
544                 self.imgAlign.val("none");
545                 self.imgMargin.val(defaultMargin);
546                 self.imgRatio[0].disabled = true;
547                 self.imgRatioValue = null;
548             }
549             self.uploadForm[0].reset();
550             self.imgLocalUrl.val(warning);
551             self.tab['switchTo'](active);
552         },

558         },

562         }
