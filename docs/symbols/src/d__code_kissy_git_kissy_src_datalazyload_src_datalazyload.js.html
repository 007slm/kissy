 20     function isValidContainer(c) {
 21         return c.nodeType != 9;
 22     }
 23 
 24     function getContainer(elem, cs) {
 25         for (var i = 0; i < cs.length; i++) {
 26             if (isValidContainer(cs[i])) {
 27                 if (cs[i].contains(elem)) {
 28                     return cs[i];
 29                 }
 30             }
 31         }
 32         return 0;
 33     }

 35     /**
 36      * 加载图片 src
 37      * @static
 38      */
 39     function loadImgSrc(img, flag) {
 40         flag = flag || IMG_SRC_DATA;
 41         var dataSrc = img.getAttribute(flag);
 42 
 43         if (dataSrc && img.src != dataSrc) {
 44             img.src = dataSrc;
 45             img.removeAttribute(flag);
 46         }
 47     }

 49     function removeExisting(newed, exists) {
 50         var ret = [];
 51         for (var i = 0; i < newed.length; i++) {
 52             if (!S.inArray(newed[i], exists)) {
 53                 ret.push(newed[i]);
 54             }
 55         }
 56         return ret;
 57     }

 59     /**
 60      * 从 textarea 中加载数据
 61      * @static
 62      */
 63     function loadAreaData(area, execScript) {
 64         // 采用隐藏 textarea 但不去除方式，去除会引发 Chrome 下错乱
 65         area.style.display = NONE;
 66         area.className = ''; // clear hook
 67         var content = DOM.create('<div>');
 68         // area 直接是 container 的儿子
 69         area.parentNode.insertBefore(content, area);
 70         DOM.html(content, area.value, execScript);
 71     }

 77         return DOM.hasClass(area, AREA_DATA_CLS);

 88

 91             return new DataLazyload(containers, config);

199     // 两块区域是否相交
200     function isCross(r1, r2) {
201         var r = {};
202         r.top = Math.max(r1.top, r2.top);
203         r.bottom = Math.min(r1.bottom, r2.bottom);
204         r.left = Math.max(r1.left, r2.left);
205         r.right = Math.min(r1.right, r2.right);
206         return r.bottom >= r.top && r.right >= r.left;
207     }

225             /**
226              * 获取并初始化需要延迟的 _images 和 _areaes
227              */
228             _filterItems:function () {
229                 var self = this,
230                     containers = self.get("containers"),
231                     n, N, imgs, _areaes, i, img,
232                     lazyImgs = [], lazyAreas = [];
233 
234                 for (n = 0, N = containers.length; n < N; ++n) {
235                     imgs = removeExisting(DOM.query('img', containers[n]), lazyImgs);
236                     lazyImgs = lazyImgs.concat(S.filter(imgs, self._filterImg, self));
237 
238                     _areaes = removeExisting(DOM.query('textarea', containers[n]), lazyAreas);
239                     lazyAreas = lazyAreas.concat(S.filter(_areaes, filterArea, self));
240                 }
241 
242                 self._images = lazyImgs;
243                 self._areaes = lazyAreas;
244             },

246             /**
247              * filter for lazyload image
248              */
249             _filterImg:function (img) {
250                 var self = this,
251                     dataSrc = img.getAttribute(IMG_SRC_DATA),
252                     placeholder = self.get("placeholder"),
253                     isManualMod = self.get("mod") === MANUAL;
254 
255                 // 手工模式，只处理有 data-src 的图片
256                 if (isManualMod) {
257                     if (dataSrc) {
258                         if (placeholder !== NONE) {
259                             img.src = placeholder;
260                         }
261                         return true;
262                     }
263                 }
264                 // 自动模式，只处理 threshold 外无 data-src 的图片
265                 else {
266                     // 注意：已有 data-src 的项，可能已有其它实例处理过，不用再次处理
267                     if (!dataSrc && !self._checkElemInViewport(img)) {
268                         DOM.attr(img, IMG_SRC_DATA, img.src);
269                         if (placeholder !== NONE) {
270                             img.src = placeholder;
271                         } else {
272                             img.removeAttribute('src');
273                         }
274                         return true;
275                     }
276                 }

280             /**
281              * 初始化加载事件
282              */
283             _initLoadEvent:function () {
284                 var self = this,
285                     autoDestroy = self.get("autoDestroy"),
286                     // 加载延迟项

303                     if (isValidContainer(c)) {
304                         Event.on(c, SCROLL, load);

306                     }
307                 });

310 
311                 // 需要立即加载一次，以保证第一屏的延迟项可见
312                 if (self._getItemsLength()) {
313                     S.ready(loadItems);
314                 }

368             /**
369              * 触发回调
370              */
371             _fireCallbacks:function () {
372                 var self = this,
373                     callbacks = self._callbacks,
374                     els = callbacks.els,
375                     fns = callbacks.fns,
376                     i, el, fn, remainEls = [], remainFns = [];
377

381                     } else {

391             /**
392              * Register callback function.
393              * When el is in viewport, then fn is called.
394              * @param {HTMLElement|String} el Html element to be monitored.
395              * @param {Function} fn Callback function to be called when el is in viewport.

398                 var self = this,
399                     callbacks = self._callbacks;
400                 el = DOM.get(el);
401 
402                 if (el && S.isFunction(fn)) {
403                     callbacks.els.push(el);
404                     callbacks.fns.push(fn);
405                 }
406 
407                 // add 立即检测，防止首屏元素问题
408                 self._fireCallbacks();

411             /**
412              * Remove a callback function. See {@link DataLazyload#addCallback}
413              * @param {HTMLElement|String} el Html element to be monitored.
414              * @param {Function} [fn] Callback function to be called when el is in viewport.
415              *                        If not specified, remove all callbacks associated with el.
416              * @since 1.3
417              */
418             removeCallback:function (el, fn) {
419                 var callbacks = this._callbacks,
420                     els = [],
421                     fns = [],
422                     curFns = callbacks.fns;
423 
424                 el = DOM.get(el);
425 
426                 S.each(callbacks.els, function (curEl, index) {
427                     if (curEl == el) {
428                         if (fn === undefined || fn == curFns[index]) {
429                             return;
430                         }
431                     }
432 
433                     els.push(curEl);
434                     fns.push(curFns[index]);
435                 });
436 
437                 callbacks.fns = fns;
438                 callbacks.els = els;
439             },

441             /**
442              * Add a array of imgs or textareas to be lazy loaded to monitor list.
443              * @param {HTMLElement[]} els Array of imgs or textareas to be lazy loaded
444              * @since 1.3
445              */
446             addElements:function (els) {
447                 if (!S.isArray(els)) {
448                     els = [els];
449                 }
450                 var self = this,
451                     imgs = self._images || [],
452                     areaes = self._areaes || [];

457                             imgs.push(el);

459                     } else if (nodeName == "textarea") {
460                         if (!S.inArray(el, areaes)) {
461                             areaes.push(el);
462                         }
463                     }
464                 });
465                 self._images = imgs;
466                 self._areaes = areaes;

494             /**
495              * 获取 c 的有效渲染区域（加上预加载差值）
496              * @protected
497              */
498             _getBoundingRect:function (c) {
499                 var vh, vw, left, top;
500 
501                 if (c !== undefined &&
502                     !S.isWindow(c) &&
503                     c.nodeType != 9) {
504                     vh = DOM.outerHeight(c);
505                     vw = DOM.outerWidth(c);
506                     var offset = DOM.offset(c);
507                     left = offset.left;
508                     top = offset.top;
509                 } else {
510                     vh = DOM.viewportHeight();
511                     vw = DOM.viewportWidth();
512                     left = DOM.scrollLeft();
513                     top = DOM.scrollTop();
514                 }
515 
516                 var diff = this.get("diff"),
517                     diffX = diff === DEFAULT ? vw : diff,
518                     diffX0 = 0,
519                     diffX1 = diffX,
520                     diffY = diff === DEFAULT ? vh : diff,
521                     // 兼容，默认只向下预读
522                     diffY0 = 0,
523                     diffY1 = diffY,
524                     right = left + vw,
525                     bottom = top + vh;
526 
527                 if (S.isObject(diff)) {
528                     diffX0 = diff.left || 0;
529                     diffX1 = diff.right || 0;
530                     diffY0 = diff.top || 0;
531                     diffY1 = diff.bottom || 0;
532                 }
533 
534                 left -= diffX0;
535                 right += diffX1;
536                 top -= diffY0;
537                 bottom += diffY1;
538 
539                 return {
540                     left:left,
541                     top:top,
542                     right:right,
543                     bottom:bottom
544                 };
545             },

556             /**
557              * 判断 textarea 元素是否一部分在可视区域内（容器内并且在窗口 viewport 内）
558              * @private
559              * @param elem
560              */
561             _checkElemInViewport:function (elem) {
562                 // 注：elem 可能处于 display: none 状态，DOM.offset(elem).top 返回 0
563                 // 这种情况下用 elem.parentNode 的 Y 值来替代
564                 elem = DOM.css(elem, DISPLAY) === NONE ? elem.parentNode : elem;
565 
566                 var self = this,
567                     elemOffset = DOM.offset(elem),
568                     inContainer = true,
569                     container = getContainer(elem, self.get("containers")),
570                     windowRegion = self._getBoundingRect(),
571                     inWin,
572                     containerRegion,
573                     left = elemOffset.left,
574                     top = elemOffset.top,
575                     elemRegion = {
576                         left:left,
577                         top:top,
578                         right:left + DOM.outerWidth(elem),
579                         bottom:top + DOM.outerHeight(elem)
580                     };
581 
582                 if (container) {
583                     containerRegion = self._getBoundingRect(container);
584                     inContainer = isCross(containerRegion, elemRegion);
585                 }
586 
587                 // 确保在容器内出现
588                 // 并且在视窗内也出现
589                 inWin = isCross(windowRegion, elemRegion);
590                 return inContainer && inWin;
591             },

616     /**
617      * Load lazyload textarea and imgs manually.
618      * @name loadCustomLazyData
619      * @function
620      * @memberOf DataLazyload
621      * @param {HTMLElement[]} containers Containers with in which lazy loaded elements are loaded.
622      * @param {String} type Type of lazy loaded element. "img" or "textarea"
623      * @param {String} [flag] flag which will be searched to find lazy loaded elements from containers.
624      * Default "data-ks-lazyload-custom" for img attribute and "ks-lazyload-custom" for textarea css class.
625      */
626     function loadCustomLazyData(containers, type, flag) {
627         var imgs;
628 
629         if (type === 'img-src') {
630             type = 'img';
631         }
632 
633         // 支持数组
634         if (!S.isArray(containers)) {
635             containers = [DOM.get(containers)];
636         }
637 
638         // 遍历处理
639         S.each(containers, function (container) {
640             switch (type) {
641                 case 'img':
642                     if (container.nodeName === 'IMG') { // 本身就是图片
643                         imgs = [container];
644                     } else {
645                         imgs = DOM.query('img', container);
646                     }
647 
648                     S.each(imgs, function (img) {
649                         loadImgSrc(img, flag || (IMG_SRC_DATA + CUSTOM));
650                     });
651                     break;
652 
653                 default:
654                     DOM.query('textarea', container).each(function (area) {
655                         if (DOM.hasClass(area, flag || (AREA_DATA_CLS + CUSTOM))) {
656                             loadAreaData(area, true);
657                         }
658                     });
659             }
660         });
661     }

672 /**
673  * NOTES:
674  *
675  * 模式为 auto 时：
676  *  1. 在 Firefox 下非常完美。脚本运行时，还没有任何图片开始下载，能真正做到延迟加载。
677  *  2. 在 IE 下不尽完美。脚本运行时，有部分图片已经与服务器建立链接，这部分 abort 掉，
678  *     再在滚动时延迟加载，反而增加了链接数。
679  *  3. 在 Safari 和 Chrome 下，因为 webkit 内核 bug，导致无法 abort 掉下载。该
680  *     脚本完全无用。
681  *  4. 在 Opera 下，和 Firefox 一致，完美。
682  *  5. 2010-07-12: 发现在 Firefox 下，也有导致部分 Aborted 链接。
683  *
684  * 模式为 manual 时：（要延迟加载的图片，src 属性替换为 data-lazyload-src, 并将 src 的值赋为 placeholder ）
685  *  1. 在任何浏览器下都可以完美实现。
686  *  2. 缺点是不渐进增强，无 JS 时，图片不能展示。
687  *
688  * 缺点：
689  *  1. 对于大部分情况下，需要拖动查看内容的页面（比如搜索结果页），快速滚动时加载有损用
690  *     户体验（用户期望所滚即所得），特别是网速不好时。
691  *  2. auto 模式不支持 Webkit 内核浏览器；IE 下，有可能导致 HTTP 链接数的增加。
692  *
693  * 优点：
694  *  1. 可以很好的提高页面初始加载速度。
695  *  2. 第一屏就跳转，延迟加载图片可以减少流量。
696  *
697  * 参考资料：
698  *  1. http://davidwalsh.name/lazyload MooTools 的图片延迟插件
699  *  2. http://vip.qq.com/ 模板输出时，就替换掉图片的 src
700  *  3. http://www.appelsiini.net/projects/lazyload jQuery Lazyload
701  *  4. http://www.dynamixlabs.com/2008/01/17/a-quick-look-add-a-loading-icon-to-your-larger-_images/
702  *  5. http://www.nczonline.net/blog/2009/11/30/empty-image-src-can-destroy-your-site/
703  *
704  * 特别要注意的测试用例:
705  *  1. 初始窗口很小，拉大窗口时，图片加载正常
706  *  2. 页面有滚动位置时，刷新页面，图片加载正常
707  *  3. 手动模式，第一屏有延迟图片时，加载正常
708  *
709  * 2009-12-17 补充：
710  *  1. textarea 延迟加载约定：页面中需要延迟的 dom 节点，放在
711  *       <textarea class='ks-datalazysrc invisible'>dom code</textarea>
