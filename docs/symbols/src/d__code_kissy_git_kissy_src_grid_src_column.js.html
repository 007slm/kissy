 74                 method = v ? 'addClass' : 'removeClass',
 75                 ascCls = SORT_PREFIX + 'asc',
 76                 desCls = SORT_PREFIX + 'desc';
 77             el.removeClass(ascCls + ' ' + desCls);
 78             if (v === 'ASC') {
 79                 el.addClass(ascCls);

 81                 el.addClass(desCls);

 99     /**
100      * This class specifies the definition for a column inside a {@link Grid}.
101      * It encompasses both the grid header configuration as well as displaying data within the grid itself.
102      * If the columns configuration is specified, this column will become a column group and can contain other columns inside.
103      * In general, this class will not be created directly, rather an array of column configurations will be passed to the grid
104      * @name Grid.Column
105      * @constructor
106      * @extends Component.Controller
107      */
108     var column = Component.Controller.extend(
109         /**
110          * @lends Grid.Column.prototype
111          */
112         {    //toggle sort state of this column ,if no sort state set 'ASC',else toggle 'ASC' and 'DESC'
113             _toggleSortState:function () {
114                 var _self = this,
115                     sortState = _self.get('sortState'),
116                     v = sortState ? (sortState === SORT_ASC ? SORT_DESC : SORT_ASC) : SORT_ASC;
117                 _self.set('sortState', v);
118             },
119             /**
120              * @see {Component.Controller#bindUI}
121              */
122             bindUI:function () {
123                 var _self = this,
124                     events = _self.get('events');
125                 S.each(events, function (event) {
126                     _self.publish(event, {
127                         bubbles:1
128                     });
129                 });
130             },
131             /**
132              * {Component.Controller#performActionInternal}
133              * @private
134              */
135             performActionInternal:function (ev) {
136                 var _self = this,
137                     sender = S.one(ev.target),
138                     prefix = _self.get('prefixCls');
139                 if (sender.hasClass(prefix + CLS_HD_TRIGGER)) {
140 
141                 } else {
142                     if (_self.get('sortable')) {
143                         _self._toggleSortState();
144                     }
145                 }
146                 _self.fire('click');
147             }
