  5 KISSY.add("ajax/base", function (S, JSON, Event, XhrObject, undefined) {

  7     var rlocalProtocol = /^(?:about|app|app\-storage|.+\-extension|file|widget)$/,
  8         rspace = /\s+/,
  9         mirror = function (s) {
 10             return s;
 11         },
 12         rnoContent = /^(?:GET|HEAD)$/,
 13         curLocation,
 14         Uri = S.Uri,
 15         win = S.Env.host,
 16         doc = win.document,
 17         location = win.location,
 18         simulatedLocation;
 19 
 20     try {
 21         curLocation = location.href;
 22     } catch (e) {
 23         S.log("ajax/base get curLocation error : ");
 24         S.log(e);
 25         // Use the href attribute of an A element
 26         // since IE will modify it given document.location
 27         curLocation = doc.createElement("a");
 28         curLocation.href = "";
 29         curLocation = curLocation.href;
 30     }
 31 
 32     simulatedLocation = new Uri(curLocation);
 33 
 34     var isLocal = rlocalProtocol.test(simulatedLocation.getScheme()),
 35         transports = {},
 36         defaultConfig = {
 37             type:"GET",
 38             contentType:"application/x-www-form-urlencoded; charset=UTF-8",
 39             async:true,
 40             serializeArray:true,
 41             processData:true,
 42             accepts:{
 43                 xml:"application/xml, text/xml",
 44                 html:"text/html",
 45                 text:"text/plain",
 46                 json:"application/json, text/javascript",
 47                 *:"*/*"
 48             },
 49             converters:{
 50                 text:{
 51                     json:JSON.parse,
 52                     html:mirror,
 53                     text:mirror,
 54                     xml:S.parseXML
 55                 }
 56             },
 57             contents:{
 58                 xml:/xml/,
 59                 html:/html/,
 60                 json:/json/
 61             }
 62         };
 63 
 64     defaultConfig.converters.html = defaultConfig.converters.text;
 65 
 66     function setUpConfig(c) {
 67         // deep mix,exclude context!
 68 
 69         var context= c.context;
 70         delete c.context;
 71         c = S.mix(S.clone(defaultConfig), c, {
 72             deep:true
 73         });
 74         c.context=context||c;
 75 
 76         var data, uri, type = c.type, dataType = c.dataType, query;
 77 
 78         query = c.query = new S.Uri.Query();
 79 
 80         uri = c.uri = simulatedLocation.resolve(c.url);
 81 
 82         if (!("crossDomain" in c)) {
 83             c.crossDomain = !c.uri.hasSameDomainAs(simulatedLocation);
 84         }
 85 
 86         if (c.processData && (data = c.data)) {
 87             // 必须 encodeURIComponent 编码 utf-8
 88             if (S.isObject(data)) {
 89                 query.add(data);
 90             } else {
 91                 query.reset(data);
 92             }
 93         }
 94 
 95         type = c.type = type.toUpperCase();
 96         c.hasContent = !rnoContent.test(type);
 97 
 98         // 数据类型处理链，一步步将前面的数据类型转化成最后一个
 99         dataType = c.dataType = S.trim(dataType || "*").split(rspace);
100 
101         if (!("cache" in c) && S.inArray(dataType[0], ["script", "jsonp"])) {
102             c.cache = false;
103         }
104 
105         if (!c.hasContent) {
106             if (query.count()) {
107                 uri.query.add(query);
108             }
109             if (c.cache === false) {
110                 uri.query.set("_ksTS", (S.now() + "_" + S.guid()));
111             }
112         }
113         return c;
114     }

127          * @name IO#success
128          * @description  fired after request succeeds
129          * @event
130          * @param {Event.Object} e
131          * @param {Object} e.ajaxConfig current request 's config
132          * @param {IO.XhrObject} e.xhr current xhr object
133          */
134 
135         /**
136          * @name IO#error
137          * @description fired after request occurs error
138          * @event
139          * @param {Event.Object} e
140          * @param {Object} e.ajaxConfig current request 's config

143         io.fire(eventType, { ajaxConfig:xhrObject.config, xhr:xhrObject});
144     }

146     /**
147      * @name IO
148      * @namespace Provides utility that brokers HTTP requests through a simplified interface
149      * @function
150      *
151      * @param {Object} c <br/>name-value of object to config this io request.<br/>
152      *  all values are optional.<br/>
153      *  default value can be set through {@link io.setupConfig}<br/>
154      *
155      * @param {String} c.url <br/>request destination
156      *
157      * @param {String} c.type <br/>request type.
158      * eg: "get","post"<br/>
159      * Default: "get"<br/>
160      *
161      * @param {String} c.contentType <br/>
162      * Default: "application/x-www-form-urlencoded; charset=UTF-8"<br/>
163      * Data will always be transmitted to the server using UTF-8 charset<br/>
164      *
165      * @param {Object} c.accepts <br/>
166      * Default: depends on DataType.<br/>
167      * The content type sent in request header that tells the server<br/>
168      * what kind of response it will accept in return.<br/>
169      * It is recommended to do so once in the {@link io.setupConfig}
170      *
171      * @param {Boolean} c.async <br/>
172      * Default: true<br/>
173      * whether request is sent asynchronously<br/>
174      *
175      * @param {Boolean} c.cache <br/>
176      * Default: true ,false for dataType "script" and "jsonp"<br/>
177      * if set false,will append _ksTs=Date.now() to url automatically<br/>
178      *
179      * @param {Object} c.contents <br/>
180      * a name-regexp map to determine request data's dataType<br/>
181      * It is recommended to do so once in the {@link io.setupConfig}<br/>
182      *
183      * @param {Object} c.context <br/>
184      * specify the context of this request's callback (success,error,complete)
185      *
186      * @param {Object} c.converters <br/>
187      * Default:{text:{json:JSON.parse,html:mirror,text:mirror,xml:KISSY.parseXML}}<br/>
188      * specified how to transform one dataType to another dataType<br/>
189      * It is recommended to do so once in the {@link io.setupConfig}
190      *
191      * @param {Boolean} c.crossDomain <br/>
192      * Default: false for same-domain request,true for cross-domain request<br/>
193      * if server-side jsonp redirect to another domain ,you should set this to true
194      *
195      * @param {Object} c.data <br/>
196      * Data sent to server.if processData is true,data will be serialized to String type.<br/>
197      * if value if an Array, serialization will be based on serializeArray.
198      *
199      * @param {String} c.dataType <br/>
200      * return data as a specified type<br/>
201      * Default: Based on server contentType header<br/>
202      * "xml" : a XML document<br/>
203      * "text"/"html": raw server data <br/>
204      * "script": evaluate the return data as script<br/>
205      * "json": parse the return data as json and return the result as final data<br/>
206      * "jsonp": load json data via jsonp
207      *
208      * @param {Object} c.headers <br/>
209      * additional name-value header to send along with this request.
210      *
211      * @param {String} c.jsonp <br/>
212      * Default: "callback"<br/>
213      * Override the callback function name in a jsonp request. eg:<br/>
214      * set "callback2" , then jsonp url will append  "callback2=?".
215      *
216      * @param {String} c.jsonpCallback <br/>
217      * Specify the callback function name for a jsonp request.<br/>
218      * set this value will replace the auto generated function name.<br/>
219      * eg:<br/>
220      * set "customCall" , then jsonp url will append "callback=customCall"
221      *
222      * @param {String} c.mimeType <br/>
223      * override xhr's mime type
224      *
225      * @param {Boolean} c.processData <br/>
226      * Default: true<br/>
227      * whether data will be serialized as String
228      *
229      * @param {String} c.scriptCharset <br/>
230      * only for dataType "jsonp" and "script" and "get" type.<br/>
231      * force the script to certain charset.
232      *
233      * @param {Function} c.beforeSend <br/>
234      * beforeSend(xhrObject,config)<br/>
235      * callback function called before the request is sent.this function has 2 arguments<br/>
236      * 1. current KISSY xhrObject<br/>
237      * 2. current io config<br/>
238      * note: can be used for add progress event listener for native xhr's upload attribute
239      * see <a href="http://www.w3.org/TR/XMLHttpRequest/#event-xhr-progress">XMLHttpRequest2</a>
240      *
241      * @param {Function} c.success <br/>
242      * success(data,textStatus,xhr)<br/>
243      * callback function called if the request succeeds.this function has 3 arguments<br/>
244      * 1. data returned from this request with type specified by dataType<br/>
245      * 2. status of this request with type String<br/>
246      * 3. XhrObject of this request , for details {@link IO.XhrObject}
247      *
248      * @param {Function} c.error <br/>
249      * success(data,textStatus,xhr) <br/>
250      * callback function called if the request occurs error.this function has 3 arguments<br/>
251      * 1. null value<br/>
252      * 2. status of this request with type String,such as "timeout","Not Found","parsererror:..."<br/>
253      * 3. XhrObject of this request , for details {@link IO.XhrObject}
254      *
255      * @param {Function} c.complete <br/>
256      * success(data,textStatus,xhr)<br/>
257      * callback function called if the request finished(success or error).this function has 3 arguments<br/>
258      * 1. null value if error occurs or data returned from server<br/>
259      * 2. status of this request with type String,such as success:"ok",
260      * error:"timeout","Not Found","parsererror:..."<br/>
261      * 3. XhrObject of this request , for details {@link IO.XhrObject}
262      *
263      * @param {Number} c.timeout <br/>
264      * Set a timeout(in seconds) for this request.if will call error when timeout
265      *
266      * @param {Boolean} c.serializeArray <br/>
267      * whether add [] to data's name when data's value is array in serialization
268      *
269      * @param {Object} c.xhrFields <br/>
270      * name-value to set to native xhr.set as xhrFields:{withCredentials:true}
271      *
272      * @param {String} c.username <br/>
273      * a username tobe used in response to HTTP access authentication request
274      *
275      * @param {String} c.password <br/>
276      * a password tobe used in response to HTTP access authentication request
277      *
278      * @param {Object} c.xdr <br/>
279      * cross domain request config object
280      *
281      * @param {String} c.xdr.src <br/>
282      * Default: KISSY 's flash url
283      * flash sender url
284      *
285      * @param {String} c.xdr.use <br/>
286      * if set to "use", it will always use flash for cross domain request even in chrome/firefox
287      *
288      * @param {Object} c.xdr.subDomain <br/>
289      * cross sub domain request config object
290      *
291      * @param {String} c.xdr.subDomain.proxy <br/>
292      * proxy page,eg:<br/>
293      * a.t.cn/a.htm send request to b.t.cn/b.htm: <br/>
294      * 1. a.htm set document.domain='t.cn'<br/>
295      * 2. b.t.cn/proxy.htm 's content is <script>document.domain='t.cn'</script><br/>
296      * 3. in a.htm , call io({xdr:{subDomain:{proxy:'/proxy.htm'}}})
297      *
298      * @returns {IO.XhrObject} current request object
299      */
300     function io(c) {
301 
302         if (!c.url) {
303             return undefined;
304         }
305 
306         c = setUpConfig(c);
307 
308         var xhrObject = new XhrObject(c),
309             transportConstructor,
310             transport;
311 
312         /**
313          * @name IO#start
314          * @description fired before generating request object
315          * @event
316          * @param {Event.Object} e
317          * @param {Object} e.ajaxConfig current request 's config
318          * @param {IO.XhrObject} e.xhr current xhr object
319          */
320 
321         fire("start", xhrObject);
322 
323         transportConstructor = transports[c.dataType[0]] || transports["*"];
324         transport = new transportConstructor(xhrObject);
325 
326         xhrObject.transport = transport;
327 
328         if (c.contentType) {
329             xhrObject.setRequestHeader("Content-Type", c.contentType);
330         }
331 
332         var dataType = c.dataType[0],
333             timeoutTimer,
334             i,
335             timeout = c.timeout,
336             context = c.context,
337             headers = c.headers,
338             accepts = c.accepts;
339 
340         // Set the Accepts header for the server, depending on the dataType
341         xhrObject.setRequestHeader(
342             "Accept",
343             dataType && accepts[dataType] ?
344                 accepts[ dataType ] + (dataType === "*" ? "" : ", */*; q=0.01"  ) :
345                 accepts[ "*" ]
346         );
347 
348         // Check for headers option
349         for (i in headers) {
350             if (headers.hasOwnProperty(i)) {
351                 xhrObject.setRequestHeader(i, headers[ i ]);
352             }
353         }
354 
355 
356         // allow setup native listener
357         // such as xhr.upload.addEventListener('progress', function (ev) {})
358         if (c.beforeSend && ( c.beforeSend.call(context, xhrObject, c) === false)) {
359             return undefined;
360         }
361 
362         function genHandler(handleStr) {
363             return function (v) {
364                 if (timeoutTimer = xhrObject.timeoutTimer) {
365                     clearTimeout(timeoutTimer);
366                     xhrObject.timeoutTimer = 0;
367                 }
368                 var h = c[handleStr];
369                 h && h.apply(context, v);
370                 fire(handleStr, xhrObject);
371             };
372         }
373 
374         xhrObject.then(genHandler("success"), genHandler("error"));
375 
376         xhrObject.fin(genHandler("complete"));
377 
378         xhrObject.readyState = 1;
379 
380         /**
381          * @name IO#send
382          * @description fired before sending request
383          * @event
384          * @param {Event.Object} e
385          * @param {Object} e.ajaxConfig current request 's config
386          * @param {IO.XhrObject} e.xhr current xhr object
387          */
388 
389         fire("send", xhrObject);
390 
391         // Timeout
392         if (c.async && timeout > 0) {
393             xhrObject.timeoutTimer = setTimeout(function () {
394                 xhrObject.abort("timeout");
395             }, timeout * 1000);
396         }
397 
398         try {
399             // flag as sending
400             xhrObject.state = 1;
401             transport.send();
402         } catch (e) {
403             // Propagate exception as error if not done
404             if (xhrObject.state < 2) {
405                 xhrObject._xhrReady(-1, e);
406                 // Simply rethrow otherwise
407             } else {
408                 S.error(e);
409             }
410         }
411 
412         return xhrObject;
413     }

449             },

451              * get default config value for io request
452              * @returns {Object}
453              */
454             getConfig:function () {
455                 return defaultConfig;
456             }
457         });
