1 /**
  2  * @fileOverview a scalable client io framework
  3  * @author  yiminghe@gmail.com
  4  */
  5 KISSY.add("ajax/base", function (S, JSON, Event, XhrObject) {
  6 
  7         var rlocalProtocol = /^(?:about|app|app\-storage|.+\-extension|file|widget):$/,
  8             rspace = /\s+/,
  9             rurl = /^([\w\+\.\-]+:)(?:\/\/([^\/?#:]*)(?::(\d+))?)?/,
 10             mirror = function (s) {
 11                 return s;
 12             },
 13             HTTP_PORT = 80,
 14             HTTPS_PORT = 443,
 15             rnoContent = /^(?:GET|HEAD)$/,
 16             curLocation,
 17             curLocationParts;
 18 
 19 
 20         try {
 21             curLocation = location.href;
 22         } catch (e) {
 23             S.log("ajax/base get curLocation error : ");
 24             S.log(e);
 25             // Use the href attribute of an A element
 26             // since IE will modify it given document.location
 27             curLocation = document.createElement("a");
 28             curLocation.href = "";
 29             curLocation = curLocation.href;
 30         }
 31 
 32         curLocationParts = rurl.exec(curLocation);
 33 
 34         var isLocal = rlocalProtocol.test(curLocationParts[1]),
 35             transports = {},
 36             defaultConfig = {
 37                 // isLocal:isLocal,
 38                 type:"GET",
 39                 // only support utf-8 when post, encoding can not be changed actually
 40                 contentType:"application/x-www-form-urlencoded; charset=UTF-8",
 41                 async:true,
 42                 // whether add []
 43                 serializeArray:true,
 44                 // whether param data
 45                 processData:true,
 46                 /*
 47                  url:"",
 48                  context:null,
 49                  // 单位秒!!
 50                  timeout: 0,
 51                  data: null,
 52                  // 可取json | jsonp | script | xml | html | text | null | undefined
 53                  dataType: null,
 54                  username: null,
 55                  password: null,
 56                  cache: null,
 57                  mimeType:null,
 58                  xdr:{
 59                  subDomain:{
 60                  proxy:'http://xx.t.com/proxy.html'
 61                  },
 62                  src:''
 63                  },
 64                  headers: {},
 65                  xhrFields:{},
 66                  // jsonp script charset
 67                  scriptCharset:null,
 68                  crossdomain:false,
 69                  forceScript:false,
 70                  */
 71 
 72                 accepts:{
 73                     xml:"application/xml, text/xml",
 74                     html:"text/html",
 75                     text:"text/plain",
 76                     json:"application/json, text/javascript",
 77                     *:"*/*"
 78                 },
 79                 converters:{
 80                     text:{
 81                         json:JSON.parse,
 82                         html:mirror,
 83                         text:mirror,
 84                         xml:S.parseXML
 85                     }
 86                 },
 87                 contents:{
 88                     xml:/xml/,
 89                     html:/html/,
 90                     json:/json/
 91                 }
 92             };
 93 
 94         defaultConfig.converters.html = defaultConfig.converters.text;
 95 
 96         function setUpConfig(c) {
 97             // deep mix
 98             c = S.mix(S.clone(defaultConfig), c || {}, undefined, undefined, true);
 99             if (!S.isBoolean(c.crossDomain)) {
100                 var parts = rurl.exec(c.url.toLowerCase());
101                 c.crossDomain = !!( parts &&
102                     ( parts[ 1 ] != curLocationParts[ 1 ] || parts[ 2 ] != curLocationParts[ 2 ] ||
103                         ( parts[ 3 ] || ( parts[ 1 ] === "http:" ? HTTP_PORT : HTTPS_PORT ) )
104                             !=
105                             ( curLocationParts[ 3 ] || ( curLocationParts[ 1 ] === "http:" ? HTTP_PORT : HTTPS_PORT ) ) )
106                     );
107             }
108 
109             if (c.processData && c.data && !S.isString(c.data)) {
110                 // 必须 encodeURIComponent 编码 utf-8
111                 c.data = S.param(c.data, undefined, undefined, c.serializeArray);
112             }
113 
114             c.type = c.type.toUpperCase();
115             c.hasContent = !rnoContent.test(c.type);
116 
117             if (!c.hasContent) {
118                 if (c.data) {
119                     c.url += ( /\?/.test(c.url) ? "&" : "?" ) + c.data;
120                 }
121                 if (c.cache === false) {
122                     c.url += ( /\?/.test(c.url) ? "&" : "?" ) + "_ksTS=" + (S.now() + "_" + S.guid());
123                 }
124             }
125 
126             // 数据类型处理链，一步步将前面的数据类型转化成最后一个
127             c.dataType = S.trim(c.dataType || "*").split(rspace);
128 
129             c.context = c.context || c;
130             return c;
131         }
132 
133         function fire(eventType, xhr) {
134 
135 
136             /**
137              * @name io#complete
138              * @description 请求完成（成功或失败）后触发
139              * @event
140              * @param {Event.Object} e
141              * @param {Object} e.ajaxConfig 当前请求的配置
142              * @param {io.XhrObject} e.xhr 当前请求对象
143              */
144 
145             /**
146              * @name io#success
147              * @description 请求成功后触发
148              * @event
149              * @param {Event.Object} e
150              * @param {Object} e.ajaxConfig 当前请求的配置
151              * @param {io.XhrObject} e.xhr 当前请求对象
152              */
153 
154             /**
155              * @name io#error
156              * @description 请求失败后触发
157              * @event
158              * @param {Event.Object} e
159              * @param {Object} e.ajaxConfig 当前请求的配置
160              * @param {io.XhrObject} e.xhr 当前请求对象
161              */
162 
163             io.fire(eventType, { ajaxConfig:xhr.config, xhr:xhr});
164         }
165 
166         function handleXhrEvent(e) {
167             var xhr = this,
168                 c = xhr.config,
169                 type = e.type;
170             if (this.timeoutTimer) {
171                 clearTimeout(this.timeoutTimer);
172             }
173             if (c[type]) {
174                 c[type].call(c.context, xhr.responseData, xhr.statusText, xhr);
175             }
176             fire(type, xhr);
177         }
178 
179         /**
180          * @name io
181          * @description kissy io framework
182          * @namespace io framework
183          * @function
184          * @param {Object} c 发送请求配置选项
185          * @param {String} c.url 请求地址
186          */
187         function io(c) {
188             if (!c.url) {
189                 return undefined;
190             }
191             c = setUpConfig(c);
192             var xhr = new XhrObject(c);
193 
194 
195             /**
196              * @name io#start
197              * @description 生成请求对象前触发
198              * @event
199              * @param {Event.Object} e
200              * @param {Object} e.ajaxConfig 当前请求的配置
201              * @param {io.XhrObject} e.xhr 当前请求对象
202              */
203 
204             fire("start", xhr);
205             var transportContructor = transports[c.dataType[0]] || transports["*"],
206                 transport = new transportContructor(xhr);
207             xhr.transport = transport;
208 
209             if (c.contentType) {
210                 xhr.setRequestHeader("Content-Type", c.contentType);
211             }
212             var dataType = c.dataType[0],
213                 accepts = c.accepts;
214             // Set the Accepts header for the server, depending on the dataType
215             xhr.setRequestHeader(
216                 "Accept",
217                 dataType && accepts[dataType] ?
218                     accepts[ dataType ] + (dataType === "*" ? "" : ", */*; q=0.01"  ) :
219                     accepts[ "*" ]
220             );
221 
222             // Check for headers option
223             for (var i in c.headers) {
224                 xhr.setRequestHeader(i, c.headers[ i ]);
225             }
226 
227             xhr.on("complete success error", handleXhrEvent);
228 
229             xhr.readyState = 1;
230 
231             /**
232              * @name io#send
233              * @description 发送请求前触发
234              * @event
235              * @param {Event.Object} e
236              * @param {Object} e.ajaxConfig 当前请求的配置
237              * @param {io.XhrObject} xhr 当前请求对象
238              */
239 
240             fire("send", xhr);
241 
242             // Timeout
243             if (c.async && c.timeout > 0) {
244                 xhr.timeoutTimer = setTimeout(function () {
245                     xhr.abort("timeout");
246                 }, c.timeout * 1000);
247             }
248 
249             try {
250                 // flag as sending
251                 xhr.state = 1;
252                 transport.send();
253             } catch (e) {
254                 // Propagate exception as error if not done
255                 if (xhr.status < 2) {
256                     xhr.callback(-1, e);
257                     // Simply rethrow otherwise
258                 } else {
259                     S.error(e);
260                 }
261             }
262 
263             return xhr;
264         }
265 
266         S.mix(io, Event.Target);
267         S.mix(io, {
268             isLocal:isLocal,
269             setupConfig:function (setting) {
270                 S.mix(defaultConfig, setting, undefined, undefined, true);
271             },
272             setupTransport:function (name, fn) {
273                 transports[name] = fn;
274             },
275             getTransport:function (name) {
276                 return transports[name];
277             },
278             getConfig:function () {
279                 return defaultConfig;
280             }
281         });
282 
283 
284         return io;
285     },
286     {
287         requires:["json", "event", "./xhrobject"]
288     });
