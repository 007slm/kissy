1 /**
  2  * @fileOverview  a specialized toolbar that is bound to a Grid.Store and provides automatic paging control.
  3  * @author dxq613@gmail.com, yiminghe@gmail.com
  4  */
  5 KISSY.add('grid/pagingbar', function (S, Component, Bar, Bindable) {
  6 
  7     var ID_FIRST = 'first',
  8         ID_PREV = 'prev',
  9         ID_NEXT = 'next',
 10         ID_LAST = 'last',
 11         ID_SKIP = 'skip',
 12         ID_TOTAL_PAGE = 'totalPage',
 13         ID_CURRENT_PAGE = 'curPage',
 14         ID_TOTAL_COUNT = 'totalCount';
 15 
 16     /**
 17      * specialized toolbar that is bound to a Grid.Store and provides automatic paging control.
 18      * Paging Toolbar is typically used as one of the Grid's toolbars.
 19      * @name PagingBar
 20      * @constructor
 21      * @extends Grid.Bar
 22      * @memberOf Grid
 23      */
 24     var PagingBar = Bar.extend([Bindable],
 25         /** @lends Grid.PagingBar.prototype*/
 26         {
 27             /**
 28              * From Bar, Initialize this paging bar items.
 29              * @override
 30              * @protected
 31              */
 32             initializer:function () {
 33                 var _self = this,
 34                     children = _self.get('children'),
 35                     items = _self._getItems(),
 36                     store = _self.get('store');
 37                 S.each(items, function (item) {
 38                     children.push(_self._createItem(item));
 39                 });
 40                 if (store && store.pageSize) {
 41                     _self.set('pageSize', store.pageSize);
 42                 }
 43             },
 44             /**
 45              * bind page change and store events
 46              * @override
 47              * @protected
 48              */
 49             bindUI:function () {
 50                 var _self = this;
 51                 _self._bindButtonEvent();
 52                 //_self._bindStoreEvents();
 53 
 54             },
 55             /**
 56              * skip to page
 57              * this method can fire "beforepagechange" event,
 58              * if you return false in the handler the action will be canceled
 59              * @param {Number} page target page
 60              */
 61             jumpToPage:function (page) {
 62                 if (page <= 0 || page > this.get('totalPage')) {
 63                     return;
 64                 }
 65                 var _self = this,
 66                     store = _self.get('store'),
 67                     pageSize = _self.get('pageSize'),
 68                     index = page - 1,
 69                     start = index * pageSize;
 70                 var result = _self.fire('beforepagechange', {from:_self.get('curPage'), to:page});
 71                 if (store && result !== false) {
 72                     store.load({ start:start, limit:pageSize, pageIndex:index });
 73                 }
 74             },
 75             //after store loaded data,reset the information of paging bar and buttons state
 76             _afterStoreLoad:function (store, params) {
 77                 var _self = this,
 78                     pageSize = _self.get('pageSize'),
 79                     start = 0, //页面的起始记录
 80                     end, //页面的结束记录
 81                     totalCount, //记录的总数
 82                     curPage, //当前页
 83                     totalPage;//总页数;
 84                 if (params) {
 85                     start = params.start || 0;
 86                 } else {
 87                     start = 0;
 88                 }
 89                 //设置加载数据后翻页栏的状态
 90                 totalCount = store.getTotalCount();
 91                 end = totalCount - start > pageSize ? start + store.getCount() : totalCount;
 92                 totalPage = parseInt((totalCount + pageSize - 1) / pageSize, 10);
 93                 totalPage = totalPage > 0 ? totalPage : 1;
 94                 curPage = parseInt(start / pageSize, 10) + 1;
 95 
 96                 _self.set('start', start);
 97                 _self.set('end', end);
 98                 _self.set('totalCount', totalCount);
 99                 _self.set('curPage', curPage);
100                 _self.set('totalPage', totalPage);
101 
102                 //设置按钮状态
103                 _self._setAllButtonsState();
104                 _self._setNumberPages();
105             },
106 
107             //bind page change events
108             _bindButtonEvent:function () {
109                 var _self = this;
110 
111                 //first page handler
112                 _self._bindButtonItemEvent(ID_FIRST, function () {
113                     _self.jumpToPage(1);
114                 });
115 
116                 //previous page handler
117                 _self._bindButtonItemEvent(ID_PREV, function () {
118                     _self.jumpToPage(_self.get('curPage') - 1);
119                 });
120 
121                 //previous page next
122                 _self._bindButtonItemEvent(ID_NEXT, function () {
123                     _self.jumpToPage(_self.get('curPage') + 1);
124                 });
125 
126                 //previous page next
127                 _self._bindButtonItemEvent(ID_LAST, function () {
128                     _self.jumpToPage(_self.get('totalPage'));
129                 });
130                 //skip to one page
131                 _self._bindButtonItemEvent(ID_SKIP, function () {
132                     handleSkip();
133                 });
134                 //input page number and press key "enter"
135                 _self.getItem(ID_CURRENT_PAGE).get('el').on('keyup', function (event) {
136                     event.stopPropagation();
137                     if (event.keyCode === 13) {
138                         handleSkip();
139                     }
140                 });
141                 //when click skip button or press key "enter",cause an action of skipping page
142                 function handleSkip() {
143                     var value = parseInt(_self._getCurrentPageValue(), 10);
144                     if (_self._isPageAllowRedirect(value)) {
145                         _self.jumpToPage(value);
146                     } else {
147                         _self._setCurrentPageValue(_self.get('curPage'));
148                     }
149                 }
150             },
151             // bind button item event
152             _bindButtonItemEvent:function (id, func) {
153                 var _self = this,
154                     item = _self.getItem(id);
155                 if (item) {
156                     item.get('el').on('click', func);
157                 }
158             },
159             onLoad:function (params) {
160                 var _self = this,
161                     store = _self.get('store');
162                 _self._afterStoreLoad(store, params);
163             },
164             //get the items of paging bar
165             _getItems:function () {
166                 var _self = this,
167                     items = _self.get('items');
168                 if (items) {
169                     return items;
170                 }
171                 //default items
172                 items = [];
173                 //first item
174                 items.push(_self._getButtonItem(ID_FIRST));
175                 //previous item
176                 items.push(_self._getButtonItem(ID_PREV));
177                 //separator item
178                 items.push(_self._getSeparator());
179                 //total page of store
180                 items.push(_self._getTextItem(ID_TOTAL_PAGE));
181                 //current page of store
182                 items.push(_self._getTextItem(ID_CURRENT_PAGE));
183                 //button for skip to
184                 items.push(_self._getButtonItem(ID_SKIP));
185                 //separator item
186                 items.push(_self._getSeparator());
187                 //next item
188                 items.push(_self._getButtonItem(ID_NEXT));
189                 //last item
190                 items.push(_self._getButtonItem(ID_LAST));
191                 //separator item
192                 items.push(_self._getSeparator());
193                 //current page of store
194                 items.push(_self._getTextItem(ID_TOTAL_COUNT));
195                 return items;
196             },
197             //get item which the xtype is button
198             _getButtonItem:function (id) {
199                 var _self = this;
200                 return {
201                     id:id,
202                     xtype:'button',
203                     text:_self.get(id + 'Text'),
204                     disabled:true,
205                     elCls:_self.get(id + 'Cls')
206                 };
207             },
208             //get separator item
209             _getSeparator:function () {
210                 return {xtype:'separator'};
211             },
212             //get text item
213             _getTextItem:function (id) {
214                 var _self = this;
215                 return {
216                     id:id,
217                     xtype:'text',
218                     text:_self._getTextItemTpl(id)
219                 };
220             },
221             //get text item's template
222             _getTextItemTpl:function (id) {
223                 var _self = this,
224                     obj = {};
225                 obj[id] = _self.get(id);
226                 return S.substitute(this.get(id + 'Tpl'), obj);
227             },
228             //Whether to allow jump, if it had been in the current page or not within the scope of effective page, not allowed to jump
229             _isPageAllowRedirect:function (value) {
230                 var _self = this;
231                 return value && value > 0 && value <= _self.get('totalPage') && value !== _self.get('curPage');
232             },
233             //when page changed, reset all buttons state
234             _setAllButtonsState:function () {
235                 var _self = this,
236                     store = _self.get('store');
237                 if (store) {
238                     _self._setButtonsState([ID_PREV, ID_NEXT, ID_FIRST, ID_LAST, ID_SKIP], true);
239                 }
240 
241                 if (_self.get('curPage') === 1) {
242                     _self._setButtonsState([ID_PREV, ID_FIRST], false);
243                 }
244                 if (_self.get('curPage') === _self.get('totalPage')) {
245                     _self._setButtonsState([ID_NEXT, ID_LAST], false);
246                 }
247             },
248             //if button id in the param buttons,set the button state
249             _setButtonsState:function (buttons, enable) {
250                 var _self = this,
251                     children = _self.get('children');
252                 S.each(children, function (child) {
253                     if (child.get('xtype') === 'button' && S.inArray(child.get('id'), buttons)) {
254                         child.set('disabled', !enable);
255                     }
256                 });
257             },
258             //show the information of current page , total count of pages and total count of records
259             _setNumberPages:function () {
260                 var _self = this,
261                     totalPageItem = _self.getItem(ID_TOTAL_PAGE),
262                     totalCountItem = _self.getItem(ID_TOTAL_COUNT);
263                 if (totalPageItem) {
264                     totalPageItem.set('html', _self._getTextItemTpl(ID_TOTAL_PAGE));
265                 }
266                 _self._setCurrentPageValue(_self.get(ID_CURRENT_PAGE));
267                 if (totalCountItem) {
268                     totalCountItem.set('html', _self._getTextItemTpl(ID_TOTAL_COUNT));
269                 }
270             },
271             _getCurrentPageValue:function (curItem) {
272                 var _self = this;
273                 curItem = curItem || _self.getItem(ID_CURRENT_PAGE);
274                 var textEl = curItem.get('el').one('input');
275                 return textEl.val();
276             },
277             //show current page in textbox
278             _setCurrentPageValue:function (value, curItem) {
279                 var _self = this;
280                 curItem = curItem || _self.getItem(ID_CURRENT_PAGE);
281                 var textEl = curItem.get('el').one('input');
282                 textEl.val(value);
283             }
284         }, {
285             ATTRS:/** @lends Grid.PagingBar.prototype*/
286             {
287                 /**
288                  * the text of button for first page
289                  * @default {String} "首 页"
290                  */
291                 firstText:{
292                     value:'首 页'
293                 },
294                 /**
295                  * the cls of button for first page
296                  * @default {String} "ks-pb-first"
297                  */
298                 fistCls:{
299                     value:'ks-pb-first'
300                 },
301                 /**
302                  * the text for previous page button
303                  * @default {String} "前一页"
304                  */
305                 prevText:{
306                     value:'上一页'
307                 },
308                 /**
309                  * the cls for previous page button
310                  * @default {String} "ks-pb-prev"
311                  */
312                 prevCls:{
313                     value:'ks-pb-prev'
314                 },
315                 /**
316                  * the text for next page button
317                  * @default {String} "下一页"
318                  */
319                 nextText:{
320                     value:'下一页'
321                 },
322                 /**
323                  * the cls for next page button
324                  * @default {String} "ks-pb-next"
325                  */
326                 nextCls:{
327                     value:'ks-pb-next'
328                 },
329                 /**
330                  * the text for last page button
331                  * @default {String} "末 页"
332                  */
333                 lastText:{
334                     value:'末 页'
335                 },
336                 /**
337                  * the cls for last page button
338                  * @default {String} "ks-pb-last"
339                  */
340                 lastCls:{
341                     value:'ks-pb-last'
342                 },
343                 /**
344                  * the text for skip page button
345                  * @default {String} "跳 转"
346                  */
347                 skipText:{
348                     value:'跳 转'
349                 },
350                 /**
351                  * the cls for skip page button
352                  * @default {String} "ks-pb-last"
353                  */
354                 skipCls:{
355                     value:'ks-pb-skip'
356                 },
357                 /**
358                  * the template of total page info
359                  * @default {String} '共 {totalPage} 页'
360                  */
361                 totalPageTpl:{
362                     value:'共 {totalPage} 页'
363                 },
364                 /**
365                  * the template of current page info
366                  * @default {String} '第 <input type="text" autocomplete="off" class="ks-pb-page" size="20" name="inputItem"> 页'
367                  */
368                 curPageTpl:{
369                     value:'第 <input type="text" '+
370                         'autocomplete="off" class="ks-pb-page" size="20" name="inputItem"> 页'
371                 },
372                 /**
373                  * the template of total count info
374                  * @default {String} '第 <input type="text" autocomplete="off" class="ks-pb-page" size="20" name="inputItem"> 页'
375                  */
376                 totalCountTpl:{
377                     value:'共{totalCount}条记录'
378                 },
379                 /**
380                  * current page of the paging bar
381                  * @private
382                  * @default {Number} 0
383                  */
384                 curPage:{
385                     value:0
386                 },
387                 /**
388                  * total page of the paging bar
389                  * @private
390                  * @default {Number} 0
391                  */
392                 totalPage:{
393                     value:0
394                 },
395                 /**
396                  * total count of the store that the paging bar bind to
397                  * @private
398                  * @default {Number} 0
399                  */
400                 totalCount:{
401                     value:0
402                 },
403                 /**
404                  * The number of records considered to form a 'page'.
405                  * if store set the property ,override this value by store's pageSize
406                  * @private
407                  */
408                 pageSize:{
409                     value:30
410                 },
411                 /**
412                  * The {@link Grid.Store} the paging toolbar should use as its data source.
413                  */
414                 store:{
415 
416                 }
417             },
418             ID_FIRST:ID_FIRST,
419             ID_PREV:ID_PREV,
420             ID_NEXT:ID_NEXT,
421             ID_LAST:ID_LAST,
422             ID_SKIP:ID_SKIP,
423             ID_TOTAL_PAGE:ID_TOTAL_PAGE,
424             ID_CURRENT_PAGE:ID_CURRENT_PAGE,
425             ID_TOTAL_COUNT:ID_TOTAL_COUNT
426         }, {
427             xclass:'pagingbar',
428             priority:2
429         });
430 
431     return PagingBar;
432 
433 }, {
434     requires:['component', './bar', './bindable']
