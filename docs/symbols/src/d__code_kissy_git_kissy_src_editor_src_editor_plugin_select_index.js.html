 85     Select.decorate = function (el) {

 87             items = [];
 88         el.all("option").each(function (opt) {
 89             items.push({
 90                 name:opt.html(),
 91                 value:opt.val()
 92             });
 93         });

 96             title:el.attr("title"),

 98             items:items,

107     function getTipText(str) {
108         if (str && str.indexOf("<") == -1) {
109             return str;

114     S.extend(Select, S.Base, {
115         _init:function () {
116             var self = this,
117                 container = self.get("container"),
118                 fakeEl = self.get("el"),
119                 el = $(SELECT_MARKUP),
120                 titleA = el.one("a"),
121                 title = self.get("title") || "",
122                 cls = self.get("cls"),
123                 text = el.one(".ks-editor-select-text"),
124                 innerText = el.one(".ks-editor-select-text-inner"),
125                 drop = el.one(".ks-editor-select-drop");
126 
127             if (self.get("value") !== undefined) {
128                 innerText.html(self._findNameByV(self.get("value")));
129             } else {
130                 innerText.html(title);
131             }
132 
133             el.unselectable(undefined);
134 
135             text.css("width", self.get("width"));
136 
137             if (title) {
138                 el.attr("title", title);
139             }
140             titleA.attr("href", "javascript:void('" + getTipText(title) + "')");
141             if (cls) {
142                 el.addClass(cls, undefined);
143             }
144             if (fakeEl) {
145                 fakeEl[0].parentNode.replaceChild(el[0], fakeEl[0]);
146             } else if (container) {
147                 el.appendTo(container, undefined);
148             }
149             el.on("click", self._click, self);
150             el.on("keydown", self._keydown, self);
151             self.el = el;
152             self.title = innerText;
153             self._focusA = el.one("a.ks-editor-select");
154             self._focusA.on("blur", function () {
155                 if (self.menu) {
156                     self.menu.hide();
157                 }
158             });
159             Editor.Utils.lazyRun(this, "_prepare", "_real");
160             self.on("afterValueChange", self._valueChange, self);
161             self.on("afterStateChange", self._stateChange, self);
162         },

164         _keydown:function (e) {
165 
166             var keyCode = e.keyCode,
167                 current,
168                 next,
169                 self = this;
170 
171             switch (keyCode) {
172                 case Event.KeyCodes.DOWN:
173                     e.halt();
174                     if (!self._isShown()) {
175                         self._click();
176                         if (!self._findCurrent()) {
177                             $(self.as[0]).addClass(MENU_SELECTED_CLASS, undefined);
178                         }
179                         return;
180                     }
181                     current = $(self._findCurrent());
182                     current.removeClass(MENU_SELECTED_CLASS, undefined);
183                     if (next = current.next()) {
184                         next.addClass(MENU_SELECTED_CLASS, undefined);
185                     } else {
186                         $(self.as[0]).addClass(MENU_SELECTED_CLASS, undefined);
187                     }
188                     break;
189 
190 
191                 case Event.KeyCodes.UP:
192                     e.halt();
193                     if (!self._isShown()) {
194                         return;
195                     }
196                     current = $(self._findCurrent());
197                     current.removeClass(MENU_SELECTED_CLASS, undefined);
198                     if (next = current.prev(undefined, undefined)) {
199                         next.addClass(MENU_SELECTED_CLASS, undefined);
200                     } else {
201                         $(self.as[self.as.length - 1]).addClass(MENU_SELECTED_CLASS, undefined);
202                     }
203                     break;
204 
205 
206                 case Event.KeyCodes.ENTER:
207                     e.halt();
208                     if (!self._isShown()) {
209                         self._click();
210                         if (!self._findCurrent()) {
211                             $(self.as[0]).addClass(MENU_SELECTED_CLASS, undefined);
212                         }
213                         return;
214                     }
215                     current = $(self._findCurrent());
216                     if (!current) {
217                         return;
218                     }
219                     self._select({
220                         target:current
221                     });
222                     break;
223 
224                 case Event.KeyCodes.ESC:
225                     e.halt();
226                     if (!self._isShown()) {
227                         return;
228                     }
229                     self.menu.hide();
230                     break;
231             }
232         },
233 
234         _findCurrent:function () {
235             var as = this.as, ret = undefined;
236             as.each(function (a) {
237                 if (a.hasClass(MENU_SELECTED_CLASS, undefined)) {
238                     ret = a;
239                     return false;
240                 }
241             });
242             return ret;
243         },
244 
245         _isShown:function () {
246             return this.menu && this.menu.get("visible");
247         },
248 
249         _findNameByV:function (v) {
250             var self = this,
251                 name = self.get("title") || "",
252                 items = self.get("items");
253             //显示值，防止下拉描述过多
254             if (self.get("showValue")) {
255                 return v || name;
256             }
257             S.each(items, function (item) {
258                 if (item.value == v) {
259                     name = item.name;
260                     return false;
261                 }
262             });
263             return name;
264         },

266         /**
267          * 当逻辑值变化时，更新select的显示值
268          * @param ev
269          */
270         _valueChange:function (ev) {
271             var v = ev.newVal,
272                 self = this,
273                 name = self._findNameByV(v);
274             self.title.html(name);
275         },
276 
277         _itemsChange:function (ev) {
278             var self = this,
279                 empty,
280                 items = ev.newVal,
281                 _selectList = self._selectList;
282             _selectList.html("");
283             if (items && items.length) {
284                 for (var i = 0; i < items.length; i++) {
285                     var item = items[i], a = new Node(
286                         S.substitute(MENU_ITEM_TPL, {
287                             tip:getTipText(item.name),
288                             value:item.value,
289                             name:item.name
290                         }), item.attrs)
291                         // make menu do not get focus in ie
292                         .unselectable(undefined)
293                         .appendTo(_selectList, undefined);
294                 }
295             } else if (empty = self.get("emptyText")) {
296                 $(
297                     S.substitute(MENU_ITEM_TPL, {
298                         tip:getTipText(empty),
299                         value:"",
300                         name:empty
301                     }))
302                     .appendTo(_selectList, undefined);
303             }
304             self.as = _selectList.all("a");
305         },
306         val:function (v) {
307             var self = this;
308             if (v !== undefined) {
309                 self.set("value", v);
310                 return self;
311             } else {
312                 return self.get("value");
313             }
314         },
315         _resize:function () {
316             var self = this,
317                 menu = self.menu;
318             if (menu.get("visible")) {
319                 self._real();
320             }
321         },
322         _prepare:function () {
323             var self = this,
324                 el = self.el,
325                 focusA = self._focusA,
326                 popUpWidth = self.get("popUpWidth"),
327                 menuNode;
328             //要在适当位置插入 !!!
329             var menu = new Overlay({
330                 prefixCls:'ks-editor-',
331                 autoRender:true,
332                 render:self.get("menuContainer"),
333                 content:SELECT_MENU_MARKUP,
334                 elCls:"ks-editor-menu",
335                 width:popUpWidth ? popUpWidth : el.width(),
336                 zIndex:Editor.baseZIndex(Editor.zIndexManager.SELECT)
337             }), items = self.get("items");
338 
339             menu.on("hide", function () {
340                 focusA.removeClass(SELECT_ACTIVE_CLASS, undefined);
341             });
342 
343             menu.on("show", function () {
344                 focusA.addClass(SELECT_ACTIVE_CLASS, undefined);
345             });
346 
347             addRes.call(self, menu);
348             menuNode = menu.get("contentEl").one("div");
349             self.menu = menu;
350 
351             // make menu do not get focus in non-ie
352             menu.get("el").on("mousedown", function (e) {
353                 e.halt();
354             });
355 
356 
357             //缩放，下拉框跟随
358             Event.on(window, "resize", self._resize, self);
359 
360             addRes.call(self, function () {
361                 Event.remove(window, "resize", self._resize, self);
362             });
363 
364             if (self.get("title")) {
365                 $("<div " +
366                     "class='ks-editor-menu-title ks-editor-select-menuitem' " +
367                     "style='margin-top:-6px;' " +
368                     ">" + self.get("title") + "</div>").appendTo(menuNode, undefined);
369             }
370             self._selectList = $("<div>").appendTo(menuNode, undefined);
371 
372             self._itemsChange({
373                 newVal:items
374             });
375 
376             menuNode.on("click", self._select, self);
377 
378             self.as = self._selectList.all("a");
379 
380             menuNode.on('mouseenter', function () {
381                 self.as.removeClass(MENU_SELECTED_CLASS, undefined);
382             });
383 
384             addRes.call(self, menuNode);
385 
386             self.on("afterItemsChange", self._itemsChange, self);
387 
388             self.menuNode = menuNode;
389         },
390         _stateChange:function (ev) {
391             var v = ev.newVal,
392                 el = this.el;
393             if (v == ENABLED) {
394                 el.removeClass(SELECT_DISABLED_CLASS, undefined);
395             } else {
396                 el.addClass(SELECT_DISABLED_CLASS, undefined);
397             }
398         },
399         enable:function () {
400             this.set("state", ENABLED);
401         },
402         disable:function () {
403             this.set("state", DISABLED);
404         },
405         _select:function (ev) {
406             if (ev && ev.halt) {
407                 ev.halt();
408             }
409             var self = this,
410                 menu = self.menu,
411                 menuNode = self.menuNode,
412                 t = $(ev.target),
413                 a = t.closest(function (n) {
414                     return DOM.contains(menuNode, n) && DOM.nodeName(n) == "a";
415                 }, undefined);
416 
417             if (!a || !a.attr("data-value")) {
418                 return;
419             }
420 
421             var preVal = self.get("value"),
422                 newVal = a.attr("data-value");
423 
424             //更新逻辑值
425             self.set("value", newVal);
426 
427             //触发 click 事件，必要时可监听 afterValueChange
428             self.fire("click", {
429                 newVal:newVal,
430                 prevVal:preVal,
431                 name:a.html()
432             });
433 
434             menu.hide();
435         },
436         _real:function () {
437             var self = this,
438                 el = self.el,
439                 xy = el.offset(),
440                 orixy = S.clone(xy),
441                 menuHeight = self.menu.get("el").height(),
442                 menuWidth = self.menu.get("el").width(),
443                 wt = DOM.scrollTop(),
444                 wl = DOM.scrollLeft(),
445                 wh = DOM.viewportHeight() ,
446                 ww = DOM.viewportWidth(),
447             //右边界坐标,60 is buffer
448                 wr = wl + ww - 60,
449             //下边界坐标
450                 wb = wt + wh,
451             //下拉框向下弹出的y坐标
452                 sb = xy.top + (el.height() - 2),
453             //下拉框右对齐的最右边x坐标
454                 sr = xy.left + el.width() - 2,
455                 align = self.get("align"),
456                 xAlign = align[0],
457                 yAlign = align[1];
458 
459             if (yAlign == "b") {
460                 //向下弹出优先
461                 xy.top = sb;
462                 if (
463                 //不能显示完全
464                     (xy.top + menuHeight) > wb
465                         &&
466                         //向上弹能显示更多
467                         (orixy.top - wt) > (wb - sb)
468                     ) {
469                     xy.top = orixy.top - menuHeight;
470                 }
471             } else {
472                 //向上弹出优先
473                 xy.top = orixy.top - menuHeight;
474 
475                 if (
476                 //不能显示完全
477                     xy.top < wt
478                         &&
479                         //向下弹能显示更多
480                         (orixy.top - wt) < (wb - sb)
481                     ) {
482                     xy.top = sb;
483                 }
484             }
485 
486             if (xAlign == "l") {
487                 //左对其优先
488                 if (
489                 //左对齐不行
490                     (xy.left + menuWidth) > wr
491                         &&
492                         //右对齐可以弹出更多
493 
494                         (sr - wl) > (wr - orixy.left)
495                     ) {
496                     xy.left = sr - menuWidth;
497                 }
498             } else {
499                 //右对齐优先
500                 xy.left = sr - menuWidth;
501                 if (
502                 //右对齐不行
503                     xy.left < wl
504                         &&
505                         //左对齐可以弹出更多
506                         (sr - wl) < (wr - orixy.left)
507                     ) {
508                     xy.left = orixy.left;
509                 }
510             }
511             self.menu.set("xy", [xy.left, xy.top]);
512             self.menu.show();
513         },
514         _click:function (e) {
515             if (e) {
516                 e.halt();
517             }
518             var self = this,
519                 el = self.el,
520                 menu = self.menu,
521                 v = self.get("value");
522 
523             if (menu && menu.get("visible")) {
524                 menu.hide();
525                 return;
526             }
527 
528             // chrome will make body focus !!
529             self._focusA[0].focus();
530 
531             if (el.hasClass(SELECT_DISABLED_CLASS, undefined)) {
532                 return false;
533             }
534 
535             self.fire("select");
536 
537             self._prepare();
538 
539             //可能的话当显示层时，高亮当前值对应option
540             if (v && menu) {
541                 var as = self.as;
542                 as.each(function (a) {
543                     if (a.attr("data-value") == v) {
544                         a.addClass(MENU_SELECTED_CLASS, undefined);
545                     } else {
546                         a.removeClass(MENU_SELECTED_CLASS, undefined);
547                     }
548                 });
549             }
550         },
551         destroy:function () {
552             destroyRes.call(this);
553             this.el.remove();
554         }
555     });

557     /**
558      * 将button ui 和点击功能分离
559      * 按钮必须立刻显示出来，功能可以慢慢加载
560      */
561     Editor.prototype.addSelect = function (cfg, methods, SelectType) {
562 
563         SelectType = SelectType || Select;
564 
565         var self = this;
566 
567         cfg.editor = self;
568 
569         var s = new SelectType(S.mix({
570             container:self.get("toolBarEl"),
571             doc:self.get("document")[0]
572         }, cfg));
573 
574         S.mix(s, methods);
575 
576         self.on("selectionChange", function () {
577             if (self.get("mode") == Editor.SOURCE_MODE) {
578                 return;
579             }
580             s.selectionChange && s.selectionChange.apply(s, arguments);
581         });
582 
583         self.on("destroy", function () {
584             s.destroy();
585         });
586 
587         s.on("click", function (ev) {
588             var t = ev.type;
589             if (s[t]) {
590                 s[t].apply(s, arguments);
591             }
592             ev.halt();
593         });
594 
595         if (cfg.mode == Editor.WYSIWYG_MODE) {
596             self.on("wysiwygMode", s.enable, s);
597             self.on("sourceMode", s.disable, s);
598         }
599 
600         s.init && s.init();
601         return s;
602 
603     };
604 
605     return Select;
