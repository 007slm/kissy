1 /**
  2  * @fileOverview scalable event framework for kissy (refer DOM3 Events)
  3  *               how to fire event just like browser?
  4  * @author  yiminghe@gmail.com,lifesinger@gmail.com
  5  */
  6 KISSY.add('event/base', function (S, DOM, EventObject, Utils, handle, special) {
  7 
  8     var isValidTarget = Utils.isValidTarget,
  9         splitAndRun = Utils.splitAndRun,
 10         nodeName = DOM._nodeName,
 11         trim = S.trim,
 12         TRIGGERED_NONE = Utils.TRIGGERED_NONE;
 13 
 14     /**
 15      * @namespace
 16      * @name Event
 17      */
 18     var Event =
 19     /**
 20      * @lends Event
 21      */
 22     {
 23         /**
 24          * fire event,simulate bubble in browser. similar to dispatchEvent in DOM3 Events
 25          * @memberOf Event
 26          * @param targets html nodes
 27          * @param {String|Event.Object} eventType event type
 28          * @param [eventData] additional event data
 29          * @param {boolean} [onlyHandlers] only fire handlers
 30          * @returns {boolean} The return value of fire/dispatchEvent indicates
 31          *                 whether any of the listeners which handled the event called preventDefault.
 32          *                 If preventDefault was called the value is false, else the value is true.
 33          */
 34         fire:function (targets, eventType, eventData, onlyHandlers/*internal usage*/) {
 35             var ret = true, r;
 36             // custom event firing moved to target.js
 37             eventData = eventData || {};
 38             if (S.isString(eventType)) {
 39                 eventType = trim(eventType);
 40                 if (eventType.indexOf(" ") > -1) {
 41                     splitAndRun(eventType, function (t) {
 42                         r = Event.fire(targets, t, eventData, onlyHandlers);
 43                         if (ret !== false) {
 44                             ret = r;
 45                         }
 46                     });
 47                     return ret;
 48                 }
 49                 // protect event type
 50                 eventData.type = eventType;
 51             } else if (eventType instanceof EventObject) {
 52                 eventData = eventType;
 53                 eventType = eventData.type;
 54             }
 55 
 56             var typedGroups = Utils.getTypedGroups(eventType),
 57                 _ks_groups = typedGroups[1];
 58 
 59             if (_ks_groups) {
 60                 _ks_groups = Utils.getGroupsRe(_ks_groups);
 61             }
 62 
 63             eventType = typedGroups[0];
 64 
 65             S.mix(eventData, {
 66                 type:eventType,
 67                 _ks_groups:_ks_groups
 68             });
 69 
 70             DOM.query(targets).each(function (target) {
 71                 r = fireDOMEvent(target, eventType, eventData, onlyHandlers);
 72                 if (ret !== false) {
 73                     ret = r;
 74                 }
 75             });
 76             return ret;
 77         },
 78 
 79         /**
 80          * does not cause default behavior to occur
 81          * does not bubble up the DOM hierarchy
 82          * @param targets
 83          * @param {Event.Object | String} eventType
 84          * @param [eventData]
 85          */
 86         fireHandler:function (targets, eventType, eventData) {
 87             return Event.fire(targets, eventType, eventData, 1);
 88         }
 89     };
 90 
 91     /**
 92      * fire dom event from bottom to up , emulate dispatchEvent in DOM3 Events
 93      * @return boolean The return value of dispatchEvent indicates
 94      *                 whether any of the listeners which handled the event called preventDefault.
 95      *                 If preventDefault was called the value is false, else the value is true.
 96      */
 97     function fireDOMEvent(target, eventType, eventData, onlyHandlers) {
 98         if (!isValidTarget(target)) {
 99             return false;
100         }
101         var s = special[eventType];
102         // TODO bug : when fire mouseenter , it also fire mouseover in firefox/chrome
103         if (s && s['onFix']) {
104             eventType = s['onFix'];
105         }
106 
107         var event,
108             ret = true;
109         if (eventData instanceof EventObject) {
110             event = eventData;
111         } else {
112             event = new EventObject(target, undefined, eventType);
113             S.mix(event, eventData);
114         }
115         /**
116          * identify event as fired manually
117          */
118         event._ks_fired = 1;
119         /*
120          The target of the event is the EventTarget on which dispatchEvent is called.
121          */
122         // TODO: protect target , but incompatible
123         // event.target=target;
124         // protect type
125         event.type = eventType;
126 
127         // onlyHandlers is equal to event.halt()
128         // but we can not call event.halt()
129         // because handle will check event.isPropagationStopped
130 
131         var cur = target,
132             t,
133             ontype = "on" + eventType;
134 
135         //bubble up dom tree
136         do {
137             event.currentTarget = cur;
138             t = handle(cur, event);
139             if (ret !== false) {
140                 ret = t;
141             }
142             // Trigger an inline bound script
143             if (cur[ ontype ] &&
144                 cur[ ontype ].call(cur) === false) {
145                 event.preventDefault();
146             }
147             // Bubble up to document, then to window
148             cur = cur.parentNode ||
149                 cur.ownerDocument ||
150                 cur === target.ownerDocument && window;
151         } while (!onlyHandlers && cur && !event.isPropagationStopped);
152 
153         if (!onlyHandlers && !event.isDefaultPrevented) {
154             if (!(eventType === "click" &&
155                 nodeName(target, "a"))) {
156                 var old;
157                 try {
158                     // execute default action on dom node
159                     // so exclude window
160                     // exclude focus/blue on hidden element
161                     if (ontype &&
162                         target[ eventType ] &&
163                         (
164                             (eventType !== "focus" && eventType !== "blur") ||
165                                 target.offsetWidth !== 0
166                             ) &&
167                         !S.isWindow(target)) {
168                         // Don't re-trigger an onFOO event when we call its FOO() method
169                         old = target[ ontype ];
170 
171                         if (old) {
172                             target[ ontype ] = null;
173                         }
174 
175                         // 记录当前 trigger 触发
176                         Utils.Event_Triggered = eventType;
177 
178                         // 只触发默认事件，而不要执行绑定的用户回调
179                         // 同步触发
180                         target[ eventType ]();
181                     }
182                 } catch (ieError) {
183                     S.log("trigger action error : ");
184                     S.log(ieError);
185                 }
186 
187                 if (old) {
188                     target[ ontype ] = old;
189                 }
190 
191                 Utils.Event_Triggered = TRIGGERED_NONE;
192             }
193         }
194         return ret;
195     }
196 
197     return Event;
198 }, {
199     requires:["dom", "./object", "./utils", './handle', './special']
200 });
