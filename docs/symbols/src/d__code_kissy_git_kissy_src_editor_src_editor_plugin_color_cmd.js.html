1 KISSY.add("editor/plugin/color/cmd", function (S, Editor) {
  2     function applyColor(editor, c, styles) {
  3         var doc = editor.get("document")[0];
  4         editor.execCommand("save");
  5         if (c) {
  6             new Editor.Style(styles, {
  7                 color:c
  8             }).apply(doc);
  9         } else {
 10             // Value 'inherit'  is treated as a wildcard,
 11             // which will match any value.
 12             //清除已设格式
 13             new Editor.Style(styles, {
 14                 color:"inherit"
 15             }).remove(doc);
 16         }
 17         editor.execCommand("save");
 18     }
 19 
 20     return {
 21         applyColor:applyColor
 22     };
 23 }, {
 24     requires:['editor']
 25 });
