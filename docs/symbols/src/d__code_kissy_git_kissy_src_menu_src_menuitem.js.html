1 /**
  2  * @fileOverview menu item ,child component for menu
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("menu/menuitem", function (S, Component, MenuItemRender) {
  6 
  7     var $ = S.all;
  8 
  9     /**
 10      * A menu item component which menu is consisted of.
 11      * @class
 12      * @name Item
 13      * @memberOf Menu
 14      * @extends Component.Controller
 15      */
 16     var MenuItem = Component.Controller.extend(
 17         /**
 18          * @lends Menu.Item#
 19          */
 20         {
 21 
 22             /**
 23              * Handle mouseenter event. Make parent menu to highlight itself.
 24              * Protected, should only be overridden by subclasses.
 25              * @param {Event.Object} e Mouseenter event object.
 26              * @protected
 27              * @override
 28              */
 29             handleMouseEnter:function (e) {
 30                 // 父亲不允许自己处理
 31                 if (MenuItem.superclass.handleMouseEnter.call(this, e)) {
 32                     return true;
 33                 }
 34                 this.get("parent").set("highlightedItem", this);
 35             },
 36 
 37             /**
 38              * Handle mouseleave event. Make parent menu to unhighlight itself.
 39              * Protected, should only be overridden by subclasses.
 40              * @param {Event.Object} e Mouseleave event object.
 41              * @protected
 42              * @override
 43              */
 44             handleMouseLeave:function (e) {
 45                 // 父亲不允许自己处理
 46                 if (MenuItem.superclass.handleMouseLeave.call(this, e)) {
 47                     return true;
 48                 }
 49                 this.get("parent").set("highlightedItem", null);
 50             },
 51 
 52             /**
 53              * Perform default action when click on enter on this menuitem.
 54              * If selectable, then make it selected.
 55              * If checkable, then toggle it.
 56              * Finally fire click on its parent menu.
 57              * @protected
 58              * @override
 59              */
 60             performActionInternal:function () {
 61                 var self = this;
 62                 // 可选
 63                 if (self.get("selectable")) {
 64                     self.set("selected", true);
 65                 }
 66                 // 可选中，取消选中
 67                 if (self.get("checkable")) {
 68                     self.set("checked", !self.get("checked"));
 69                 }
 70                 self.get("parent").fire("click", {
 71                     target:self
 72                 });
 73                 return true;
 74             },
 75 
 76             _uiSetHighlighted:function (v) {
 77                 // 是否要滚动到当前菜单项(横向，纵向)
 78                 if (v) {
 79                     var el = this.get("el"),
 80                     // 找到向上路径上第一个可以滚动的容器，直到父组件节点（包括）
 81                     // 找不到就放弃，为效率考虑不考虑 parent 的嵌套可滚动 div
 82                         p = el.parent(function (e) {
 83                             return $(e).css("overflow") != "visible";
 84                         }, this.get("parent").get("el").parent());
 85                     if (!p) {
 86                         return;
 87                     }
 88                     el.scrollIntoView(p, undefined, undefined, true);
 89                 }
 90             },
 91 
 92             /**
 93              * Check whether this menu item contains specified element.
 94              * @param {NodeList} element Element to be tested.
 95              */
 96             containsElement:function (element) {
 97                 return this.get('view') && this.get('view').containsElement(element);
 98             }
 99 
100         }, {
101             ATTRS:/**
102              * @lends Menu.Item#
103              */
104             {
105 
106                 focusable:{
107                     value:false
108                 },
109 
110                 visibleMode:{
111                     value:"display"
112                 },
113 
114                 handleMouseEvents:{
115                     value:false
116                 },
117 
118                 /**
119                  * Whether the menu item is selectable or not.
120                  * Set to true for option.
121                  * @type Boolean
122                  */
123                 selectable:{
124                     view:true
125                 },
126 
127                 /**
128                  * Whether the menu item is checkable or not.
129                  * Set to true for checkbox option.
130                  * @type Boolean
131                  */
132                 checkable:{
133                     view:true
134                 },
135 
136                 /**
137                  * The value associated with the menu item.
138                  */
139                 value:{},
140 
141                 /**
142                  * Whether the menu item is checked.
143                  * @type Boolean
144                  */
145                 checked:{
146                     view:true
147                 },
148 
149                 /**
150                  * Whether the menu item is selected.
151                  * @type Boolean
152                  */
153                 selected:{
154                     view:true
155                 },
156 
157                 xrender:{
158                     value:MenuItemRender
159                 }
160             },
161 
162             HTML_PARSER:{
163                 selectable:function (el) {
164                     var cls = this.getCssClassWithPrefix("menuitem-selectable");
165                     return el.hasClass(cls);
166                 }
167             }
168         }, {
169             xclass:"menuitem",
170             priority:10
171         });
172 
173     return MenuItem;
174 }, {
175     requires:['component', './menuitemRender']
176 });
