 12             return doc.getElementsByTagName('head')[0] || doc.documentElement;
 13         },
 14         isWebKit:!!ua.match(/AppleWebKit/),
 15         IE : !!ua.match(/MSIE/),
 16         isCss:function(url) {
 17             return /\.css(?:\?|$)/i.test(url);
 18         },
 19         isLinkNode:function(n) {
 20             return n.nodeName.toLowerCase() == 'link';
 21         },
 22         /**
 23          * resolve relative part of path
 24          * x/../y/z -> y/z
 25          * x/./y/z -> x/y/z
 26          * @param path uri path
 27          * @return {string} resolved path
 28          * @description similar to path.normalize in nodejs
 29          */
 30         normalizePath:function(path) {
 31             var paths = path.split("/"),
 32                 re = [],
 33                 p;
 34             for (var i = 0; i < paths.length; i++) {
 35                 p = paths[i];
 36                 if (p == ".") {
 37                 } else if (p == "..") {
 38                     re.pop();
 39                 } else {
 40                     re.push(p);
 41                 }
 42             }
 43             return re.join("/");
 44         },
 45 
 46         /**
 47          * 根据当前模块以及依赖模块的相对路径，得到依赖模块的绝对路径
 48          * @param moduleName 当前模块
 49          * @param depName 依赖模块
 50          * @return {string|Array} 依赖模块的绝对路径
 51          * @description similar to path.resolve in nodejs
 52          */
 53         normalDepModuleName:function normalDepModuleName(moduleName, depName) {
 54             if (!depName) {
 55                 return depName;
 56             }
 57             if (S.isArray(depName)) {
 58                 for (var i = 0; i < depName.length; i++) {
 59                     depName[i] = normalDepModuleName(moduleName, depName[i]);
 60                 }
 61                 return depName;
 62             }
 63             if (startsWith(depName, "../") || startsWith(depName, "./")) {
 64                 var anchor = "",index;
 65                 // x/y/z -> x/y/
 66                 if ((index = moduleName.lastIndexOf("/")) != -1) {
 67                     anchor = moduleName.substring(0, index + 1);
 68                 }
 69                 return normalizePath(anchor + depName);
 70             } else if (depName.indexOf("./") != -1
 71                 || depName.indexOf("../") != -1) {
 72                 return normalizePath(depName);
 73             } else {
 74                 return depName;
 75             }
 76         },
 77         //去除后缀名，要考虑时间戳?
 78         removePostfix:function (path) {
 79             return path.replace(/(-min)?\.js[^/]*$/i, "");
 80         },
 81         /**
 82          * 路径正则化，不能是相对地址
 83          * 相对地址则转换成相对页面的绝对地址
 84          * 用途:
 85          * package path 相对地址则相对于当前页面获取绝对地址
 86          */
 87         normalBasePath:function (path) {
 88             path = S.trim(path);
 89 
 90             // path 为空时，不能变成 "/"
 91             if (path && path.charAt(path.length - 1) != '/') {
 92                 path += "/";
 93             }
 94 
 95             /**
 96              * 一定要正则化，防止出现 ../ 等相对路径
 97              * 考虑本地路径
 98              */
 99             if (!path.match(/^(http(s)?)|(file):/i)
100                 && !startsWith(path, "/")) {
101                 path = loader.__pagePath + path;
102             }
103             return normalizePath(path);
104         },
105 
106         /**
107          * 相对路径文件名转换为绝对路径
108          * @param path
109          */
110         absoluteFilePath:function(path) {
111             path = utils.normalBasePath(path);
112             return path.substring(0, path.length - 1);
113         },
114 
115         //http://wiki.commonjs.org/wiki/Packages/Mappings/A

120                     names[i] += "index";
121                 }
122             }
123             return names;
124         }
