1 /**
  2  * @fileOverview mask extension for kissy
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("component/uibase/maskrender", function (S, UA, Node) {
  6 
  7     /**
  8      * 每组相同 prefixCls 的 position 共享一个遮罩
  9      */
 10     var maskMap = {
 11             /**
 12              * {
 13              *  node:
 14              *  num:
 15              * }
 16              */
 17 
 18         },
 19         ie6 = (UA['ie'] === 6),
 20         $ = Node.all;
 21 
 22     function getMaskCls(self) {
 23         return self.get("prefixCls") + "ext-mask";
 24     }
 25 
 26     function docWidth() {
 27         return  ie6 ? ("expression(KISSY.DOM.docWidth())") : "100%";
 28     }
 29 
 30     function docHeight() {
 31         return ie6 ? ("expression(KISSY.DOM.docHeight())") : "100%";
 32     }
 33 
 34     function initMask(maskCls) {
 35         var mask = $("<div " +
 36             " style='width:" + docWidth() + ";" +
 37             "left:0;" +
 38             "top:0;" +
 39             "height:" + docHeight() + ";" +
 40             "position:" + (ie6 ? "absolute" : "fixed") + ";'" +
 41             " class='" +
 42             maskCls +
 43             "'>" +
 44             (ie6 ? "<" + "iframe " +
 45                 "style='position:absolute;" +
 46                 "left:" + "0" + ";" +
 47                 "top:" + "0" + ";" +
 48                 "background:red;" +
 49                 "width: expression(this.parentNode.offsetWidth);" +
 50                 "height: expression(this.parentNode.offsetHeight);" +
 51                 "filter:alpha(opacity=0);" +
 52                 "z-index:-1;'></iframe>" : "") +
 53             "</div>")
 54             .prependTo("body");
 55         /**
 56          * 点 mask 焦点不转移
 57          */
 58         mask.unselectable();
 59         mask.on("mousedown", function (e) {
 60             e.preventDefault();
 61         });
 62         return mask;
 63     }
 64 
 65     function Mask() {
 66     }
 67 
 68     Mask.prototype = {
 69 
 70         _maskExtShow:function () {
 71             var self = this,
 72                 maskCls = getMaskCls(self),
 73                 maskDesc = maskMap[maskCls],
 74                 maskShared = self.get("maskShared"),
 75                 mask = self.get("maskNode");
 76             if (!mask) {
 77                 if (maskShared) {
 78                     if (maskDesc) {
 79                         mask = maskDesc.node;
 80                     } else {
 81                         mask = initMask(maskCls);
 82                         maskDesc = maskMap[maskCls] = {
 83                             num:0,
 84                             node:mask
 85                         };
 86                     }
 87                 } else {
 88                     mask = initMask(maskCls);
 89                 }
 90                 self.__set("maskNode", mask);
 91             }
 92             mask.css("z-index", self.get("zIndex") - 1);
 93             if (maskShared) {
 94                 maskDesc.num++;
 95             }
 96             if (!maskShared || maskDesc.num == 1) {
 97                 mask.show();
 98             }
 99         },
100 
101         _maskExtHide:function () {
102             var self = this,
103                 maskCls = getMaskCls(self),
104                 maskDesc = maskMap[maskCls],
105                 maskShared = self.get("maskShared"),
106                 mask = self.get("maskNode");
107             if (maskShared && maskDesc) {
108                 maskDesc.num = Math.max(maskDesc.num - 1, 0);
109                 if (maskDesc.num == 0) {
110                     mask.hide();
111                 }
112             } else {
113                 mask.hide();
114             }
115         },
116 
117         __destructor:function () {
118             var self = this,
119                 maskShared = self.get("maskShared"),
120                 mask = self.get("maskNode");
121             if (self.get("maskNode")) {
122                 if (maskShared) {
123                     if (self.get("visible")) {
124                         self._maskExtHide();
125                     }
126                 } else {
127                     mask.remove();
128                 }
129             }
130         }
131 
132     };
133 
134     return Mask;
135 }, {
136     requires:["ua", "node"]
137 });
