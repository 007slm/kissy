1 /**
  2  * @fileOverview collection of models
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("mvc/collection", function (S, Event, Model, sync, Base) {
  6 
  7     function findModelIndex(mods, mod, comparator) {
  8         var i = mods.length;
  9         if (comparator) {
 10             var k = comparator(mod);
 11             for (i = 0; i < mods.length; i++) {
 12                 var k2 = comparator(mods[i]);
 13                 if (k < k2) {
 14                     break;
 15                 }
 16             }
 17         }
 18         return i;
 19     }
 20 
 21     /**
 22      * Collection. A list of model.
 23      * @class
 24      * @memberOf MVC
 25      * @extends Base
 26      */
 27     function Collection() {
 28         Collection.superclass.constructor.apply(this, arguments);
 29     }
 30 
 31     Collection.ATTRS =
 32     /**
 33      * @lends MVC.Collection#
 34      */
 35     {
 36         /**
 37          * Model constructor with in current collection.
 38          * @type MVC.Model
 39          */
 40         model:{
 41             value:Model
 42         },
 43         /**
 44          * Model list.
 45          * @type MVC.Model[]
 46          */
 47         models:{
 48             /*
 49              normalize model list
 50              @param models
 51              */
 52             setter:function (models) {
 53                 var prev = this.get("models");
 54                 this.remove(prev, {silent:1});
 55                 this.add(models, {silent:1});
 56                 return this.get("models");
 57             },
 58             value:[]
 59         },
 60         /**
 61          * Get url for sending data to server.
 62          * @type String|Function
 63          */
 64         url:{
 65             value:""
 66         },
 67         /**
 68          * Comparator function for index getter when adding model.
 69          * default to append to last of current model list.
 70          * @type Function
 71          */
 72         comparator:{},
 73         /**
 74          * Sync function to sync data with server.
 75          * Default to call {@link MVC.sync}
 76          * @type Function
 77          */
 78         sync:{
 79             value:function () {
 80                 sync.apply(this, arguments);
 81             }
 82         },
 83         /**
 84          * Get structured data from raw data returned from server.
 85          * default to return raw data from server.
 86          * @type Function
 87          */
 88         parse:{
 89             value:function (resp) {
 90                 return resp;
 91             }
 92         }
 93     };
 94 
 95     S.extend(Collection, Base,
 96         /**
 97          * @lends MVC.Collection#
 98          */
 99         {
100             /**
101              * Sort model list according {@link MVC.Collection#comparator}.
102              */
103             sort:function () {
104                 var comparator = this.get("comparator");
105                 if (comparator) {
106                     this.get("models").sort(function (a, b) {
107                         return comparator(a) - comparator(b);
108                     });
109                 }
110             },
111 
112             /**
113              * Get json representation of this collection.
114              * @return Object[]
115              */
116             toJSON:function () {
117                 return S.map(this.get("models"), function (m) {
118                     return m.toJSON();
119                 });
120             },
121 
122             /**
123              * Add a model to current collection.
124              * @param {Object|MVC.Model} model Model or json data to be added.
125              * @param {object} [opts] Add config
126              * @param {Function} opts.silent Whether to fire add event.
127              */
128             add:function (model, opts) {
129                 var self = this,
130                     ret = true;
131                 if (S.isArray(model)) {
132                     var orig = [].concat(model);
133                     S.each(orig, function (m) {
134                         var t = self._add(m, opts);
135                         ret = ret && t;
136                     });
137                 } else {
138                     ret = self._add(model, opts);
139                 }
140                 return ret;
141             },
142 
143             /**
144              * Remove an existing model from current collection.
145              * @param {MVC.Model} model Model to be removed.
146              * @param {object} [opts] Remove config.
147              * @param {Function} opts.silent Whether to fire remove event.
148              */
149             remove:function (model, opts) {
150                 var self = this;
151                 if (S.isArray(model)) {
152                     var orig = [].concat(model);
153                     S.each(orig, function (m) {
154                         self._remove(m, opts);
155                     });
156                 } else if (model) {
157                     self._remove(model, opts);
158                 }
159             },
160 
161             /**
162              * Get model at specified index.
163              * @param {Number} i Specified index.
164              */
165             at:function (i) {
166                 return this.get("models")[i];
167             },
168 
169             _normModel:function (model) {
170                 var ret = true;
171                 if (!(model instanceof Model)) {
172                     var data = model,
173                         modelConstructor = this.get("model");
174                     model = new modelConstructor();
175                     ret = model.set(data, {
176                         silent:1
177                     });
178                 }
179                 return ret && model;
180             },
181 
182             /**
183              * Initialize model list by loading data using sync mechanism.
184              * @param {object} opts Load config.
185              * @param {Function} opts.success Callback when load is successful.
186              * @param {Function} opts.error Callback when error occurs on loading.
187              * @param {Function} opts.complete Callback when load is complete.
188              */
189             load:function (opts) {
190                 var self = this;
191                 opts = opts || {};
192                 var success = opts.success;
193                 /**
194                  * @ignore
195                  */
196                 opts.success = function (resp) {
197                     if (resp) {
198                         var v = self.get("parse").call(self, resp);
199                         if (v) {
200                             self.set("models", v, opts);
201                         }
202                     }
203                     // https://github.com/kissyteam/kissy/issues/138
204                     S.each(self.get("models"), function (m) {
205                         m.__isModified = 0;
206                     });
207                     success && success.apply(this, arguments);
208                 };
209                 self.get("sync").call(self, self, 'read', opts);
210                 return self;
211             },
212 
213             /**
214              * Add a model to current collection by provide json data.
215              * @param {Object} model Json data represent model data.
216              * @param {object} opts Create config.
217              * @param {Function} opts.success Callback when create is successful.
218              * @param {Function} opts.error Callback when error occurs on creating.
219              * @param {Function} opts.complete Callback when create is complete.
220              * @param {Function} opts.silent Whether to fire add event.
221              */
222             create:function (model, opts) {
223                 var self = this;
224                 opts = opts || {};
225                 model = this._normModel(model);
226                 if (model) {
227                     model.addToCollection(self);
228                     var success = opts.success;
229                     opts.success = function () {
230                         self.add(model, opts);
231                         success && success();
232                     };
233                     model.save(opts);
234                 }
235                 return model;
236             },
237 
238             _add:function (model, opts) {
239                 model = this._normModel(model);
240                 if (model) {
241                     opts = opts || {};
242                     var index = findModelIndex(this.get("models"), model, this.get("comparator"));
243                     this.get("models").splice(index, 0, model);
244                     model.addToCollection(this);
245                     if (!opts['silent']) {
246                         this.fire("add", {
247                             model:model
248                         });
249                     }
250                 }
251                 return model;
252             },
253 
254             /**
255              * not call model.destroy ,maybe model belongs to multiple collections
256              * @private
257              */
258             _remove:function (model, opts) {
259                 opts = opts || {};
260                 var index = S.indexOf(model, this.get("models"));
261                 if (index != -1) {
262                     this.get("models").splice(index, 1);
263                     model.removeFromCollection(this);
264                 }
265                 if (!opts['silent']) {
266                     this.fire("remove", {
267                         model:model
268                     });
269                 }
270             },
271 
272             /**
273              * Get model instance by id.
274              * @param {String} id
275              */
276             getById:function (id) {
277                 var models = this.get("models");
278                 for (var i = 0; i < models.length; i++) {
279                     var model = models[i];
280                     if (model.getId() === id) {
281                         return model;
282                     }
283                 }
284                 return null;
285             },
286 
287             /**
288              * Get model instance by client id.
289              * @param {String} cid Client id auto generated by model.
290              */
291             getByCid:function (cid) {
292                 var models = this.get("models");
293                 for (var i = 0; i < models.length; i++) {
294                     var model = models[i];
295                     if (model.get("clientId") === cid) {
296                         return model;
297                     }
298                 }
299                 return null;
300             }
301 
302         });
303 
304     return Collection;
305 
306 }, {
307     requires:['event', './model', './sync', 'base']
308 });
