  9     var ALIGN = {
 10         points:["bl", "tl"],
 11         overflow:{
 12             adjustX:1,
 13             adjustY:1
 14         }
 15     };

 17     function alignMenuImmediately(self) {
 18         var menu = self.get("menu"),
 19             menuCfg = self.get("menuCfg") || {};
 20         menu.set("align", S.merge({
 21             node:self.get("el")
 22         }, ALIGN, menuCfg.align));
 23     }
 24 
 25     function alignWithTokenImmediately(self) {
 26         var inputDesc = self._getInputDesc(),
 27             tokens = inputDesc.tokens,
 28             menu = self.get("menu"),
 29             cursorPosition = inputDesc.cursorPosition,
 30             tokenIndex = inputDesc.tokenIndex,
 31             tokenCursorPosition,
 32             el = self.get("el");
 33         tokenCursorPosition = tokens.slice(0, tokenIndex).join("").length;
 34         if (tokenCursorPosition > 0) {
 35             // behind separator
 36             ++tokenCursorPosition;
 37         }
 38         el.prop("selectionStart", tokenCursorPosition);
 39         el.prop("selectionEnd", tokenCursorPosition);
 40         var cursorOffset = el.prop("KsCursorOffset");
 41         el.prop("selectionStart", cursorPosition);
 42         el.prop("selectionEnd", cursorPosition);
 43         menu.set("xy", [cursorOffset.left, cursorOffset.top]);
 44     }
 45 
 46     function alignImmediately(self) {
 47         if (self.get("multiple") &&
 48             self.get("alignWithCursor")) {
 49             alignWithTokenImmediately(self);
 50         } else {
 51             alignMenuImmediately(self);
 52         }
 53     }

 55     /**
 56      * Input/Textarea Wrapper for autoComplete
 57      * @name AutoComplete
 58      * @extends Component.Controller
 59      * @class
 60      */
 61     AutoComplete = UIBase.create(Component.Controller, [],
 62         /**
 63          * @lends AutoComplete
 64          */
 65         {
 66 
 67             // user's input text
 68             _savedInputValue:null,
 69 
 70             _stopNotify:0,
 71 
 72             bindUI:function () {
 73                 var self = this,
 74                     el = self.get("el");
 75                 el.on("valuechange", self._onValueChange, self);
 76             },
 77             /**
 78              * fetch autoComplete list by value and show autoComplete list
 79              * @param {String} value value for fetching autoComplete list
 80              */
 81             sendRequest:function (value) {
 82                 var self = this,
 83                     dataSource = self.get("dataSource");
 84                 dataSource.fetchData(value, self._renderData, self);
 85             },
 86             _onValueChange:function () {
 87                 var self = this;
 88                 if (self._stopNotify) {
 89                     return;
 90                 }
 91                 var value = self._getValue();
 92 
 93                 if (value === undefined) {
 94                     var autoCompleteMenu = self.get("menu");
 95                     if (autoCompleteMenu) {
 96                         autoCompleteMenu.hide();
 97                     }
 98                     return;
 99                 }
100                 self._savedInputValue = value;
101                 S.log("value change: " + value);
102                 self.sendRequest(value);
103             },
104 
105             handleBlur:function () {
106                 AutoComplete.superclass.handleBlur.apply(this, arguments);
107                 var autoCompleteMenu = this.get("menu");
108                 // S.log("input blur!!!!!!!");
109                 if (autoCompleteMenu) {
110                     // 通知 menu
111                     autoCompleteMenu._hideForAutoComplete();
112                 }
113             },
114 
115             _renderData:function (data) {
116                 var self = this,
117                     autoCompleteMenu = self.get("menu");
118                 autoCompleteMenu.removeChildren(true);
119                 if (data && data.length) {
120                     data = data.slice(0, self.get("maxItemCount"));
121                     var menuCfg = self.get("menuCfg") || {};
122                     var v;
123                     // 同步当前 width
124                     autoCompleteMenu.set("width",
125                         menuCfg.width === undefined ?
126                             self.get("el").css("width") :
127                             menuCfg.width);
128                     var contents;
129                     if (self.get("format")) {
130                         contents = self.get("format").call(self,
131                             self._getValue(),
132                             data);
133                     } else {
134                         contents = [];
135                     }
136                     for (var i = 0; i < data.length; i++) {
137                         v = data[i];
138                         autoCompleteMenu.addChild(new Menu.Item(S.mix({
139                             prefixCls:self.get("prefixCls") + "autocomplete-",
140                             content:v,
141                             textContent:v,
142                             value:v
143                         }, contents[i])))
144                     }
145                     self._showMenu();
146                 } else {
147                     autoCompleteMenu.hide();
148                 }
149             },
150 
151             _showMenu:function () {
152                 var self = this;
153                 var menu = self.get("menu");
154                 menu._input = self;
155                 menu._clearDismissTimer();
156                 alignImmediately(self);
157                 menu.show();
158                 // make menu item (which textContent is same as input) active
159                 var children = menu.get("children"),
160                     val = self._getValue();
161                 for (var i = 0; i < children.length; i++) {
162                     if (children[i].get("textContent") == val) {
163                         menu.set("highlightedItem", children[i]);
164                         return;
165                     }
166                 }
167                 // Whether or not the first row should be highlighted by default.
168                 if (self.get("autoHighlightFirst")) {
169                     for (i = 0; i < children.length; i++) {
170                         if (!children[i].get("disabled")) {
171                             menu.set("highlightedItem", children[i]);
172                             break;
173                         }
174                     }
175                 }
176             },
177 
178             _onWindowResize:function () {
179                 alignImmediately(this);
180             },
181 
182             handleKeyEventInternal:function (e) {
183                 var self = this,
184                     el = self.get("el"),
185                     autoCompleteMenu = self.get("menu");
186                 if (!autoCompleteMenu) {
187                     return;
188                 }
189                 var updateInputOnDownUp = self.get("updateInputOnDownUp");
190 
191                 if (updateInputOnDownUp) {
192                     // autocomplete will change input value
193                     // but it does not need to reload data
194                     if (S.inArray(e.keyCode, [
195                         KeyCodes.UP,
196                         KeyCodes.DOWN,
197                         KeyCodes.ESC
198                     ])) {
199                         self._stopNotify = 1;
200                     } else {
201                         self._stopNotify = 0;
202                     }
203                 }
204                 var activeItem;
205                 if (autoCompleteMenu.get("visible")) {
206                     var handledByMenu = autoCompleteMenu.handleKeydown(e);
207 
208                     if (updateInputOnDownUp) {
209                         if (S.inArray(e.keyCode, [KeyCodes.DOWN, KeyCodes.UP])) {
210                             // update menu's active value to input just for show
211                             self._setValue(autoCompleteMenu.get("activeItem").get("textContent"));
212                         }
213                     }
214                     // esc
215                     if (e.keyCode == KeyCodes.ESC) {
216                         autoCompleteMenu.hide();
217                         if (updateInputOnDownUp) {
218                             // restore original user's input text
219                             self._setValue(self._savedInputValue);
220                         }
221                         return true;
222                     }
223 
224                     // tab
225                     // if menu is open and an menuitem is highlighted, see as click/enter
226                     if (e.keyCode == KeyCodes.TAB) {
227                         if (activeItem = autoCompleteMenu.get("activeItem")) {
228                             activeItem.performActionInternal();
229                             // only prevent focus change in multiple mode
230                             if (self.get("multiple")) {
231                                 return true;
232                             }
233                         }
234                     }
235                     return handledByMenu;
236                 } else if ((e.keyCode == KeyCodes.DOWN || e.keyCode == KeyCodes.UP) &&
237                     self.get("reFetchOnDownUp")) {
238                     // re-fetch , consider multiple input
239                     S.log("refetch : " + self._getValue());
240                     self.sendRequest(self._getValue());
241                     return true;
242                 }
243             },
244 
245             _uiSetSelectedItem:function (item) {
246                 var self = this;
247                 if (item) {
248                     var textContent = item.get("textContent");
249                     self._setValue(textContent + self.get("strAppendedOnComplete"));
250                     self._savedInputValue = textContent;
251                     /**
252                      * @name AutoComplete#select
253                      * @description fired when user select from suggestion list
254                      * @event
255                      * @param e
256                      * @param e.value value of selected menuItem
257                      * @param e.content content of selected menuItem
258                      * @param e.input current active input
259                      */
260                     self.fire("select", {
261                         value:item.get("value"),
262                         content:item.get("content"),
263                         textContent:textContent
264                     });
265                 }
266             },

268             /**
269              * Consider multiple mode , get token at current cursor position
270              */
271             _getValue:function () {
272                 var self = this,
273                     el = self.get("el"),
274                     inputVal = el.val();
275                 if (self.get("multiple")) {
276                     var inputDesc = self._getInputDesc();
277                     var tokens = inputDesc.tokens, tokenIndex = inputDesc.tokenIndex;
278                     var separator = self.get("separator");
279                     var token = tokens[tokenIndex] || "";
280                     // only if token starts with separator , then token has meaning!
281                     // token can not be empty
282                     if (token && separator.indexOf(token.charAt(0)) != -1) {
283                         // remove separator
284                         return token.substring(1);
285                     }
286                     if (
287                         self.get("autoCompleteOnInitial") &&
288                             // cursor is at the beginning of textarea
289                             // whether to trigger autoComplete
290                             (tokenIndex == 0 || tokenIndex == -1)
291                         ) {
292                         return token;
293                     }
294                     return undefined;
295                 } else {
296                     return inputVal;
297                 }
298             },
299 
300             _setValue:function (value) {
301                 var self = this;
302                 var el = self.get("el");
303                 if (self.get("multiple")) {
304                     var inputDesc = self._getInputDesc();
305                     var tokens = inputDesc.tokens, tokenIndex = inputDesc.tokenIndex;
306                     var separator = self.get("separator");
307                     var cursorPosition;
308                     var appendSeparatorOnComplete = self.get("appendSeparatorOnComplete");
309 
310                     var token = tokens[tokenIndex];
311 
312                     if (separator.indexOf(token.charAt(0)) != -1) {
313                         tokens[tokenIndex] = token.charAt(0);
314                     } else {
315                         tokens[tokenIndex] = "";
316                     }
317 
318                     tokens[tokenIndex] += value;
319 
320                     var nextToken = tokens[tokenIndex + 1];
321 
322                     // appendSeparatorOnComplete if next token does not start with separator
323                     if (appendSeparatorOnComplete &&
324                         (!nextToken || separator.indexOf(nextToken.charAt(0)) == -1 )) {
325                         tokens[tokenIndex] += separator.charAt(0);
326                     }
327 
328                     cursorPosition = tokens.slice(0, tokenIndex + 1).join("").length;
329 
330                     el.val(tokens.join(""));
331 
332                     el.prop("selectionStart", cursorPosition);
333                     el.prop("selectionEnd", cursorPosition);
334                 } else {
335                     el.val(value);
336                 }
337             },
338 
339             _getInputDesc:function () {
340                 var self = this,
341                     el = self.get("el"),
342                     inputVal = el.val(),
343                     tokens = [],
344                     cache = [],
345                     literal = self.get("literal"),
346                     separator = self.get("separator"),
347                     inLiteral = false,
348                     whitespace = self.get("whitespace"),
349                     cursorPosition = el.prop('selectionStart'),
350                     tokenIndex = -1;
351 
352                 for (var i = 0; i < inputVal.length; i++) {
353                     var c = inputVal.charAt(i);
354                     if (i == cursorPosition) {
355                         // current token index
356                         tokenIndex = tokens.length;
357                     }
358                     if (!inLiteral) {
359                         // whitespace is not part of token value
360                         // then separate
361                         if (!whitespace && /\s|\xa0/.test(c)) {
362                             tokens.push(cache.join(""));
363                             cache = [];
364                         }
365 
366                         if (separator.indexOf(c) != -1) {
367                             tokens.push(cache.join(""));
368                             cache = [];
369                         }
370                     }
371                     if (literal) {
372                         if (c == literal) {
373                             inLiteral = !inLiteral;
374                         }
375                     }
376                     cache.push(c);
377                 }
378 
379                 if (cache.length) {
380                     tokens.push(cache.join(""));
381                 }
382                 if (tokenIndex == -1) {
383                     tokenIndex = tokens.length - 1;
384                 }
385                 return {
386                     tokens:tokens,
387                     cursorPosition:cursorPosition,
388                     tokenIndex:tokenIndex
389                 };
390             },

393                 var self = this,
394                     autoCompleteMenu = self.get("menu");
395                 autoCompleteMenu.detachInput(self, self.get("destroyMenu"));
396                 self.__set("menu", null);
