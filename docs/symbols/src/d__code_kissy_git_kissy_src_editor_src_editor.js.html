550             /**
551              * Insert a element into current editor.
552              * @param {NodeList} element
553              */
554             insertElement:function (element) {
555 
556                 var self = this;
557 
558                 if (self.get("mode") !== WYSIWYG_MODE) {
559                     return;
560                 }
561 
562                 self.focus();
563 
564                 var clone,
565                     elementName = element['nodeName'](),
566                     xhtml_dtd = Editor.XHTML_DTD,
567                     isBlock = xhtml_dtd['$block'][ elementName ],
568                     KER = Editor.RANGE,
569                     selection = self.getSelection(),
570                     ranges = selection && selection.getRanges(),
571                     range,
572                     notWhitespaceEval,
573                     i,
574                     next,
575                     nextName,
576                     lastElement;
577 
578                 if (!ranges || ranges.length == 0) {
579                     return;
580                 }
581 
582                 self.execCommand("save");
583 
584                 for (i = ranges.length - 1; i >= 0; i--) {
585                     range = ranges[ i ];
586                     // Remove the original contents.
587 
588                     clone = !i && element || element['clone'](TRUE);
589                     range.insertNodeByDtd(clone);
590                     // Save the last element reference so we can make the
591                     // selection later.
592                     if (!lastElement) {
593                         lastElement = clone;
594                     }
595                 }
596 
597                 if (!lastElement) {
598                     return;
599                 }
600 
601                 range.moveToPosition(lastElement, KER.POSITION_AFTER_END);
602                 // If we're inserting a block element immediately followed by
603                 // another block element, the selection must move there. (#3100,#5436)
604                 if (isBlock) {
605                     notWhitespaceEval = Editor.Walker.whitespaces(true);
606                     next = lastElement.next(notWhitespaceEval, 1);
607                     nextName = next && next[0].nodeType == DOM.ELEMENT_NODE
608                         && next.nodeName();
609                     // Check if it's a block element that accepts text.
610                     if (nextName &&
611                         xhtml_dtd.$block[ nextName ] &&
612                         xhtml_dtd[ nextName ]['#text']) {
613                         range.moveToElementEditablePosition(next);
614                     }
615                 }
616                 selection.selectRanges([ range ]);
617                 self.focus();
618                 // http://code.google.com/p/kissy/issues/detail?can=1&start=100&id=121
619                 // only tag can scroll
620                 if (clone && clone[0].nodeType == 1) {
621                     clone.scrollIntoView(undefined, false);
622                 }
623                 saveLater.call(self);
624                 return clone;
625             },

627             /**
628              * Insert html string into current editor.
629              * @param data {string}
630              */
631             insertHtml:function (data, dataFilter) {
632                 var self = this,
633                     htmlDataProcessor,
634                     editorDoc = self.get("document")[0];
635 
636                 if (self.get("mode") !== WYSIWYG_MODE) {
637                     return;
638                 }
639 
640                 if (htmlDataProcessor = self.htmlDataProcessor) {
641                     data = htmlDataProcessor.toDataFormat(data, dataFilter);
642                 }
643 
644                 self.focus();
645                 self.execCommand("save");
646 
647                 // ie9 仍然需要这样！
648                 // ie9 标准 selection 有问题，连续插入不能定位光标到插入内容后面
649                 if (IS_IE) {
650                     var $sel = editorDoc.selection;
651                     if ($sel.type == 'Control') {
652                         $sel.clear();
653                     }
654                     try {
655                         $sel.createRange().pasteHTML(data);
656                     } catch (e) {
657                         S.log("insertHtml error in ie");
658                     }
659                 } else {
660                     // ie9 仍然没有
661                     // 1.webkit insert html 有问题！会把标签去掉，算了直接用 insertElement.
662                     // 10.0 修复？？
663                     // firefox 初始编辑器无焦点报异常
664                     try {
665                         editorDoc.execCommand('inserthtml', FALSE, data);
666                     } catch (e) {
667                         setTimeout(function () {
668                             // still not ok in ff!
669                             // 手动选择 body 的第一个节点
670                             if (self.getSelection().getRanges().length == 0) {
671                                 var r = new Editor.Range(editorDoc),
672                                     node = DOM.first(editorDoc.body, function (el) {
673                                         return el.nodeType == 1 && DOM.nodeName(el) != "br";
674                                     });
675                                 if (!node) {
676                                     node = new Node(editorDoc.createElement("p"));
677                                     node._4e_appendBogus(undefined);
678                                     editorDoc.body.appendChild(node[0]);
679                                 }
680                                 r.setStartAt(node, Editor.RANGE.POSITION_AFTER_START);
681                                 r.select();
682                             }
683                             editorDoc.execCommand('inserthtml', FALSE, data);
684                         }, 50);
685                     }
686                 }
687                 // bug by zjw2004112@163.com :
688                 // 有的浏览器 ： chrome , ie67 貌似不会自动滚动到粘贴后的位置
689                 setTimeout(function () {
690                     self.getSelection().scrollIntoView();
691                 }, 50);
692                 saveLater.call(self);
693             }
694         });

696     /**
697      * 初始化iframe内容以及浏览器间兼容性处理，
698      * 必须等待iframe内的脚本向父窗口通知
699      *
700      * @param id {string}
701      */
702     Editor["_initIFrame"] = function (id) {
703 
704         var self = focusManager.getInstance(id),
705             doc = self.get("document")[0],
706         // Remove bootstrap script from the DOM.
707             script = doc.getElementById("ke_active_script");
708 
709         DOM.remove(script);
710 
711         fixByBindIframeDoc(self);
712 
713         var body = doc.body;
714 
715         /**
716          * from kissy editor 1.0
717          *
718          * // 注1：在 tinymce 里，designMode = "on" 放在 try catch 里。
719          //     原因是在 firefox 下，当iframe 在 display: none 的容器里，会导致错误。
720          //     但经过我测试，firefox 3+ 以上已无此现象。
721          // 注2： ie 用 contentEditable = true.
722          //     原因是在 ie 下，IE needs to use contentEditable or
723          // it will display non secure items for HTTPS
724          // Ref:
725          //   - Differences between designMode and contentEditable
726          //     http://74.125.153.132/search?q=cache:5LveNs1yHyMJ:nagoon97.wordpress.com/2008/04/20/differences-between-designmode-and-contenteditable/+ie+contentEditable+designMode+different&cd=6&hl=en&ct=clnk
727          */
728 
729         // 这里对主流浏览器全部使用 contenteditable
730         // 那么不同于 kissy editor 1.0
731         // 在body范围外右键，不会出现 复制，粘贴等菜单
732         // 因为这时右键作用在document而不是body
733         // 1.0 document.designMode='on' 是编辑模式
734         // 2.0 body.contentEditable=true body外不是编辑模式
735         if (IS_IE) {
736             // Don't display the focus border.
737             body['hideFocus'] = TRUE;
738             // Disable and re-enable the body to avoid IE from
739             // taking the editing focus at startup. (#141 / #523)
740             body.disabled = TRUE;
741             body['contentEditable'] = TRUE;
742             body.removeAttribute('disabled');
743         } else {
744             // Avoid opening design mode in a frame window thread,
745             // which will cause host page scrolling.(#4397)
746             setTimeout(function () {
747                 // Prefer 'contentEditable' instead of 'designMode'. (#3593)
748                 if (UA['gecko'] || UA['opera']) {
749                     body['contentEditable'] = TRUE;
750                 }
751                 else if (UA['webkit'])

857     function blinkCursor(doc, retry) {
858         var body = doc.body;
859         tryThese(
860             function () {
861                 doc['designMode'] = 'on';
862                 //异步引起时序问题，尽可能小间隔
863                 setTimeout(function () {
864                     doc['designMode'] = 'off';
865                     body.focus();
866                     // Try it again once..
867                     if (!arguments.callee.retry) {
868                         arguments.callee.retry = TRUE;
869                         //arguments.callee();
870                     }
871                 }, 50);
872             },
873             function () {
874                 // The above call is known to fail when parent DOM
875                 // tree layout changes may break design mode. (#5782)
876                 // Refresh the 'contentEditable' is a cue to this.
877                 doc['designMode'] = 'off';
878                 DOM.attr(body, 'contentEditable', FALSE);
879                 DOM.attr(body, 'contentEditable', TRUE);
880                 // Try it again once..
881                 !retry && blinkCursor(doc, 1);
882             }
883         );
884     }

886     function fixByBindIframeDoc(self) {
887         var iframe = self.get("iframe"),
888             textarea = self.get("textarea")[0],
889             win = self.get("window")[0],
890             doc = self.get("document")[0];
891 
892         // Gecko need a key event to 'wake up' the editing
893         // ability when document is empty.(#3864)
894         // activateEditing 删掉，初始引起屏幕滚动了
895         // Webkit: avoid from editing form control elements content.
896         if (UA['webkit']) {
897             Event.on(doc, "click", function (ev) {
898                 var control = new Node(ev.target);
899                 if (S.inArray(control.nodeName(), ['input', 'select'])) {
900                     ev.preventDefault();
901                 }
902             });
903             // Prevent from editig textfield/textarea value.
904             Event.on(doc, "mouseup", function (ev) {
905                 var control = new Node(ev.target);
906                 if (S.inArray(control.nodeName(), ['input', 'textarea'])) {
907                     ev.preventDefault();
908                 }
909             });
910         }
911 
912 
913         // Create an invisible element to grab focus.
914         if (UA['gecko'] || IS_IE || UA['opera']) {
915             var focusGrabber;
916             focusGrabber = new Node(
917                 // Use 'span' instead of anything else to fly under the screen-reader radar. (#5049)
918                 '<span ' +
919                     'tabindex="-1" ' +
920                     'style="position:absolute; left:-10000"' +
921                     ' role="presentation"' +
922                     '></span>').insertAfter(textarea);
923             focusGrabber.on('focus', function () {
924                 self.focus();
925             });
926             self.activateGecko = function () {
927                 if (UA['gecko'] && self.__iframeFocus)
928                     focusGrabber[0].focus();
929             };
930             self.on('destroy', function () {
931                 focusGrabber.detach();
932                 focusGrabber.remove();
933             });
934         }
935 
936 
937         Event.on(win, 'focus', function () {
938             /**
939              * yiminghe特别注意：firefox光标丢失bug
940              * blink后光标出现在最后，这就需要实现保存range
941              * focus后再恢复range
942              */
943             if (UA['gecko']) {
944                 blinkCursor(doc, FALSE);
945             }
946             else if (UA['opera']) {
947                 doc.body.focus();
948             }
949             // focus 后强制刷新自己状态
950             self.notifySelectionChange();
951         });
952 
953 
954         if (UA['gecko']) {
955             /**
956              * firefox 焦点丢失后，再点编辑器区域焦点会移不过来，要点两下
957              */
958             Event.on(doc, "mousedown", function () {
959                 if (!self.__iframeFocus) {
960                     blinkCursor(doc, FALSE);
961                 }
962             });
963         }
964 
965         if (IS_IE) {
966             // Override keystrokes which should have deletion behavior
967             // on control types in IE . (#4047)
968             /**
969              * 选择img，出现缩放框后不能直接删除
970              */
971             Event.on(doc, 'keydown', function (evt) {
972                 var keyCode = evt.keyCode;
973                 // Backspace OR Delete.
974                 if (keyCode in { 8:1, 46:1 }) {
975                     var sel = self.getSelection(),
976                         control = sel.getSelectedElement();
977                     if (control) {
978                         // Make undo snapshot.
979                         self.execCommand('save');
980                         // Delete any element that 'hasLayout' (e.g. hr,table) in IE8 will
981                         // break up the selection, safely manage it here. (#4795)
982                         var bookmark = sel.getRanges()[ 0 ].createBookmark();
983                         // Remove the control manually.
984                         control.remove();
985                         sel.selectBookmarks([ bookmark ]);
986                         self.execCommand('save');
987                         evt.preventDefault();
988                     }
989                 }
990             });
991 
992             // PageUp/PageDown scrolling is broken in document
993             // with standard doctype, manually fix it. (#4736)
994             // ie8 主窗口滚动？？
995             if (doc.compatMode == 'CSS1Compat') {
996                 var pageUpDownKeys = { 33:1, 34:1 };
997                 Event.on(doc, 'keydown', function (evt) {
998                     if (evt.keyCode in pageUpDownKeys) {
999                         setTimeout(function () {
1000                             self.getSelection().scrollIntoView();
1001                         }, 0);
1002                     }
1003                 });
1004             }
1005         }
1006 
1007         // Gecko/Webkit need some help when selecting control type elements. (#3448)
1008         if (UA['webkit']) {
1009             Event.on(doc, "mousedown", function (ev) {
1010                 var control = new Node(ev.target);
1011                 if (S.inArray(control.nodeName(), ['img', 'hr', 'input', 'textarea', 'select'])) {
1012                     self.getSelection().selectElement(control);
1013                 }
1014             });
1015         }
1016 
1017 
1018         if (UA['gecko']) {
1019             Event.on(doc, "dragstart", function (ev) {
1020                 var control = new Node(ev.target);
1021                 if (control.nodeName() === 'img' && /ke_/.test(control[0].className)) {
1022                     // firefox禁止拖放
1023                     ev.preventDefault();
1024                 }
1025             });
1026         }
1027         //注意：必须放在这个位置，等iframe加载好再开始运行
1028         //加入焦点管理，和其他实例联系起来
1029         focusManager.add(self);
1030 
1031     }

1033     function prepareIFrameHtml(id, customStyle, customLink, data) {
1034         var links = "",
1035             i,
1036             innerCssFile = Utils.debugUrl("theme/editor-iframe.css");
1037 
1038         for (i = 0; i < customLink.length; i++) {
1039             links += S.substitute('<link href="' + '{href}" rel="stylesheet" />', {
1040                 href:customLink[i]
1041             });
1042         }
1043 
1044         return S.substitute(IFRAME_HTML_TPL, {
1045             // kissy-editor #12
1046             // IE8 doesn't support carets behind images(empty content after image's block)
1047             // setting ie7 compatible mode would force IE8+ to run in IE7 compat mode.
1048             doctype:DOC.documentMode === 8 ?
1049                 '<meta http-equiv="X-UA-Compatible" content="IE=7" />' :
1050                 "",
1051             title:"${title}",
1052             href:innerCssFile,
1053             style:customStyle,
1054             // firefox 必须里面有东西，否则编辑前不能删除!
1055             data:data || " ",
1056             script:id ?
1057                 // The script that launches the bootstrap logic on 'domReady', so the document
1058                 // is fully editable even before the editing iframe is fully loaded (#4455).
1059                 // 确保iframe确实载入成功,过早的话 document.domain 会出现无法访问
1060                 ('<script id="ke_active_script">' +
1061                     ( DOM.isCustomDomain() ? ( 'document.domain="' + DOC.domain + '";' ) : '' ) +
1062                     'parent.KISSY.Editor._initIFrame("' + id + '");' +
1063                     '</script>') :
1064                 ''
1065

1073     function setUpIFrame(self, data) {
1074         var iframe = self.get("iframe"),
1075             html = prepareIFrameHtml(self._UUID,
1076                 self.get('customStyle'),
1077                 self.get('customLink'), data),
1078             iframeDom = iframe[0],
1079             win = iframeDom.contentWindow,
1080             doc;
1081         iframe.__loaded = 1;
1082         try {
1083             // In IE, with custom document.domain, it may happen that
1084             // the iframe is not yet available, resulting in "Access
1085             // Denied" for the following property access.
1086             //ie 设置domain 有问题：yui也有
1087             //http://yuilibrary.com/projects/yui2/ticket/2052000
1088             //http://waelchatila.com/2007/10/31/1193851500000.html
1089             //http://nagoon97.wordpress.com/tag/designmode/
1090             doc = win.document;
1091         } catch (e) {
1092             // Trick to solve this issue, forcing the iframe to get ready
1093             // by simply setting its "src" property.
1094             //noinspection SillyAssignmentJS
1095             iframeDom.src = iframeDom.src;
1096             // In IE6 though, the above is not enough, so we must pause the
1097             // execution for a while, giving it time to think.
1098             if (IS_IE < 7) {
1099                 setTimeout(run, 10);
1100                 return;
1101             }
1102         }
1103         run();
1104         function run() {
1105             doc = win.document;
1106             self.__set("document", new Node(doc));
1107             self.__set("window", new Node(win));
1108             iframe.detach();
1109             // Don't leave any history log in IE. (#5657)
1110             doc['open']("text/html", "replace");
1111             doc.write(html);
1112             doc.close();
1113         }
1114     }

1116     function createIframe(self, afterData) {
1117         var iframe = new Node(IFRAME_TPL),
1118             textarea = self.get("textarea");
1119         if (textarea.hasAttr("tabindex")) {
1120             iframe.attr("tabIndex", UA['webkit'] ? -1 : textarea.attr("tabIndex"));
1121         }
1122         textarea.parent().prepend(iframe);
1123         self.set("iframe", iframe);
1124         self.__docReady = 0;
1125         // With FF, it's better to load the data on iframe.load. (#3894,#4058)
1126         if (UA['gecko'] && !iframe.__loaded) {
1127             iframe.on('load', function () {
1128                 setUpIFrame(self, afterData);
1129             }, self);
1130         } else {
1131             // webkit(chrome) load等不来！
1132             setUpIFrame(self, afterData);
