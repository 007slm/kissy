1 /**
  2  * @fileOverview  a specialized pagingbar which use number buttons
  3  * @author dxq613@gmail.com, yiminghe@gmail.com
  4  */
  5 KISSY.add('grid/numberpagingbar', function (S,Component,PBar,Bar) {
  6 
  7 	var NUMBER_CONTAINER = 'numberContainer',
  8 	    CLS_NUMBER_BUTTON = 'ks-grid-button-number',
  9 		CLS_ACTIVE = 'ks-active';
 10 
 11 	/**
 12 	* specialized paging bar auto show numberic buttons
 13 	* Paging Toolbar is typically used as one of the Grid's toolbars.
 14 	* @name Number
 15     * @constructor
 16     * @extends Grid.PagingBar
 17     * @memberOf Grid.PagingBar
 18 	*/
 19 	var numberPagingBar = PBar.extend({
 20 		/**
 21 		* get the initial items of paging bar
 22 		* @protected
 23 		* @override
 24 		*/
 25 		_getItems : function(){
 26 			var _self = this,
 27 				items = _self.get('items'),
 28 				numberContainerBar;
 29 			if(items)
 30 			{
 31 				return items;
 32 			}
 33 			//default items
 34 			items = [];
 35 
 36 			//previous item
 37 			items.push(_self._getButtonItem(PBar.ID_PREV));
 38 
 39 			//the container of number buttons 
 40 			numberContainerBar = new Bar({id : NUMBER_CONTAINER,elCls:'ks-inline-block'});
 41 			items.push(numberContainerBar);
 42 			_self.set(NUMBER_CONTAINER,numberContainerBar);
 43 			//next item
 44 			items.push(_self._getButtonItem(PBar.ID_NEXT));
 45 			//total page of store
 46 			items.push(_self._getTextItem(PBar.ID_TOTAL_PAGE));
 47 			//current page of store
 48 			items.push(_self._getTextItem(PBar.ID_CURRENT_PAGE));
 49 			//button for skip to
 50 			items.push(_self._getButtonItem(PBar.ID_SKIP));
 51 			return items;
 52 		},
 53 		/**
 54 		* bind buttons event
 55 		* @protected
 56 		* @override
 57 		*/
 58 		_bindButtonEvent : function(){
 59 			var _self = this,
 60 				numberContainerBar = _self.get(NUMBER_CONTAINER);
 61 			_self.constructor.superclass._bindButtonEvent.call(this);
 62 			numberContainerBar.get('el').delegate('click','.' + CLS_NUMBER_BUTTON,function(event){
 63 				var btn = S.one(event.target),
 64 					page = parseInt(btn.text(),10);
 65 				_self.jumpToPage(page);
 66 			});
 67 		},
 68 		//设置页码信息，设置 页数 按钮
 69 		_setNumberPages : function(){
 70 			var _self = this;
 71 			_self.constructor.superclass._setNumberPages.call(_self);
 72 			_self._setNumberButtons();
 73 
 74 		},
 75 		//设置 页数 按钮
 76 		_setNumberButtons : function(){
 77 			var _self = this,
 78 				numberContainerBar = _self.get(NUMBER_CONTAINER),
 79 				curPage = _self.get('curPage'),
 80 				totalPage = _self.get('totalPage'),
 81 				numberItems = _self._getNumberItems(curPage,totalPage),
 82 				curItem;
 83 			numberContainerBar.removeChildren(true);
 84 
 85 			S.each(numberItems,function(item){
 86 				numberContainerBar.addChild(item);
 87 			});
 88 			curItem = numberContainerBar.getItem(curPage);
 89 			if(curItem){
 90 				//curItem.get('el').addClass(CLS_ACTIVE);
 91 				curItem.set('checked',true);
 92 			}
 93 		},
 94 		//获取所有页码按钮的配置项
 95 		_getNumberItems : function(curPage, totalPage){
 96 			var _self = this,
 97 				result = [],
 98 				maxLimitCount = _self.get('maxLimitCount'),
 99 				showRangeCount = _self.get('showRangeCount'),
100 				maxPage;
101 
102 			function addNumberItem(from,to){
103 				for(var i = from ;i<=to;i++){
104 					result.push(_self._getNumberItem(i));
105 				}
106 			}
107 
108 			function addEllipsis(){
109 				result.push(_self._getEllipsisItem());
110 			}
111 
112 			if(totalPage < maxLimitCount){
113 				maxPage = totalPage;
114 				addNumberItem(1,totalPage);
115 			}else{
116 				var startNum = (curPage <= maxLimitCount) ? 1 : (curPage - showRangeCount),
117                     lastLimit = curPage + showRangeCount,
118                     endNum = lastLimit < totalPage ? (lastLimit > maxLimitCount ? lastLimit : maxLimitCount) : totalPage;
119                 if (startNum > 1) {
120                     addNumberItem(1, 1);
121 					if(startNum > 2){
122 						addEllipsis();
123 					}
124                 }
125                 maxPage = endNum;
126                 addNumberItem(startNum, endNum);
127 			}
128 
129 			if (maxPage < totalPage) {
130 				if(maxPage < totalPage -1){
131 					addEllipsis();
132 				}
133                 addNumberItem(totalPage, totalPage);
134             }
135 
136 			return result;
137 		},
138 		//获取省略号
139 		_getEllipsisItem : function(){
140 			var _self = this;
141 			return {
142 				xclass:'grid-bar-item-text',
143 				text : _self.get('ellipsisTpl')
144 			};
145 		},
146 		//生成页面按钮配置项
147 		_getNumberItem : function(page){
148 			var _self = this;
149 			return {
150 				id : page,
151 				xclass:'grid-bar-item-button',
152 				text : ''+page+'',
153 				elCls : _self.get('numberButtonCls')
154 			};
155 		}
156 		
157 	},{
158 		ATTRS:{
159 			/**
160 			* the text for skip page button
161 			* @override
162 			* @default {String} "确定"
163 			*/
164 			skipText : {
165 				value :'确定'
166 			},
167 			/**
168 			* if the number of page is smaller then this value,show all number buttons,else show ellipsis button
169 			* @default {Number} 4
170 			*/
171 			maxLimitCount : {
172 				value : 4
173 			},
174 			/**
175 			* 
176 			*/
177 			showRangeCount : {
178 				value : 1	
179 			},
180 			/**
181 			* the css used on number button
182 			*/
183 			numberButtonCls:{
184 				value : CLS_NUMBER_BUTTON
185 			},
186 			/**
187 			* the template of ellipsis which represent the omitted pages number
188 			*/
189 			ellipsisTpl : {
190 				value : '...'
191 			},
192 			/**
193 			* the template of current page info
194 			* @override
195 			* @default {String} '到第 <input type="text" autocomplete="off" class="ks-pb-page" size="20" name="inputItem"> 页'
196 			*/
197 			curPageTpl : {
198 				value : '到第 <input type="text" '+
199                     'auto'+
200                     'complete="off" class="ks-pb-page" size="20" name="inputItem"> 页'
201 			}
202 		}
203 	},{
204 		xclass : 'grid-pagingbar-number',
205 		priority : 3	
206 	});
207 
208 
209 	PBar.Number = numberPagingBar;
210 	return numberPagingBar;
211 },{
212 	 requires:['component','./pagingbar','./bar']
213 });
