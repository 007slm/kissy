 24     //定义通用的函数交换两个结点的位置
 25     function swapNode(node1, node2) {
 26         //获取父结点
 27         var _parent = node1.parentNode;
 28         //获取两个结点的相对位置
 29         var _t1 = node1.nextSibling;
 30         var _t2 = node2.nextSibling;
 31         //将node2插入到原来node1的位置
 32         _parent.insertBefore(node2, _t1);
 33         //将node1插入到原来node2的位置
 34         _parent.insertBefore(node1, _t2);
 35     }
