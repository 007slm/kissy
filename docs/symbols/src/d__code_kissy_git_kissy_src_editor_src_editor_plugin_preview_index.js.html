1 /**
  2  * preview for kissy editor
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/preview/index", function () {
  6     var win = window;
  7     return {
  8         init:function (editor) {
  9             editor.addButton("preview", {
 10                 tooltip:"预览",
 11                 listeners:{
 12                     click:function () {
 13                             try {
 14                                 var screen = win.screen,
 15                                     iWidth = Math.round(screen.width * 0.8),
 16                                     iHeight = Math.round(screen.height * 0.7),
 17                                     iLeft = Math.round(screen.width * 0.1);
 18                             } catch (e) {
 19                                 iWidth = 640; // 800 * 0.8,
 20                                 iHeight = 420; // 600 * 0.7,
 21                                 iLeft = 80;	// (800 - 0.8 * 800) /2 = 800 * 0.1.
 22                             }
 23                             var sHTML = editor.getDocHtml()
 24                                     .replace(/\${title}/, "预览"),
 25                                 sOpenUrl = '',
 26                                 oWindow = win.open(sOpenUrl,
 27                                     // 每次都弹出新窗口
 28                                     '',
 29                                     'toolbar=yes,' +
 30                                         'location=no,' +
 31                                         'status=yes,' +
 32                                         'menubar=yes,' +
 33                                         'scrollbars=yes,' +
 34                                         'resizable=yes,' +
 35                                         'width=' +
 36                                         iWidth +
 37                                         ',height='
 38                                         + iHeight
 39                                         + ',left='
 40                                         + iLeft), winDoc = oWindow.document;
 41                             winDoc.open();
 42                             winDoc.write(sHTML);
 43                             winDoc.close();
 44                             //ie 重新显示
 45                             oWindow.focus();
 46                         }
 47 
 48                 }
 49             });
 50         }};
 51 });
