1 /**
  2  * @fileOverview A collection of commonly used function buttons or controls represented in compact visual form.
  3  * @author dxq613@gmail.com, yiminghe@gmail.com
  4  */
  5 KISSY.add("grid/bar", function (S,Toolbar,BarRender,BarItem) {
  6 
  7 	/**
  8 	 * This class specifies the definition for a toolbar. 
  9      * Bar class is a collection of buttons,links and other command control.
 10      * @name Bar
 11      * @constructor
 12      * @extends Toolbar
 13      * @memberOf Grid
 14      */
 15 	var Bar = Toolbar.extend(
 16 	 /**
 17 	 * @lends Grid.Bar.prototype
 18 	 */	
 19 	{
 20 
 21 		/**
 22 		* get bar item by id
 23 		* @param {String|Number} id the id of item 
 24 		* @return {BarItem}
 25 		*/
 26 		getItem : function(id){
 27 			var _self = this,
 28 				children = _self.get('children'),
 29 				result = null;
 30 			S.each(children,function(child){
 31 				if(child.get('id') === id){
 32 					result = child;
 33 					return false;
 34 				}
 35 			});
 36 			return result;
 37 		}
 38 	},{
 39 		ATTRS:/** @lends Grid.Bar.prototype*/
 40 		{
 41 			/**
 42 			* @protected
 43 			*/
 44 			focusable : {
 45 				value : false
 46 			},
 47 			/**
 48 			* @private
 49 			*/
 50 			xrender : {
 51 				value : BarRender	
 52 			}
 53 		},
 54 		BarItem : BarItem
 55 	},{
 56 		xclass : 'grid-bar',
 57 		priority : 1	
 58 	});
 59 
 60 	return Bar;
 61 }, {
 62     requires:['toolbar','./barrender', './baritem']
 63 });
