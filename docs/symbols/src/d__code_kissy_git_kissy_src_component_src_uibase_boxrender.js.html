1 /**
  2  * @fileOverview UIBase.Box
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add('component/uibase/boxrender', function (S) {
  6 
  7     var $ = S.all, doc = S.Env.host.document;
  8 
  9     function BoxRender() {
 10     }
 11 
 12     BoxRender.ATTRS = {
 13         el:{
 14             //容器元素
 15             setter:function (v) {
 16                 return $(v);
 17             }
 18         },
 19 
 20         // 构建时批量生成，不需要执行单个
 21         elCls:{
 22         },
 23 
 24         elStyle:{
 25         },
 26 
 27         width:{
 28         },
 29 
 30         height:{
 31         },
 32 
 33         elTagName:{
 34             // 生成标签名字
 35             value:"div"
 36         },
 37 
 38         elAttrs:{
 39         },
 40 
 41         content:{
 42         },
 43 
 44         elBefore:{
 45             // better named to renderBefore, too late !
 46         },
 47 
 48         render:{},
 49 
 50         visible:{
 51             value:true
 52         },
 53 
 54         visibleMode:{
 55             value:"display"
 56         },
 57         // content 设置的内容节点,默认根节点
 58         contentEl:{
 59             valueFn:function () {
 60                 return this.get("el");
 61             }
 62         }
 63     };
 64 
 65     BoxRender.HTML_PARSER = {
 66         el:function (srcNode) {
 67             return srcNode;
 68         },
 69         content:function (el) {
 70             // 从 contentElCls 的标志中读取
 71             var contentElCls = this.get("contentElCls");
 72             return (contentElCls ? el.one("." + contentElCls) : el).html();
 73         }
 74     };
 75 
 76     BoxRender.prototype =
 77     /**
 78      * @lends Component.UIBase.Box.Render#
 79      */
 80     {
 81 
 82         __renderUI:function () {
 83             var self = this;
 84             // 新建的节点才需要摆放定位
 85             if (!self.get("srcNode")) {
 86                 var render = self.get("render"),
 87                     el = self.get("el"),
 88                     renderBefore = self.get("elBefore");
 89                 if (renderBefore) {
 90                     el.insertBefore(renderBefore, undefined);
 91                 } else if (render) {
 92                     el.appendTo(render, undefined);
 93                 } else {
 94                     el.appendTo(doc.body, undefined);
 95                 }
 96             }
 97         },
 98 
 99         /**
100          * 只负责建立节点，如果是 decorate 过来的，甚至内容会丢失
101          * 通过 render 来重建原有的内容
102          */
103         __createDom:function () {
104             var self = this;
105             if (!self.get("srcNode")) {
106                 var el,
107                     contentEl = self.get("contentEl");
108 
109                 el = $("<" + self.get("elTagName") + ">");
110 
111                 if (contentEl) {
112                     el.append(contentEl);
113                 }
114 
115                 self.__set("el", el);
116 
117                 if (!contentEl) {
118                     // 没取到,这里设下值, uiSet 时可以 set("content")  取到
119                     self.__set("contentEl", el);
120                 }
121             }
122         },
123 
124         _uiSetElAttrs:function (attrs) {
125             this.get("el").attr(attrs);
126         },
127 
128         _uiSetElCls:function (cls) {
129             this.get("el").addClass(cls);
130         },
131 
132         _uiSetElStyle:function (style) {
133             this.get("el").css(style);
134         },
135 
136         _uiSetWidth:function (w) {
137             this.get("el").width(w);
138         },
139 
140         _uiSetHeight:function (h) {
141             var self = this;
142             self.get("el").height(h);
143         },
144 
145         _uiSetContent:function (c) {
146             var self = this, el;
147             // srcNode 时不重新渲染 content
148             // 防止内部有改变，而 content 则是老的 html 内容
149             if (self.get("srcNode") && !self.get("rendered")) {
150             } else {
151                 el = self.get("contentEl");
152                 if (typeof c == "string") {
153                     el.html(c);
154                 } else if (c) {
155                     el.empty().append(c);
156                 }
157             }
158         },
159 
160         _uiSetVisible:function (isVisible) {
161             var el = this.get("el"),
162                 visibleMode = this.get("visibleMode");
163             if (visibleMode == "visibility") {
164                 el.css("visibility", isVisible ? "visible" : "hidden");
165             } else {
166                 el.css("display", isVisible ? "" : "none");
167             }
168         },
169 
170         __destructor:function () {
171             var el = this.get("el");
172             if (el) {
173                 el.remove();
174             }
175         }
176     };
177 
178     return BoxRender;
179 }, {
180     requires:['node']
181 });
