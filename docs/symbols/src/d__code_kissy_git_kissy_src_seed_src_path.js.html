1 /**
  2  * Port Node Path Utils For KISSY.
  3  * Note: Only posix mode.
  4  * @author yiminghe@gmail.com
  5  */
  6 (function (S) {
  7 
  8     /**
  9      * @namespace
 10      * @name Path
 11      * @memberOf KISSY
 12      */
 13     var Path = {},
 14     // [root, dir, basename, ext]
 15         splitPathRe = /^(\/?)([\s\S]+\/(?!$)|\/)?((?:\.{1,2}$|[\s\S]+?)?(\.[^.\/]*)?)$/;
 16 
 17     KISSY.Path = Path;
 18 
 19     /**
 20      * Remove .. and . in path array
 21      * @param parts
 22      * @param allowAboveRoot
 23      * @return {*}
 24      */
 25     function normalizeArray(parts, allowAboveRoot) {
 26         // level above root
 27         var up = 0;
 28         for (var i = parts.length - 1; i >= 0; i--) {
 29             var last = parts[i];
 30             if (last == ".") {
 31                 parts.splice(i, 1);
 32             } else if (last === "..") {
 33                 parts.splice(i, 1);
 34                 up++;
 35             } else if (up) {
 36                 parts.splice(i, 1);
 37                 up--;
 38             }
 39         }
 40 
 41         // if allow above root, has to add ..
 42         if (allowAboveRoot) {
 43             for (; up--; up) {
 44                 parts.unshift("..");
 45             }
 46         }
 47 
 48         return parts;
 49     }
 50 
 51     S.mix(Path,
 52         /**
 53          * @lends KISSY.Path
 54          */
 55         {
 56 
 57             /**
 58              * resolve([from ...], to)
 59              * @return {String} Resolved path.
 60              */
 61             resolve:function () {
 62 
 63                 var resolvedPath = [],
 64                     resolvedPathStr,
 65                     i,
 66                     args = S.makeArray(arguments),
 67                     path,
 68                     absolute = 0;
 69 
 70                 for (i = args.length - 1; i >= 0 && !absolute; i--) {
 71                     path = args[i];
 72                     if (typeof path != "string" || !path) {
 73                         continue;
 74                     }
 75                     resolvedPath.unshift(path);
 76                     absolute = path.charAt(0) == "/";
 77                 }
 78 
 79                 resolvedPathStr = normalizeArray(resolvedPath, !absolute).join("/");
 80 
 81                 return (absolute ? "/" : "" + resolvedPathStr) || ".";
 82             },
 83 
 84             /**
 85              * normalize .. and . in path
 86              * @param {String} path Path tobe normalized
 87              * @example
 88              * <code>
 89              * "x/y/../z" => "x/z"
 90              * "x/y/z/../" => "x/y/"
 91              * </code>
 92              * @return {String}
 93              */
 94             normalize:function (path) {
 95                 var absolute = path.charAt(0) == "/",
 96                     trailingSlash = path.slice(-1) == "/";
 97 
 98                 path = normalizeArray(S.filter(path.split("/"), function (p) {
 99                     return !!p;
100                 }), !absolute).join("/");
101 
102                 if (!path && !absolute) {
103                     path = ".";
104                 }
105 
106                 if (path && trailingSlash) {
107                     path += "/";
108                 }
109 
110 
111                 return (absolute ? "/" : "") + path;
112             },
113 
114             /**
115              * join([path ...]) and normalize
116              * @return {String}
117              */
118             join:function () {
119                 var args = S.makeArray(arguments);
120                 return Path.normalize(S.filter(args,function (p) {
121                     return p && (typeof p == "string");
122                 }).join("/"));
123             },
124 
125             /**
126              * Get string which is to relative to from
127              * @param {String} from
128              * @param {String} to
129              * @example
130              * <code>
131              * relative("x/","x/y/z") => "y/z"
132              * relative("x/t/z","x/") => "../../"
133              * </code>
134              * @return {String}
135              */
136             relative:function (from, to) {
137                 from = Path.resolve(from);
138                 to = Path.resolve(to);
139 
140                 var fromParts = S.filter(from.split("/"), function (p) {
141                         return !!p;
142                     }),
143                     path = [],
144                     sameIndex,
145                     sameIndex2,
146                     toParts = S.filter(to.split("/"), function (p) {
147                         return !!p;
148                     }), commonLength = Math.min(fromParts.length, toParts.length);
149 
150                 for (sameIndex = 0; sameIndex < commonLength; sameIndex++) {
151                     if (fromParts[sameIndex] != toParts[sameIndex]) {
152                         break;
153                     }
154                 }
155 
156                 sameIndex2 = sameIndex;
157 
158                 while (sameIndex < fromParts.length) {
159                     path.push("..");
160                     sameIndex++;
161                 }
162 
163                 path = path.concat(toParts.slice(sameIndex2));
164 
165                 path = path.join("/");
166 
167                 return path;
168             },
169 
170             /**
171              * Get base name of path
172              * @param {String} path
173              * @param {String} [ext] ext to be stripped from result returned.
174              * @return {String}
175              */
176             basename:function (path, ext) {
177                 var result = path.match(splitPathRe) || [];
178                 result = result[2];
179                 if (result && result.slice(-1 * ext.length) == ext) {
180                     result = result.slice(0, -1 * ext.length);
181                 }
182                 return result;
183             },
184 
185             /**
186              * Get dirname of path
187              * @return {String}
188              */
189             dirname:function (path) {
190                 var result = path.match(splitPathRe) || [],
191                     root = result[0],
192                     dir = result[1];
193 
194                 if (!root && !dir) {
195                     // No dirname
196                     return '.';
197                 }
198 
199                 if (dir) {
200                     // It has a dirname, strip trailing slash
201                     dir = dir.substring(0, dir.length - 1);
202                 }
203 
204                 return root + dir;
205             },
206 
207             /**
208              * Get extension name of file in path
209              * @param {String} path
210              * @return {String}
211              */
212             extname:function (path) {
213                 return (path.match(splitPathRe) || [])[3];
214             }
215 
216         });
217 
218 })(KISSY);
