1 /**
  2  * triple state button for kissy editor
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/button/index", function (S, Editor, UIBase) {
  6     var ON = "on",
  7         OFF = "off",
  8         DISABLED = "disabled",
  9         BUTTON_CLASS = "ke-triplebutton",
 10         ON_CLASS = "ke-triplebutton-on",
 11         OFF_CLASS = "ke-triplebutton-off",
 12         ACTIVE_CLASS = "ke-triplebutton-active",
 13         DISABLED_CLASS = "ke-triplebutton-disabled";
 14 
 15     function getTipText(str) {
 16         if (str && str.indexOf("<") == -1) {
 17             return str;
 18         }
 19         return 0;
 20     }
 21 
 22     var TripleButton = UIBase.create([UIBase['Box']['Render']], {
 23         _updateHref:function () {
 24             var self = this;
 25             self.get("el").attr("href", "javascript:void('" +
 26                 (getTipText(self.get("text")) || getTipText(self.get("title")) ) + "')");
 27         },
 28         bindUI:function () {
 29             var self = this,
 30                 keepFocus = self.get("keepFocus"),
 31                 el = self.get("el");
 32             el.on("click", self['_action'], self);
 33             // make ie do not lose focus
 34             if (keepFocus) {
 35                 el.unselectable();
 36             }
 37             // 添加鼠标点击视觉效果
 38             el.on("mousedown", function (e) {
 39                 if (self.get("state") == OFF) {
 40                     el.addClass(ACTIVE_CLASS);
 41                 }
 42                 // make non-ie do not lose focus
 43                 if (keepFocus) {
 44                     e.halt();
 45                 }
 46             });
 47             el.on("mouseup mouseleave", function () {
 48                 if (self.get("state") == OFF &&
 49                     el.hasClass(ACTIVE_CLASS)) {
 50                     //click 后出发
 51                     setTimeout(function () {
 52                         el.removeClass(ACTIVE_CLASS);
 53                     }, 300);
 54                 }
 55             });
 56         },
 57         _uiSetTitle:function () {
 58             var self = this;
 59             self.get("el").attr("title", self.get("title"));
 60             self._updateHref();
 61         },
 62         _uiSetContentCls:function (contentCls) {
 63             var self = this,
 64                 el = self.get("el");
 65             if (contentCls !== undefined) {
 66                 el.html("<span class='ke-toolbar-item " + contentCls + "' />");
 67                 if (self.get('keepFocus')) {
 68                     el.unselectable(undefined);
 69                 }
 70             }
 71         },
 72         _uiSetText:function (text) {
 73             var self = this,
 74                 el = self.get("el");
 75             el.html(text);
 76             self._updateHref();
 77         },
 78         _uiSetState:function (n, ev) {
 79             this["_" + n](ev);
 80         },
 81         disable:function () {
 82             var self = this;
 83             self.set("state", DISABLED);
 84         },
 85         enable:function () {
 86             var self = this;
 87             if (self.get("state") == DISABLED) {
 88                 self.set("state", self._savedState);
 89             }
 90         },
 91         _action:function (ev) {
 92             var self = this;
 93             self.fire(self.get("state") + "Click", {
 94                 TripleEvent:ev
 95             });
 96             self.fire("click", {
 97                 TripleClickType:self.get("state") + "Click"
 98             });
 99             if (!self.get("keepFocus")) {
100                 // chrome will make body focus
101                 self.get("el")[0].focus();
102             }
103         },
104         bon:function () {
105             this.set("state", ON);
106         },
107         boff:function () {
108             this.set("state", OFF);
109         },
110         _on:function () {
111             var el = this.get("el");
112             el.removeClass(OFF_CLASS + " " + DISABLED_CLASS)
113                 .addClass(ON_CLASS);
114         },
115         _off:function () {
116             var el = this.get("el");
117             el.removeClass(ON_CLASS + " " + DISABLED_CLASS)
118                 .addClass(OFF_CLASS);
119         },
120         _disabled:function (ev) {
121             var el = this.get("el");
122             this._savedState = ev.prevVal;
123             el.removeClass(OFF_CLASS + " " + ON_CLASS)
124                 .addClass(DISABLED_CLASS);
125         }
126     }, {
127         name:"editor/plugin/button",
128         ATTRS:{
129             state:{value:OFF},
130             elCls:{value:[BUTTON_CLASS, OFF_CLASS].join(" ")},
131             elTagName:{value:"a"},
132             title:{},
133             contentCls:{},
134             text:{},
135             keepFocus:{
136                 value:true
137             }
138         }
139     });
140 
141     TripleButton.ON = ON;
142     TripleButton.OFF = OFF;
143     TripleButton.DISABLED = DISABLED;
144     TripleButton.ON_CLASS = ON_CLASS;
145     TripleButton.OFF_CLASS = OFF_CLASS;
146     TripleButton.DISABLED_CLASS = DISABLED_CLASS;
147 
148 
149     Editor.TripleButton = TripleButton;
150     /**
151      * 将 button ui 和点击功能分离
152      */
153     Editor.prototype.addButton = function (cfg, methods, ButtonType) {
154         ButtonType = ButtonType || TripleButton;
155         var self = this,
156             b = new ButtonType(S.mix({
157                 render:self.get("toolBarEl"),
158                 autoRender:true,
159                 editor:self
160             }, cfg));
161 
162         S.mix(b, methods);
163 
164         if (b.init) {
165             b.init();
166         }
167 
168         self.on("selectionChange", function () {
169             if (self.get("mode") == Editor.SOURCE_MODE) {
170                 return;
171             }
172             b.selectionChange && b.selectionChange.apply(b, arguments);
173         });
174 
175         b.on("click", function (ev) {
176             var t = ev.TripleClickType;
177             if (b[t]) {
178                 b[t].apply(b, arguments);
179             }
180         });
181 
182         if (b.get("mode") == Editor.WYSIWYG_MODE) {
183             self.on("wysiwygMode", b.enable, b);
184             self.on("sourceMode", b.disable, b);
185         }
186 
187         self.on("destroy", function () {
188             b.destroy();
189         });
190 
191         return b;
192     };
193 
194     return TripleButton;
195 }, {
196     requires:['editor', 'uibase']
197 });
