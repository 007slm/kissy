1 /**
  2  *  @fileOverview mix loader into S and infer KISSy baseUrl if not set
  3  *  @author  lifesinger@gmail.com,yiminghe@gmail.com
  4  */
  5 (function (S, loader, utils) {
  6     if ("require" in this) {
  7         return;
  8     }
  9     S.mix(S, loader);
 10 
 11     /**
 12      * get base from src
 13      * @param src script source url
 14      * @return base for kissy
 15      * @example:
 16      *   http://a.tbcdn.cn/s/kissy/1.1.6/??kissy-min.js,suggest/suggest-pkg-min.js
 17      *   http://a.tbcdn.cn/??s/kissy/1.1.6/kissy-min.js,s/kissy/1.1.5/suggest/suggest-pkg-min.js
 18      *   http://a.tbcdn.cn/??s/kissy/1.1.6/suggest/suggest-pkg-min.js,s/kissy/1.1.5/kissy-min.js
 19      *   http://a.tbcdn.cn/s/kissy/1.1.6/kissy-min.js?t=20101215.js
 20      * @notice: custom combo rules, such as yui3:
 21      *  <script src="path/to/kissy" data-combo-prefix="combo?" data-combo-sep="&"></script>
 22      */
 23         // notice: timestamp
 24     var baseReg = /^(.*)(seed|kissy)(-aio)?(-min)?\.js[^/]*/i,
 25         baseTestReg = /(seed|kissy)(-aio)?(-min)?\.js/i;
 26 
 27     function getBaseUrl(script) {
 28         var src = utils.absoluteFilePath(script.src),
 29             prefix = script.getAttribute('data-combo-prefix') || '??',
 30             sep = script.getAttribute('data-combo-sep') || ',',
 31             parts = src.split(sep),
 32             base,
 33             part0 = parts[0],
 34             index = part0.indexOf(prefix);
 35         // no combo
 36         if (index == -1) {
 37             base = src.replace(baseReg, '$1');
 38         } else {
 39             base = part0.substring(0, index);
 40             var part01 = part0.substring(index + 2, part0.length);
 41             // combo first
 42             // notice use match better than test
 43             if (part01.match(baseTestReg)) {
 44                 base += part01.replace(baseReg, '$1');
 45             }
 46             // combo after first
 47             else {
 48                 S.each(parts, function (part) {
 49                     if (part.match(baseTestReg)) {
 50                         base += part.replace(baseReg, '$1');
 51                         return false;
 52                     }
 53                 });
 54             }
 55         }
 56         return base;
 57     }
 58 
 59     /**
 60      * Initializes loader.
 61      */
 62     S.__initLoader = function () {
 63         var self = this;
 64         self.Env.mods = self.Env.mods || {}; // all added mods
 65     };
 66 
 67     S.Env._loadQueue = {}; // information for loading and loaded mods
 68     S.__initLoader();
 69 
 70     (function () {
 71         // get base from current script file path
 72         var scripts = document.getElementsByTagName('script'),
 73             currentScript = scripts[scripts.length - 1],
 74             base = getBaseUrl(currentScript);
 75         S.Config.base = utils.normalBasePath(base);
 76         // the default timeout for getScript
 77         S.Config.timeout = 10;
 78     })();
 79 
 80     S.mix(S.configs, {
 81         base:function (base) {
 82             S.Config.base = utils.normalBasePath(base);
 83         },
 84         timeout:function (v) {
 85             S.Config.timeout = v;
 86         },
 87         debug:function (v) {
 88             S.Config.debug = v;
 89         }
 90     });
 91 
 92     // for S.app working properly
 93     S.each(loader, function (v, k) {
 94         S.__APP_MEMBERS.push(k);
 95     });
 96 
 97     S.__APP_INIT_METHODS.push('__initLoader');
 98 
 99 })(KISSY, KISSY.__loader, KISSY.__loaderUtils);
