1 /**
  2  * Add indent button.
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/outdent/index", function (S, Editor, indexCmd) {
  6 
  7     function outdent() {
  8 
  9     }
 10 
 11     S.augment(outdent, {
 12         renderUI:function (editor) {
 13 
 14             indexCmd.init(editor);
 15 
 16             var queryOutdent = Editor.Utils.getQueryCmd("outdent");
 17 
 18             editor.addButton("outdent", {
 19                 tooltip:"减少缩进量 ",
 20                 listeners:{
 21                     click:function () {
 22                         editor.execCommand("outdent");
 23                         editor.focus();
 24 
 25                     },
 26                     afterSyncUI:function () {
 27                         var self = this;
 28                         editor.on("selectionChange", function (e) {
 29                             if (editor.get("mode") == Editor.SOURCE_MODE) {
 30                                 return;
 31                             }
 32                             if (editor.execCommand(queryOutdent, e.path)) {
 33                                 self.set("disabled", false);
 34                             } else {
 35                                 self.set("disabled", true);
 36                             }
 37                         });
 38 
 39                     }
 40                 },
 41                 mode:Editor.WYSIWYG_MODE
 42             });
 43         }
 44     });
 45 
 46     return outdent;
 47 
 48 }, {
 49     requires:['editor', './cmd']
 50 });
