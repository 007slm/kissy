 19     function getMenu(self, init) {
 20         var m = self.get("menu");
 21         if (m && m.xclass) {
 22             if (init) {
 23                 m = Component.create(m, self);
 24                 self.__set("menu", m);
 25             } else {
 26                 return null;
 27             }
 28         }
 29         return m;
 30     }

 33     function hideMenu(self) {
 34         var menu = getMenu(self);
 35         if (menu) {
 36             menu.hide();
 37         }
 38     }

 40     function alignMenuImmediately(self) {
 41         var menu = self.get("menu");
 42         var align = S.clone(menu.get("align"));
 43         align.node = self.get("el");
 44         S.mix(align, ALIGN, false);
 45         menu.set("align", align);
 46     }
 47 
 48     function alignWithTokenImmediately(self) {
 49         var inputDesc = getInputDesc(self),
 50             tokens = inputDesc.tokens,
 51             menu = self.get("menu"),
 52             cursorPosition = inputDesc.cursorPosition,
 53             tokenIndex = inputDesc.tokenIndex,
 54             tokenCursorPosition,
 55             cursorOffset,
 56             input = self.get("input");
 57         tokenCursorPosition = tokens.slice(0, tokenIndex).join("").length;
 58         if (tokenCursorPosition > 0) {
 59             // behind separator
 60             ++tokenCursorPosition;
 61         }
 62         input.prop("selectionStart", tokenCursorPosition);
 63         input.prop("selectionEnd", tokenCursorPosition);
 64         cursorOffset = input.prop("KsCursorOffset");
 65         input.prop("selectionStart", cursorPosition);
 66         input.prop("selectionEnd", cursorPosition);
 67         menu.set("xy", [cursorOffset.left, cursorOffset.top]);
 68     }

 70     function reposition() {
 71         var self = this,
 72             menu = getMenu(self);
 73         if (menu && menu.get("visible")) {
 74             if (self.get("multiple") && self.get("alignWithCursor")) {
 75                 alignWithTokenImmediately(self);
 76             } else {
 77                 alignMenuImmediately(self);
 78             }
 79         }
 80     }
 81 
 82     var repositionBuffer = S.buffer(reposition, 50);

 84 
 85     function delayHide() {
 86         var self = this;
 87         self._focusoutDismissTimer = setTimeout(function () {
 88             self.set("collapsed", true);
 89         }, 30);
 90     }
 91 
 92     function clearDismissTimer() {
 93         var self = this, t;
 94         if (t = self._focusoutDismissTimer) {
 95             clearTimeout(t);
 96             self._focusoutDismissTimer = null;
 97         }
 98     }

101     function showMenu(self) {
102         var el = self.get("el"),
103             menu = getMenu(self, 1);
104         // 保证显示前已经 bind 好 menu 事件
105 
106         clearDismissTimer.call(self);
107         if (menu && !menu.get("visible")) {
108             // 先 render，监听 width 变化事件
109             menu.render();
110             self.bindMenu();
111             // 根据 el 自动调整大小
112             if (self.get("matchElWidth")) {
113                 menu.set("width", el.innerWidth());
114             }
115             menu.show();
116             reposition.call(self);
117             self.get("input").attr("aria-owns", menu.get("el")[0].id);
118         }
119     }

233             handleKeyEventInternal:function (e) {
234                 var self = this,
235                     input = self.get("input"),
236                     menu = getMenu(self);
237 
238                 if (!menu) {
239                     return;
240                 }
241 
242                 var updateInputOnDownUp = self.get("updateInputOnDownUp");
243 
244                 if (updateInputOnDownUp) {
245                     // combobox will change input value
246                     // but it does not need to reload data
247                     if (S.inArray(e.keyCode, [
248                         KeyCodes.UP,
249                         KeyCodes.DOWN,
250                         KeyCodes.ESC
251                     ])) {
252                         self._stopNotify = 1;
253                     } else {
254                         self._stopNotify = 0;
255                     }
256                 }
257 
258                 var activeItem;
259 
260                 if (menu.get("visible")) {
261                     var handledByMenu = menu.handleKeydown(e);
262 
263                     if (updateInputOnDownUp) {
264                         if (S.inArray(e.keyCode, [KeyCodes.DOWN, KeyCodes.UP])) {
265                             // update menu's active value to input just for show
266                             setValue(self, menu.get("activeItem").get("textContent"));
267                         }
268                     }
269                     // esc
270                     if (e.keyCode == KeyCodes.ESC) {
271                         self.set("collapsed", true);
272                         if (updateInputOnDownUp) {
273                             // restore original user's input text
274                             setValue(self, self._savedInputValue);
275                         }
276                         return true;
277                     }
278 
279                     // tab
280                     // if menu is open and an menuitem is highlighted, see as click/enter
281                     if (e.keyCode == KeyCodes.TAB) {
282                         if (activeItem = menu.get("activeItem")) {
283                             activeItem.performActionInternal();
284                             // only prevent focus change in multiple mode
285                             if (self.get("multiple")) {
286                                 return true;
287                             }
288                         }
289                     }
290                     return handledByMenu;
291                 } else if ((e.keyCode == KeyCodes.DOWN || e.keyCode == KeyCodes.UP)) {
292                     // re-fetch , consider multiple input
293                     S.log("refetch : " + getValue(self));
294                     self.sendRequest(getValue(self));
295                     return true;
296                 }
297             },

497     function setValue(self, value) {
498         var input = self.get("input");
499         if (self.get("multiple")) {
500             var inputDesc = getInputDesc(self),
501                 tokens = inputDesc.tokens,
502                 tokenIndex = Math.max(0, inputDesc.tokenIndex),
503                 separator = self.get("separator"),
504                 cursorPosition,
505                 separatorType = self.get("separatorType"),
506                 token = tokens[tokenIndex];
507 
508             if (token && separator.indexOf(token.charAt(0)) != -1) {
509                 tokens[tokenIndex] = token.charAt(0);
510             } else {
511                 tokens[tokenIndex] = "";
512             }
513 
514             tokens[tokenIndex] += value;
515 
516             var nextToken = tokens[tokenIndex + 1];
517 
518             // appendSeparatorOnComplete if next token does not start with separator
519             if (separatorType == SUFFIX && (!nextToken || separator.indexOf(nextToken.charAt(0)) == -1 )) {
520                 tokens[tokenIndex] += separator.charAt(0);
521             }
522 
523             cursorPosition = tokens.slice(0, tokenIndex + 1).join("").length;
524 
525             input.val(tokens.join(""));
526 
527             input.prop("selectionStart", cursorPosition);
528             input.prop("selectionEnd", cursorPosition);
529         } else {
530             input.val(value);
531         }
532     }

535     /**
536      * Consider multiple mode , get token at current cursor position
537      */
538     function getValue(self) {
539         var input = self.get("input"),
540             inputVal = input.val();
541         if (self.get("multiple")) {
542             var inputDesc = getInputDesc(self);
543             var tokens = inputDesc.tokens,
544                 tokenIndex = inputDesc.tokenIndex;
545             var separator = self.get("separator");
546             var separatorType = self.get("separatorType");
547             var token = tokens[tokenIndex] || "";
548             // only if token starts with separator , then token has meaning!
549             // token can not be empty
550             if (token && separator.indexOf(token.charAt(0)) != -1) {
551                 // remove separator
552                 return token.substring(1);
553             }
554             // cursor is at the beginning of textarea
555             if (separatorType == SUFFIX && (tokenIndex == 0 || tokenIndex == -1)) {
556                 return token;
557             }
558             return undefined;
559         } else {
560             return inputVal;
561         }
562     }

565     function onValueChange() {
566         var self = this;
567         if (self._stopNotify) {
568             return;
569         }
570         var value = getValue(self);
571         if (value === undefined) {
572             self.set("collapsed", true);
573             return;
574         }
575         self._savedInputValue = value;
576         S.log("value change: " + value);
577         self.sendRequest(value);
578     }

580     function renderData(data) {
581         var self = this,
582             v,
583             children = [],
584             val,
585             contents,
586             matchVal,
587             i,
588             menu = getMenu(self, 1);
589 
590 
591         menu.removeChildren(true);
592 
593         if (data && data.length) {
594             data = data.slice(0, self.get("maxItemCount"));
595             if (self.get("format")) {
596                 contents = self.get("format").call(self, getValue(self), data);
597             } else {
598                 contents = [];
599             }
600             for (i = 0; i < data.length; i++) {
601                 v = data[i];
602                 children.push(menu.addChild(S.mix({
603                     xclass:'menuitem',
604                     content:v,
605                     textContent:v,
606                     value:v
607                 }, contents[i])));
608             }
609             // make menu item (which textContent is same as input) active
610             val = getValue(self);
611             for (i = 0; i < children.length; i++) {
612                 if (children[i].get("textContent") == val) {
613                     menu.set("highlightedItem", children[i]);
614                     matchVal = true;
615                     break;
616                 }
617             }
618             // Whether or not the first row should be highlighted by default.
619             if (!matchVal && self.get("autoHighlightFirst")) {
620                 for (i = 0; i < children.length; i++) {
621                     if (!children[i].get("disabled")) {
622                         menu.set("highlightedItem", children[i]);
623                         break;
624                     }
625                 }
626             }
627             self.set("collapsed", false);
628         } else {
629             self.set("collapsed", true);
630         }
631     }

634         var self = this,
635             input = self.get("input");
636         if (!self.get('collapsed')) {
637             self.set('collapsed', true);
638         } else {

640             self.sendRequest('');

648     function getInputDesc(self) {
649         var input = self.get("input"),
650             inputVal = input.val(),
651             tokens = [],
652             cache = [],
653             literal = self.get("literal"),
654             separator = self.get("separator"),
655             inLiteral = false,
656             whitespace = self.get("whitespace"),
657             cursorPosition = input.prop('selectionStart'),
658             tokenIndex = -1;
659 
660         for (var i = 0; i < inputVal.length; i++) {
661             var c = inputVal.charAt(i);
662             if (i == cursorPosition) {
663                 // current token index
664                 tokenIndex = tokens.length;
665             }
666             if (!inLiteral) {
667                 // whitespace is not part of token value
668                 // then separate
669                 if (!whitespace && /\s|\xa0/.test(c)) {
670                     tokens.push(cache.join(""));
671                     cache = [];
672                 }
673 
674                 if (separator.indexOf(c) != -1) {
675                     tokens.push(cache.join(""));
676                     cache = [];
677                 }
678             }
679             if (literal) {
680                 if (c == literal) {
681                     inLiteral = !inLiteral;
682                 }
683             }
684             cache.push(c);
685         }
686 
687         if (cache.length) {
688             tokens.push(cache.join(""));
689         }
690         if (tokenIndex == -1) {
691             tokenIndex = tokens.length - 1;
692         }
693         return {
694             tokens:tokens,
695             cursorPosition:cursorPosition,
696             tokenIndex:tokenIndex
697         };
698     }
