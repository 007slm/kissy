 78     function replaceVariables(list, variablesValues) {
 79         for (var item in list) {
 80             if (!list.hasOwnProperty(item)) continue;
 81             if (S.isString(list[ item ])) {
 82                 list[ item ] = list[ item ].replace(varRegex, function (match, varName) {
 83                     return variablesValues[ varName ];
 84                 });
 85             } else {
 86                 replaceVariables(list[ item ], variablesValues);
 87             }
 88         }
 89     }

171         // Checks if an element, or any of its attributes, is removable by the
172         // current style definition.
173         checkElementRemovable:function (element, fullMatch) {
174             if (!element)
175                 return FALSE;
176 
177             var def = this._.definition,
178                 attribs, styles;
179 
180             // If the element name is the same as the style name.
181             if (element.nodeName() == this.element) {
182                 // If no attributes are defined in the element.
183                 if (!fullMatch && !element._4e_hasAttributes())
184                     return TRUE;
185 
186                 attribs = getAttributesForComparison(def);
187 
188                 if (attribs["_length"]) {
189                     for (var attName in attribs) {
190 
191                         if (attName == '_length')
192                             continue;
193 
194                         if (attribs.hasOwnProperty(attName)) {
195 
196                             var elementAttr = element.attr(attName) || '';
197                             if (attName == 'style' ?
198                                 compareCssText(attribs[ attName ],
199                                     normalizeCssText(elementAttr, FALSE))
200                                 : attribs[ attName ] == elementAttr) {
201                                 if (!fullMatch)
202                                     return TRUE;
203                             }
204                             else if (fullMatch)
205                                 return FALSE;
206                         }
207 
208 
209                     }
210                     if (fullMatch)
211                         return TRUE;
212                 }
213                 else
214                     return TRUE;
215             }
216 
217             // Check if the element can be somehow overriden.
218             var overrides = getOverrides(this),
219                 override = overrides[ element.nodeName() ] || overrides["*"];
220 
221             if (override) {
222                 // If no attributes have been defined, remove the element.
223                 if (!( attribs = override.attributes )
224                     &&
225                     !( styles = override.styles)
226                     )
227                     return TRUE;
228                 if (attribs) {
229                     for (var i = 0; i < attribs.length; i++) {
230                         attName = attribs[i][0];
231                         var actualAttrValue = element.attr(attName);
232                         if (actualAttrValue) {
233                             var attValue = attribs[i][1];
234                             // Remove the attribute if:
235                             //    - The override definition value is NULL;
236                             //    - The override definition value is a string that
237                             //      matches the attribute value exactly.
238                             //    - The override definition value is a regex that
239                             //      has matches in the attribute value.
240                             if (attValue === NULL ||
241                                 ( typeof attValue == 'string'
242                                     && actualAttrValue == attValue ) ||
243                                 attValue.test && attValue.test(actualAttrValue))
244                                 return TRUE;
245                         }
246                     }
247                 }
248                 if (styles) {
249                     for (i = 0; i < styles.length; i++) {
250                         var styleName = styles[i][0];
251                         var actualStyleValue = element.css(styleName);
252                         if (actualStyleValue) {
253                             var styleValue = styles[i][1];
254                             if (styleValue === NULL
255                                 //inherit wildcard !
256                                 //|| styleValue == "inherit"
257                                 || ( typeof styleValue == 'string'
258                                 && actualStyleValue == styleValue ) ||
259                                 styleValue.test && styleValue.test(actualStyleValue))
260                                 return TRUE;
261                         }
262                     }
263                 }
264             }
265             return FALSE;
266         },

268         /**
269          * Get the style state inside an element path. Returns "TRUE" if the
270          * element is active in the path.
271          */
272         checkActive:function (elementPath) {
273             switch (this.type) {
274                 case KEST.STYLE_BLOCK :
275                     return this.checkElementRemovable(elementPath.block
276                         || elementPath.blockLimit, TRUE);
277 
278                 case KEST.STYLE_OBJECT :
279                 case KEST.STYLE_INLINE :
280 
281                     var elements = elementPath.elements;
282 
283                     for (var i = 0, element; i < elements.length; i++) {
284                         element = elements[ i ];
285 
286                         if (this.type == KEST.STYLE_INLINE
287                             && ( DOM.equals(element, elementPath.block)
288                             || DOM.equals(element, elementPath.blockLimit) ))
289                             continue;
290 
291                         if (this.type == KEST.STYLE_OBJECT
292                             && !( element.nodeName() in objectElements ))
293                             continue;
294 
295                         if (this.checkElementRemovable(element, TRUE))
296                             return TRUE;
297                     }
298             }
299             return FALSE;
300         }

304     KEStyle.getStyleText = function (styleDefinition) {
305         // If we have already computed it, just return it.
306         var stylesDef = styleDefinition._ST;
307         if (stylesDef)
308             return stylesDef;
309 
310         stylesDef = styleDefinition["styles"];
311 
312         // Builds the StyleText.
313         var stylesText = ( styleDefinition["attributes"]
314             && styleDefinition["attributes"][ 'style' ] ) || '',
315             specialStylesText = '';
316 
317         if (stylesText.length)
318             stylesText = stylesText.replace(semicolonFixRegex, ';');
319 
320         for (var style in stylesDef) {
321 
322             if (stylesDef.hasOwnProperty(style)) {
323 
324                 var styleVal = stylesDef[ style ],
325                     text = ( style + ':' + styleVal ).replace(semicolonFixRegex, ';');
326 
327                 // Some browsers don't support 'inherit' property value, leave them intact. (#5242)
328                 if (styleVal == 'inherit')
329                     specialStylesText += text;
330                 else
331                     stylesText += text;
332             }
333         }
334 
335         // Browsers make some changes to the style when applying them. So, here
336         // we normalize it to the browser format.
337         if (stylesText.length)
338             stylesText = normalizeCssText(stylesText);
339 
340         stylesText += specialStylesText;
341 
342         // Return it, saving it to the next request.
343         return ( styleDefinition._ST = stylesText );
344     };

346     function getElement(style, targetDocument, element) {
347         var el,
348             //def = style._.definition,
349             elementName = style["element"];
350 
351         // The "*" element name will always be a span for this function.
352         if (elementName == '*')
353             elementName = 'span';
354 
355         // Create the element.
356         el = new Node(targetDocument.createElement(elementName));
357 
358         // #6226: attributes should be copied before the new ones are applied
359         if (element)
360             element._4e_copyAttributes(el);
361 
362         return setupElement(el, style);
363     }

365     function setupElement(el, style) {
366         var def = style._["definition"],
367             attributes = def["attributes"],
368             styles = KEStyle.getStyleText(def);
369 
370         // Assign all defined attributes.
371         if (attributes) {
372             for (var att in attributes) {
373                 if (attributes.hasOwnProperty(att)) {
374                     el.attr(att, attributes[ att ]);
375                 }
376             }
377         }
378 
379         // Assign all defined styles.
380 
381         if (styles)
382             el[0].style.cssText = styles;
383 
384         return el;
385     }

387     function applyBlockStyle(range) {
388         // Serializible bookmarks is needed here since
389         // elements may be merged.
390         var bookmark = range.createBookmark(TRUE),
391             iterator = range.createIterator();
392         iterator.enforceRealBlocks = TRUE;
393 
394         // make recognize <br /> tag as a separator in ENTER_BR mode (#5121)
395         //if (this._.enterMode)
396         iterator.enlargeBr = TRUE;//( this._.enterMode != CKEDITOR.ENTER_BR );
397 
398         var block, doc = range.document;
399         // Only one =
400         while (( block = iterator.getNextParagraph() )) {
401             var newBlock = getElement(this, doc, block);
402             replaceBlock(block, newBlock);
403         }
404         range.moveToBookmark(bookmark);
405     }

407     // Wrapper function of String::replace without considering of head/tail bookmarks nodes.
408     function replace(str, regexp, replacement) {
409         var headBookmark = '',
410             tailBookmark = '';
411 
412         str = str.replace(/(^<span[^>]+_ke_bookmark.*?\/span>)|(<span[^>]+_ke_bookmark.*?\/span>$)/gi,
413             function (str, m1, m2) {
414                 m1 && ( headBookmark = m1 );
415                 m2 && ( tailBookmark = m2 );
416                 return '';
417             });
418         return headBookmark + str.replace(regexp, replacement) + tailBookmark;
419     }

421     /**
422      * Converting from a non-PRE block to a PRE block in formatting operations.
423      */
424     function toPre(block, newBlock) {
425         // First trim the block content.
426         var preHtml = block.html();
427 
428         // 1. Trim head/tail spaces, they're not visible.
429         preHtml = replace(preHtml, /(?:^[ \t\n\r]+)|(?:[ \t\n\r]+$)/g, '');
430         // 2. Delete ANSI whitespaces immediately before and after <BR> because
431         //    they are not visible.
432         preHtml = preHtml.replace(/[ \t\r\n]*(<br[^>]*>)[ \t\r\n]*/gi, '$1');
433         // 3. Compress other ANSI whitespaces since they're only visible as one
434         //    single space previously.
435         // 4. Convert   to spaces since   is no longer needed in <PRE>.
436         preHtml = preHtml.replace(/([ \t\n\r]+| )/g, ' ');
437         // 5. Convert any <BR /> to \n. This must not be done earlier because
438         //    the \n would then get compressed.
439         preHtml = preHtml.replace(/<br\b[^>]*>/gi, '\n');
440 
441         // Krugle: IE normalizes innerHTML to <pre>, breaking whitespaces.
442         if (UA['ie']) {
443             var temp = block[0].ownerDocument.createElement('div');
444             temp.appendChild(newBlock[0]);
445             newBlock[0].outerHTML = '<pre>' + preHtml + '</pre>';
446             newBlock = new Node(temp.firstChild);
447             newBlock._4e_remove();
448         }
449         else
450             newBlock.html(preHtml);
451 
452         return newBlock;
453     }

455     /**
456      * Split into multiple <pre> blocks separated by double line-break.
457      * @param preBlock
458      */
459     function splitIntoPres(preBlock) {
460         // Exclude the ones at header OR at tail,
461         // and ignore bookmark content between them.
462         var duoBrRegex = /(\S\s*)\n(?:\s|(<span[^>]+_ck_bookmark.*?\/span>))*\n(?!$)/gi,
463             //blockName = preBlock.nodeName(),
464             splittedHtml = replace(preBlock._4e_outerHtml(),
465                 duoBrRegex,
466                 function (match, charBefore, bookmark) {
467                     return charBefore + '</pre>' + bookmark + '<pre>';
468                 });
469 
470         var pres = [];
471         splittedHtml.replace(/<pre\b.*?>([\s\S]*?)<\/pre>/gi,
472             function (match, preContent) {
473                 pres.push(preContent);
474             });
475         return pres;
476     }

478     // Replace the original block with new one, with special treatment
479     // for <pre> blocks to make sure content format is well preserved, and merging/splitting adjacent
480     // when necessary.(#3188)
481     function replaceBlock(block, newBlock) {
482         var newBlockIsPre = newBlock.nodeName == ('pre'),
483             blockIsPre = block.nodeName == ('pre'),
484             isToPre = newBlockIsPre && !blockIsPre,
485             isFromPre = !newBlockIsPre && blockIsPre;
486 
487         if (isToPre)
488             newBlock = toPre(block, newBlock);
489         else if (isFromPre)
490         // Split big <pre> into pieces before start to convert.
491             newBlock = fromPres(splitIntoPres(block), newBlock);
492         else
493             block._4e_moveChildren(newBlock);
494 
495         block[0].parentNode.replaceChild(newBlock[0], block[0]);
496         if (newBlockIsPre) {
497             // Merge previous <pre> blocks.
498             mergePre(newBlock);
499         }
500     }

502     /**
503      * Merge a <pre> block with a previous sibling if available.
504      */
505     function mergePre(preBlock) {
506         var previousBlock;
507         if (!( ( previousBlock = preBlock._4e_previousSourceNode(TRUE, KEN.NODE_ELEMENT) )
508             && previousBlock.nodeName() == 'pre' ))
509             return;
510 
511         // Merge the previous <pre> block contents into the current <pre>
512         // block.
513         //
514         // Another thing to be careful here is that currentBlock might contain
515         // a '\n' at the beginning, and previousBlock might contain a '\n'
516         // towards the end. These new lines are not normally displayed but they
517         // become visible after merging.
518         var mergedHtml = replace(previousBlock.html(), /\n$/, '') + '\n\n' +
519             replace(preBlock.html(), /^\n/, '');
520 
521         // Krugle: IE normalizes innerHTML from <pre>, breaking whitespaces.
522         if (UA['ie'])
523             preBlock[0].outerHTML = '<pre>' + mergedHtml + '</pre>';
524         else
525             preBlock.html(mergedHtml);
526 
527         previousBlock._4e_remove();
528     }

530     /**
531      * Converting a list of <pre> into blocks with format well preserved.
532      */
533     function fromPres(preHtmls, newBlock) {
534         var docFrag = newBlock[0].ownerDocument.createDocumentFragment();
535         for (var i = 0; i < preHtmls.length; i++) {
536             var blockHtml = preHtmls[ i ];
537 
538             // 1. Trim the first and last line-breaks immediately after and before <pre>,
539             // they're not visible.
540             blockHtml = blockHtml.replace(/(\r\n|\r)/g, '\n');
541             blockHtml = replace(blockHtml, /^[ \t]*\n/, '');
542             blockHtml = replace(blockHtml, /\n$/, '');
543             // 2. Convert spaces or tabs at the beginning or at the end to  
544             blockHtml = replace(blockHtml, /^[ \t]+|[ \t]+$/g, function (match, offset) {
545                 if (match.length == 1)    // one space, preserve it
546                     return ' ';
547                 else if (!offset)        // beginning of block
548                     return new Array(match.length).join(' ') + ' ';
549                 else                // end of block
550                     return ' ' + new Array(match.length).join(' ');
551             });
552 
553             // 3. Convert \n to <BR>.
554             // 4. Convert contiguous (i.e. non-singular) spaces or tabs to  
555             blockHtml = blockHtml.replace(/\n/g, '<br>');
556             blockHtml = blockHtml.replace(/[ \t]{2,}/g,
557                 function (match) {
558                     return new Array(match.length).join(' ') + ' ';
559                 });
560 
561             var newBlockClone = newBlock.clone();
562             newBlockClone.html(blockHtml);
563             docFrag.appendChild(newBlockClone[0]);
564         }
565         return docFrag;
566     }

568     /**
569      *
570      * @param range
571      */
572     function applyInlineStyle(range) {
573         var self = this,
574             document = range.document;
575 
576         if (range.collapsed) {
577             // Create the element to be inserted in the DOM.
578             var collapsedElement = getElement(this, document, undefined);
579             // Insert the empty element into the DOM at the range position.
580             range.insertNode(collapsedElement);
581             // Place the selection right inside the empty element.
582             range.moveToPosition(collapsedElement, KER.POSITION_BEFORE_END);
583             return;
584         }
585         var elementName = this["element"],
586             def = this._["definition"],
587             isUnknownElement,
588             // Get the DTD definition for the element. Defaults to "span".
589             dtd = DTD[ elementName ];
590         if (!dtd) {
591             isUnknownElement = TRUE;
592             dtd = DTD["span"];
593         }
594 
595         // Bookmark the range so we can re-select it after processing.
596         var bookmark = range.createBookmark();
597 
598         // Expand the range.
599         range.enlarge(KER.ENLARGE_ELEMENT);
600         range.trim();
601 
602         // Get the first node to be processed and the last, which concludes the
603         // processing.
604         var boundaryNodes = range.createBookmark(),
605             firstNode = boundaryNodes.startNode,
606             lastNode = boundaryNodes.endNode,
607             currentNode = firstNode,
608             styleRange;
609 
610         while (currentNode && currentNode[0]) {
611             var applyStyle = FALSE;
612 
613             if (DOM.equals(currentNode, lastNode)) {
614                 currentNode = NULL;
615                 applyStyle = TRUE;
616             }
617             else {
618                 var nodeType = currentNode[0].nodeType,
619                     nodeName = nodeType == KEN.NODE_ELEMENT ?
620                         currentNode.nodeName() : NULL;
621 
622                 if (nodeName && currentNode.attr('_ke_bookmark')) {
623                     currentNode = currentNode._4e_nextSourceNode(TRUE);
624                     continue;
625                 }
626 
627                 // Check if the current node can be a child of the style element.
628                 if (!nodeName || ( dtd[ nodeName ]
629                     && ( currentNode._4e_position(lastNode) |
630                     ( KEP.POSITION_PRECEDING |
631                         KEP.POSITION_IDENTICAL |
632                         KEP.POSITION_IS_CONTAINED) )
633                     == ( KEP.POSITION_PRECEDING +
634                     KEP.POSITION_IDENTICAL +
635                     KEP.POSITION_IS_CONTAINED )
636                     && ( !def["childRule"] || def["childRule"](currentNode) ) )) {
637                     var currentParent = currentNode.parent();
638 
639 
640                     // hack for
641                     // 1<a href='http://www.taobao.com'>2</a>3
642                     // select all ,set link to http://www.ckeditor.com
643                     // expect => <a href='http://www.ckeditor.com'>123</a> (same with tinymce)
644                     // but now => <a href="http://www.ckeditor.com">1</a>
645                     // <a href="http://www.taobao.com">2</a>
646                     // <a href="http://www.ckeditor.com">3</a>
647                     // http://dev.ckeditor.com/ticket/8470
648                     if (currentParent &&
649                         elementName == "a" &&
650                         currentParent.nodeName() == elementName) {
651                         var tmpANode = getElement(self, document, undefined);
652                         currentParent._4e_moveChildren(tmpANode);
653                         currentParent[0].parentNode.replaceChild(tmpANode[0], currentParent[0]);
654                         tmpANode._4e_mergeSiblings();
655                     }
656 
657                     // Check if the style element can be a child of the current
658                     // node parent or if the element is not defined in the DTD.
659                     else if (currentParent && currentParent[0]
660                         && ( ( DTD[currentParent.nodeName()] ||
661                         DTD["span"] )[ elementName ] ||
662                         isUnknownElement )
663                         && ( !def["parentRule"] || def["parentRule"](currentParent) )) {
664                         // This node will be part of our range, so if it has not
665                         // been started, place its start right before the node.
666                         // In the case of an element node, it will be included
667                         // only if it is entirely inside the range.
668                         if (!styleRange &&
669                             ( !nodeName
670                                 || !DTD.$removeEmpty[ nodeName ]
671                                 || ( currentNode._4e_position(lastNode) |
672                                 ( KEP.POSITION_PRECEDING |
673                                     KEP.POSITION_IDENTICAL |
674                                     KEP.POSITION_IS_CONTAINED ))
675                                 ==
676                                 ( KEP.POSITION_PRECEDING +
677                                     KEP.POSITION_IDENTICAL +
678                                     KEP.POSITION_IS_CONTAINED )
679                                 )) {
680                             styleRange = new KERange(document);
681                             styleRange.setStartBefore(currentNode);
682                         }
683 
684                         // Non element nodes, or empty elements can be added
685                         // completely to the range.
686                         if (nodeType == KEN.NODE_TEXT ||
687                             ( nodeType == KEN.NODE_ELEMENT &&
688                                 !currentNode[0].childNodes.length )) {
689                             var includedNode = currentNode,
690                                 parentNode = null;
691 
692                             // This node is about to be included completelly, but,
693                             // if this is the last node in its parent, we must also
694                             // check if the parent itself can be added completelly
695                             // to the range.
696                             //2010-11-18 fix ; http://dev.ckeditor.com/ticket/6687
697                             //<span><book/>123<book/></span> 直接越过末尾 <book/>
698 
699                             // If the included node still is the last node in its
700                             // parent, it means that the parent can't be included
701                             // in this style DTD, so apply the style immediately.
702                             while (
703                                 (applyStyle = !includedNode.next(notBookmark))
704                                     && ( (parentNode = includedNode.parent()) &&
705                                     dtd[ parentNode.nodeName() ] )

954             /*
955              * Find out the style ancestor that needs to be broken down at startNode
956              * and endNode.
957              */
958             function breakNodes() {
959                 var startPath = new ElementPath(startNode.parent()),
960                     endPath = new ElementPath(endNode.parent()),
961                     breakStart = NULL,
962                     breakEnd = NULL;
963                 for (var i = 0; i < startPath.elements.length; i++) {
964                     var element = startPath.elements[ i ];
965 
966                     if (element == startPath.block ||
967                         element == startPath.blockLimit)
968                         break;
969 
970                     if (me.checkElementRemovable(element))
971                         breakStart = element;
972                 }
973                 for (i = 0; i < endPath.elements.length; i++) {
974                     element = endPath.elements[ i ];
975 
976                     if (element == endPath.block ||
977                         element == endPath.blockLimit)
978                         break;
979 
980                     if (me.checkElementRemovable(element))
981                         breakEnd = element;
982                 }
983 
984                 if (breakEnd)
985                     endNode._4e_breakParent(breakEnd);
986                 if (breakStart)
987                     startNode._4e_breakParent(breakStart);
988             }
989 
990             breakNodes();
991 
992             // Now, do the DFS walk.
993             var currentNode = new Node(startNode[0].nextSibling);
994             while (currentNode[0] !== endNode[0]) {
995                 /*
996                  * Need to get the next node first because removeFromElement() can remove
997                  * the current node from DOM tree.
998                  */
999                 var nextNode = currentNode._4e_nextSourceNode();
1000                 if (currentNode[0] &&
1001                     currentNode[0].nodeType == KEN.NODE_ELEMENT &&
1002                     this.checkElementRemovable(currentNode)) {
1003                     // Remove style from element or overriding element.
1004                     if (currentNode.nodeName() == this["element"])
1005                         removeFromElement(this, currentNode);
1006                     else {
1007                         var overrides = getOverrides(this);
1008                         removeOverrides(currentNode,
1009                             overrides[ currentNode.nodeName() ] || overrides["*"]);
1010 
1011                     }
1012 
1013                     /*
1014                      * removeFromElement() may have merged the next node with something before
1015                      * the startNode via mergeSiblings(). In that case, the nextNode would
1016                      * contain startNode and we'll have to call breakNodes() again and also
1017                      * reassign the nextNode to something after startNode.
1018                      */
1019                     if (nextNode[0].nodeType == KEN.NODE_ELEMENT &&
1020                         nextNode.contains(startNode)) {
1021                         breakNodes();
1022                         nextNode = new Node(startNode[0].nextSibling);
1023                     }
1024                 }
1025                 currentNode = nextNode;
1026             }
1027         }
1028         range.moveToBookmark(bookmark);

1048         typeof source == 'string' && ( source = parseStyleText(source) );
1049         typeof target == 'string' && ( target = parseStyleText(target) );
1050         for (var name in source) {
1051             if (source.hasOwnProperty(name)) {
1052                 // Value 'inherit'  is treated as a wildcard,

1055                     ( target[ name ] == source[ name ]
1056                         || source[ name ] == 'inherit'

1058                     return FALSE;

1136      * defined in this style overrides, with other element. All information in

1138      * @param  style

1140     function getOverrides(style) {
1141         if (style._.overrides)
1142             return style._.overrides;
1143 
1144         var overrides = ( style._.overrides = {} ),
1145             definition = style._.definition["overrides"];
1146 
1147         if (definition) {
1148             // The override description can be a string, object or array.
1149             // Internally, well handle arrays only, so transform it if needed.
1150             if (!S.isArray(definition))
1151                 definition = [ definition ];
1152 
1153             // Loop through all override definitions.
1154             for (var i = 0; i < definition.length; i++) {
1155                 var override = definition[i];
1156                 var elementName;
1157                 var overrideEl;
1158                 var attrs, styles;
1159 
1160                 // If can be a string with the element name.
1161                 if (typeof override == 'string')
1162                     elementName = override.toLowerCase();
1163                 // Or an object.
1164                 else {
1165                     elementName = override["element"] ?
1166                         override["element"].toLowerCase() :
1167                         style.element;
1168                     attrs = override["attributes"];
1169                     styles = override["styles"];
1170                 }
1171 
1172                 // We can have more than one override definition for the same
1173                 // element name, so we attempt to simply append information to
1174                 // it if it already exists.
1175                 overrideEl = overrides[ elementName ] ||
1176                     ( overrides[ elementName ] = {} );
1177 
1178                 if (attrs) {
1179                     // The returning attributes list is an array, because we
1180                     // could have different override definitions for the same
1181                     // attribute name.
1182                     var overrideAttrs = ( overrideEl["attributes"] =
1183                         overrideEl["attributes"] || new Array() );
1184                     for (var attName in attrs) {
1185                         // Each item in the attributes array is also an array,
1186                         // where [0] is the attribute name and [1] is the
1187                         // override value.

1189                             overrideAttrs.push([ attName.toLowerCase(), attrs[ attName ] ]);

1191                 }
1192

1195                     // The returning attributes list is an array, because we
1196                     // could have different override definitions for the same

1199                         overrideEl["styles"] || new Array() );

1202                         // where [0] is the style name and [1] is the
1203                         // override value.

1205                             overrideStyles.push([ styleName.toLowerCase(),

1216     // Removes a style from an element itself, don't care about its subtree.
1217     function removeFromElement(style, element) {
1218         var def = style._.definition,
1219             overrides = getOverrides(style),
1220             attributes = Utils.mix(def["attributes"],
1221                 (overrides[ element.nodeName()] || overrides["*"] || {})["attributes"]),
1222             styles = Utils.mix(def["styles"],
1223                 (overrides[ element.nodeName()] || overrides["*"] || {})["styles"]),
1224             // If the style is only about the element itself, we have to remove the element.
1225             removeEmpty = S.isEmptyObject(attributes) &&
1226                 S.isEmptyObject(styles);
1227 
1228         // Remove definition attributes/style from the element.
1229         for (var attName in attributes) {
1230             if (attributes.hasOwnProperty(attName)) {
1231                 // The 'class' element value must match (#1318).
1232                 if (( attName == 'class' || style._.definition["fullMatch"] )
1233                     && element.attr(attName) != normalizeProperty(attName,
1234                     attributes[ attName ]))
1235                     continue;
1236                 removeEmpty = removeEmpty || !!element.hasAttr(attName);
1237                 element.removeAttr(attName);
1238             }
1239         }
1240 
1241         for (var styleName in styles) {
1242             if (styles.hasOwnProperty(styleName)) {
1243                 // Full match style insist on having fully equivalence. (#5018)
1244                 if (style._.definition["fullMatch"]
1245                     && element.style(styleName)
1246                     != normalizeProperty(styleName, styles[ styleName ], TRUE))
1247                     continue;
1248 
1249                 removeEmpty = removeEmpty || !!element.style(styleName);
1250                 //设置空即为：清除样式
1251                 element.style(styleName, "");
1252             }
1253         }
1254 
1255         //removeEmpty &&
1256         //始终检查
1257         removeNoAttribsElement(element);
1258     }

1303      *  Note: Remove the element if no attributes remain.
1304      * @param {Object} element
1305      * @param {Object} overrides
1306      */
1307     function removeOverrides(element, overrides) {
1308         var i, attributes = overrides && overrides["attributes"];
1309 
1310         if (attributes) {
1311             for (i = 0; i < attributes.length; i++) {
1312                 var attName = attributes[i][0], actualAttrValue;
1313 
1314                 if (( actualAttrValue = element.attr(attName) )) {
1315                     var attValue = attributes[i][1];
1316 
1317                     // Remove the attribute if:
1318                     //    - The override definition value is NULL ;
1319                     //    - The override definition valie is a string that
1320                     //      matches the attribute value exactly.
1321                     //    - The override definition value is a regex that
1322                     //      has matches in the attribute value.
1323                     if (attValue === NULL ||
1324                         ( attValue.test && attValue.test(actualAttrValue) ) ||
1325                         ( typeof attValue == 'string' && actualAttrValue == attValue ))
1326                         element[0].removeAttribute(attName);
1327                 }
1328             }
1329         }
1330 
1331 
1332         var styles = overrides && overrides["styles"];
1333 
1334         if (styles) {
1335             for (i = 0; i < styles.length; i++) {
1336                 var styleName = styles[i][0], actualStyleValue;
1337 
1338                 if (( actualStyleValue = element.css(styleName) )) {
1339                     var styleValue = styles[i][1];
1340                     if (styleValue === NULL ||
1341                         //styleValue === "inherit" ||
1342                         ( styleValue.test && styleValue.test(actualAttrValue) ) ||
1343                         ( typeof styleValue == 'string' && actualStyleValue == styleValue ))
1344                         element.css(styleName, "");
1345                 }
1346             }

1352     // If the element has no more attributes, remove it.
1353     function removeNoAttribsElement(element) {
1354         // If no more attributes remained in the element, remove it,
1355         // leaving its children.
1356         if (!element._4e_hasAttributes()) {
1357             // Removing elements may open points where merging is possible,
1358             // so let's cache the first and last nodes for later checking.
1359             var firstChild = element[0].firstChild,
1360                 lastChild = element[0].lastChild;

1366                 firstChild.nodeType == KEN.NODE_ELEMENT &&
1367                 DOM._4e_mergeSiblings(firstChild);

1370                     && lastChild.nodeType == KEN.NODE_ELEMENT)
1371                     DOM._4e_mergeSiblings(lastChild);
