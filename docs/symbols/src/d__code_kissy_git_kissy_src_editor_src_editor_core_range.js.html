 42     /**
 43      * Extract html content within range.
 44      * @param {Number} action
 45      * 0 : delete
 46      * 1 : extract
 47      * 2 : clone
 48      */
 49     function execContentsAction(self, action) {
 50         var startNode = self.startContainer,
 51             endNode = self.endContainer,
 52             startOffset = self.startOffset,
 53             endOffset = self.endOffset,
 54             removeStartNode,
 55             hasSplitStart = FALSE,
 56             hasSplitEnd = FALSE,
 57             t,
 58             docFrag = undefined,
 59             doc = self.document,
 60             removeEndNode;
 61 
 62         if (self.collapsed) {
 63             return docFrag;
 64         }
 65 
 66         if (action > 0) {
 67             docFrag = doc.createDocumentFragment();
 68         }
 69 
 70         // 将 bookmark 包含在选区内
 71         self.optimizeBookmark();
 72 
 73 
 74         // endNode -> end guard , not included in range
 75 
 76         // For text containers, we must simply split the node and point to the
 77         // second part. The removal will be handled by the rest of the code .
 78         //最关键：一般起始都是在文字节点中，得到起点选择右边的文字节点，只对节点处理！
 79         if (endNode[0].nodeType == KEN.NODE_TEXT) {
 80             hasSplitEnd = TRUE;
 81             endNode = endNode._4e_splitText(endOffset);
 82         } else {
 83             // If the end container has children and the offset is pointing
 84             // to a child, then we should start from it.
 85             if (endNode[0].childNodes.length > 0) {
 86                 // If the offset points after the last node.
 87                 if (endOffset >= endNode[0].childNodes.length) {
 88                     // Let's create a temporary node and mark it for removal.
 89                     endNode = new Node(
 90                         endNode[0].appendChild(doc.createTextNode(""))
 91                     );
 92                     removeEndNode = TRUE;
 93                 } else {
 94                     endNode = new Node(endNode[0].childNodes[endOffset]);
 95                 }
 96             }
 97         }
 98 
 99         // startNode -> start guard , not included in range
100 
101         // For text containers, we must simply split the node. The removal will
102         // be handled by the rest of the code .
103         if (startNode[0].nodeType == KEN.NODE_TEXT) {
104             hasSplitStart = TRUE;
105             startNode._4e_splitText(startOffset);
106         } else {
107             // If the start container has children and the offset is pointing
108             // to a child, then we should start from its previous sibling.
109 
110             // If the offset points to the first node, we don't have a
111             // sibling, so let's use the first one, but mark it for removal.
112             if (!startOffset) {
113                 // Let's create a temporary node and mark it for removal.
114                 t = new Node(doc.createTextNode(""));
115                 startNode.prepend(t);
116                 startNode = t;
117                 removeStartNode = TRUE;
118             }
119             else if (startOffset >= startNode[0].childNodes.length) {
120                 // Let's create a temporary node and mark it for removal.
121                 startNode = new Node(startNode[0]
122                     .appendChild(doc.createTextNode('')));
123                 removeStartNode = TRUE;
124             } else
125                 startNode = new Node(
126                     startNode[0].childNodes[startOffset].previousSibling
127                 );
128         }
129 
130         // Get the parent nodes tree for the start and end boundaries.
131         //从根到自己
132         var startParents = startNode._4e_parents(),
133             endParents = endNode._4e_parents();
134 
135         startParents.each(function (n, i) {
136             startParents[i] = n;
137         });
138 
139         endParents.each(function (n, i) {
140             endParents[i] = n;
141         });
142 
143 
144         // Compare them, to find the top most siblings.
145         var i, topStart, topEnd;
146 
147         for (i = 0; i < startParents.length; i++) {
148             topStart = startParents[ i ];
149             topEnd = endParents[ i ];
150 
151             // The compared nodes will match until we find the top most
152             // siblings (different nodes that have the same parent).
153             // "i" will hold the index in the parents array for the top
154             // most element.
155             if (!topStart.equals(topEnd)) {
156                 break;
157             }
158         }
159 
160         var clone = docFrag,
161             levelStartNode,
162             levelClone,
163             currentNode,
164             currentSibling;
165 
166         // Remove all successive sibling nodes for every node in the
167         // startParents tree.
168         for (var j = i; j < startParents.length; j++) {
169             levelStartNode = startParents[j];
170 
171             // For Extract and Clone, we must clone this level.
172             if (action > 0 && !levelStartNode.equals(startNode)) {
173                 // action = 0 = Delete
174                 levelClone = clone.appendChild(levelStartNode.clone()[0]);
175             } else {
176                 levelClone = null;
177             }
178 
179             // 开始节点的路径所在父节点不能 clone(TRUE)，其他节点（结束节点路径左边的节点）可以直接 clone(true)
180             currentNode = levelStartNode[0].nextSibling;
181 
182             var endParentJ = endParents[ j ],
183                 domEndNode = endNode[0],
184                 domEndParentJ = endParentJ[0];
185 
186             while (currentNode) {
187                 // Stop processing when the current node matches a node in the
188                 // endParents tree or if it is the endNode.
189                 if (domEndParentJ == currentNode || domEndNode == currentNode) {
190                     break;
191                 }
192 
193                 // Cache the next sibling.
194                 currentSibling = currentNode.nextSibling;
195 
196                 // If cloning, just clone it.
197                 if (action == 2) {
198                     // 2 = Clone
199                     clone.appendChild(currentNode.cloneNode(TRUE));
200                 } else {
201                     // Both Delete and Extract will remove the node.
202                     DOM._4e_remove(currentNode);
203 
204                     // When Extracting, move the removed node to the docFrag.
205                     if (action == 1) {
206                         // 1 = Extract
207                         clone.appendChild(currentNode);
208                     }
209                 }
210 
211                 currentNode = currentSibling;
212             }
213             // 开始节点的路径所在父节点不能 clone(TRUE)，要在后面深入子节点处理
214             if (levelClone) {
215                 clone = levelClone;
216             }
217         }
218 
219         clone = docFrag;
220 
221         // Remove all previous sibling nodes for every node in the
222         // endParents tree.
223         for (var k = i; k < endParents.length; k++) {
224             levelStartNode = endParents[ k ];
225 
226             // For Extract and Clone, we must clone this level.
227             if (action > 0 && !levelStartNode.equals(endNode)) {
228                 // action = 0 = Delete
229                 // 浅复制
230                 levelClone = clone.appendChild(levelStartNode.clone()[0]);
231             } else {
232                 levelClone = null;
233             }
234 
235             // The processing of siblings may have already been done by the parent.
236             if (
237                 !startParents[ k ] ||
238                     // 前面 startParents 循环已经处理过了
239                     levelStartNode[0].parentNode != startParents[ k ][0].parentNode
240                 ) {
241                 currentNode = levelStartNode[0].previousSibling;
242                 while (currentNode) {
243                     // Cache the next sibling.
244                     currentSibling = currentNode.previousSibling;
245 
246                     // If cloning, just clone it.
247                     if (action == 2) {    // 2 = Clone
248                         clone.insertBefore(currentNode.cloneNode(TRUE),
249                             clone.firstChild);
250                     } else {
251                         // Both Delete and Extract will remove the node.
252                         DOM._4e_remove(currentNode);
253 
254                         // When Extracting, mode the removed node to the docFrag.
255                         if (action == 1) {
256                             // 1 = Extract
257                             clone.insertBefore(currentNode, clone.firstChild);
258                         }
259                     }
260 
261                     currentNode = currentSibling;
262                 }
263             }
264 
265             if (levelClone) {
266                 clone = levelClone;
267             }
268         }
269         // 2 = Clone.
270         if (action == 2) {
271 
272             // No changes in the DOM should be done, so fix the split text (if any).
273 
274             if (hasSplitStart) {
275                 var startTextNode = startNode[0];
276                 if (startTextNode.nodeType == KEN.NODE_TEXT
277                     && startTextNode.nextSibling
278                     // careful, next sibling should be text node
279                     && startTextNode.nextSibling.nodeType == KEN.NODE_TEXT) {
280                     startTextNode.data += startTextNode.nextSibling.data;
281                     startTextNode.parentNode.removeChild(startTextNode.nextSibling);
282                 }
283             }
284 
285             if (hasSplitEnd) {
286                 var endTextNode = endNode[0];
287                 if (endTextNode.nodeType == KEN.NODE_TEXT &&
288                     endTextNode.previousSibling &&
289                     endTextNode.previousSibling.nodeType == KEN.NODE_TEXT) {
290                     endTextNode.previousSibling.data += endTextNode.data;
291                     endTextNode.parentNode.removeChild(endTextNode);
292                 }
293             }
294 
295         } else {
296 
297             // Collapse the range.
298             // If a node has been partially selected, collapse the range between
299             // topStart and topEnd. Otherwise, simply collapse it to the start.
300             // (W3C specs).
301             if (
302                 topStart && topEnd &&
303                     (
304                         !startNode.parent().equals(topStart.parent())
305                             ||
306                             !endNode.parent().equals(topEnd.parent())
307                         )
308                 ) {
309                 var startIndex = topStart._4e_index();
310 
311                 // If the start node is to be removed, we must correct the
312                 // index to reflect the removal.
313                 if (removeStartNode &&
314                     // startNode 和 topStart 同级
315                     topStart.parent().equals(removeStartNode.parent())) {
316                     startIndex--;
317                 }
318 
319                 self.setStart(topStart.parent(), startIndex + 1);
320             }
321 
322             // Collapse it to the start.
323             self.collapse(TRUE);
324 
325         }
326 
327         // Cleanup any marked node.
328         if (removeStartNode) {
329             startNode.remove();
330         }
331 
332         if (removeEndNode) {
333             endNode.remove();
334         }
335 
336         return docFrag;
337     }

339     function updateCollapsed(self) {
340         self.collapsed = (
341             self.startContainer &&
342                 self.endContainer &&
343                 self.startContainer[0] == self.endContainer[0] &&
344                 self.startOffset == self.endOffset );
345     }

642             /**
643              * Get node which is enclosed by range.
644              * @example
645              * <code>
646              * ^<book/><span/><book/>^
647              * =>
648              * ^<span/>^
649              * </code>
650              */
651             getEnclosedNode:function () {
652                 var walkerRange = this.clone();
653                 // Optimize and analyze the range to avoid DOM destructive nature of walker.
654                 walkerRange.optimize();
655                 if (walkerRange.startContainer[0].nodeType != KEN.NODE_ELEMENT ||
656                     walkerRange.endContainer[0].nodeType != KEN.NODE_ELEMENT) {
657                     return NULL;
658                 }
659 
660                 var walker = new Walker(walkerRange),
661                     isNotBookmarks = Walker.bookmark(false, true),
662                     isNotWhitespaces = Walker.whitespaces(TRUE), node, pre;
663 
664                 walker.evaluator = function (node) {
665                     return isNotWhitespaces(node) && isNotBookmarks(node);
666                 };
667 
668                 //深度优先遍历的第一个元素
669                 //        x
670                 //     y     z
671                 // x->y ,return y
672                 node = walker.next();
673                 walker.reset();
674                 pre = walker.previous();
675                 //前后相等，则脱一层皮 :)
676                 return node && node.equals(pre) ? node : NULL;

679             /**
680              * Shrink range to its innermost element.(make sure text content is unchanged)
681              * @param mode
682              * @param {Boolean} selectContents
683              */
684             shrink:function (mode, selectContents) {
685                 // Unable to shrink a collapsed range.
686                 var self = this;
687                 if (!self.collapsed) {
688                     mode = mode || KER.SHRINK_TEXT;
689 
690                     var walkerRange = self.clone(),
691                         startContainer = self.startContainer,
692                         endContainer = self.endContainer,
693                         startOffset = self.startOffset,
694                         endOffset = self.endOffset,
695                         // Whether the start/end boundary is movable.
696                         moveStart = TRUE,
697                         currentElement,
698                         walker,
699                         moveEnd = TRUE;
700 
701                     if (startContainer && startContainer[0].nodeType == KEN.NODE_TEXT) {
702                         if (!startOffset) {
703                             walkerRange.setStartBefore(startContainer);
704                         } else if (startOffset >= startContainer[0].nodeValue.length) {
705                             walkerRange.setStartAfter(startContainer);
706                         } else {
707                             // Enlarge the range properly to avoid walker making
708                             // DOM changes caused by trimming the text nodes later.
709                             walkerRange.setStartBefore(startContainer);
710                             moveStart = FALSE;
711                         }
712                     }
713 
714                     if (endContainer && endContainer[0].nodeType == KEN.NODE_TEXT) {
715                         if (!endOffset) {
716                             walkerRange.setEndBefore(endContainer);
717                         } else if (endOffset >= endContainer[0].nodeValue.length) {
718                             walkerRange.setEndAfter(endContainer);
719                         } else {
720                             walkerRange.setEndAfter(endContainer);
721                             moveEnd = FALSE;
722                         }
723                     }
724 
725                     if (moveStart || moveEnd) {
726 
727                         walker = new Walker(walkerRange);
728 
729                         walker.evaluator = function (node) {
730                             return node.nodeType == ( mode == KER.SHRINK_ELEMENT ?
731                                 KEN.NODE_ELEMENT : KEN.NODE_TEXT );
732                         };
733 
734                         walker.guard = function (node, movingOut) {
735                             node = node[0] || node;
736                             // Stop when we're shrink in element mode while encountering a text node.
737                             if (mode == KER.SHRINK_ELEMENT && node.nodeType == KEN.NODE_TEXT) {
738                                 return FALSE;
739                             }
740                             // Stop when we've already walked "through" an element.
741                             if (movingOut && node == currentElement) {
742                                 return FALSE;
743                             }
744                             if (!movingOut && node.nodeType == KEN.NODE_ELEMENT) {
