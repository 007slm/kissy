1 KISSY.add("editor/plugin/elementPath/index", function (S, Editor) {
  2     var Node = S.Node;
  3     var CLASS = "ke-element-path";
  4 
  5     function ElementPaths(cfg) {
  6         var self = this;
  7         self.cfg = cfg;
  8         self._cache = [];
  9         self._init();
 10     }
 11 
 12     S.augment(ElementPaths, {
 13         _init:function () {
 14             var self = this,
 15                 cfg = self.cfg,
 16                 editor = cfg.editor;
 17             self.holder = new Node("<span>");
 18             self.holder.appendTo(editor.get("statusBarEl"), undefined);
 19             editor.on("selectionChange", self._selectionChange, self);
 20             Editor.Utils.sourceDisable(editor, self);
 21         },
 22         disable:function () {
 23             this.holder.css("visibility", "hidden");
 24         },
 25         enable:function () {
 26             this.holder.css("visibility", "");
 27         },
 28         _selectionChange:function (ev) {
 29             var self = this,
 30                 cfg = self.cfg,
 31                 editor = cfg.editor,
 32                 statusDom = self.holder,
 33                 elementPath = ev.path,
 34                 elements = elementPath.elements,
 35                 element, i,
 36                 cache = self._cache;
 37             for (i = 0; i < cache.length; i++) {
 38                 cache[i].remove();
 39             }
 40             self._cache = [];
 41             // For each element into the elements path.
 42             for (i = 0; i < elements.length; i++) {
 43                 element = elements[i];
 44                 // 考虑 fake objects
 45                 var type = element.attr("_ke_real_element_type") || element.nodeName(),
 46                     a = new Node("<a " +
 47                         "href='javascript(\"" +
 48                         type + "\")' " +
 49                         "class='" +
 50                         CLASS + "'>" +
 51                         type +
 52                         "</a>");
 53                 self._cache.push(a);
 54                 (function (element) {
 55                     a.on("click", function (ev2) {
 56                         ev2.halt();
 57                         editor.focus();
 58                         setTimeout(function () {
 59                             editor.getSelection().selectElement(element);
 60                         }, 50);
 61                     });
 62                 })(element);
 63                 statusDom.prepend(a);
 64             }
 65         },
 66         destroy:function () {
 67             this.holder.remove();
 68         }
 69     });
 70 
 71     return {
 72         init:function (editor) {
 73             var elemPath = new ElementPaths({
 74                 editor:editor
 75             });
 76             editor.on("destroy", function () {
 77                 elemPath.destroy();
 78             });
 79         }
 80     };
 81 
 82 }, {
 83     requires:['editor']
 84 });
