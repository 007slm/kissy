1 /**
  2  * @fileOverview Declare config info for KISSY.
  3  * @author yiminghe@gmail.com
  4  */
  5 (function (S) {
  6     if (typeof require !== 'undefined') {
  7         return;
  8     }
  9     var Loader = S.Loader, utils = Loader.Utils;
 10     /*
 11      modify current module path
 12      <code>
 13      [
 14      [/(.+-)min(.js(\?t=\d+)?)$/,"$1$2"],
 15      [/(.+-)min(.js(\?t=\d+)?)$/,function(_,m1,m2){
 16      return m1+m2;
 17      }]
 18      ]
 19      </code>
 20      */
 21     S.configs.map = function (rules) {
 22         var self = this;
 23         return self.Config.mappedRules = (self.Config.mappedRules || []).concat(rules || []);
 24     };
 25 
 26     /*
 27      包声明
 28      biz -> .
 29      表示遇到 biz/x
 30      在当前网页路径找 biz/x.js
 31      @private
 32      */
 33     S.configs.packages = function (cfgs) {
 34         var self = this,
 35             name,
 36             base,
 37             Env = self.Env,
 38             ps = Env.packages = Env.packages || {};
 39         if (cfgs) {
 40             S.each(cfgs, function (cfg, key) {
 41                 // 兼容数组方式
 42                 name = cfg.name || key;
 43                 // 兼容 path
 44                 base = cfg.base || cfg.path;
 45 
 46                 // must be folder
 47                 if (!S.endsWith(base, "/")) {
 48                     base += "/";
 49                 }
 50 
 51                 // 注意正则化
 52                 cfg.name = name;
 53                 var baseUri = utils.resolveByPage(base);
 54                 cfg.base = baseUri.toString();
 55                 cfg.baseUri = baseUri;
 56                 cfg.SS = S;
 57                 delete cfg.path;
 58 
 59                 ps[ name ] = new Loader.Package(cfg);
 60             });
 61         }
 62     };
 63 
 64     /*
 65      只用来指定模块依赖信息.
 66      <code>
 67 
 68      KISSY.config({
 69      base:'',
 70      // dom-min.js
 71      debug:'',
 72      combine:true,
 73      tag:'',
 74      packages:{
 75      "biz1": {
 76      // path change to base
 77      base: "haha",
 78      // x.js
 79      debug:'',
 80      tag:'',
 81      combine:false,
 82      }
 83      },
 84      modules:{
 85      "biz1/main" : {
 86      requires: [ "biz1/part1" , "biz1/part2" ]
 87      }
 88      }
 89      });
 90      */
 91     S.configs.modules = function (modules) {
 92         var self = this;
 93         if (modules) {
 94             S.each(modules, function (modCfg, modName) {
 95                 utils.createModuleInfo(self, modName, modCfg);
 96                 S.mix(self.Env.mods[modName], modCfg);
 97             });
 98         }
 99     };
100 
101     S.configs.modules.order = 10;
102 
103     /*
104      KISSY 's base path.
105      */
106     S.configs.base = function (base) {
107         var self = this, baseUri, Config = self.Config;
108         if (!base) {
109             return Config.base;
110         }
111         baseUri = utils.resolveByPage(base);
112         Config.base = baseUri.toString();
113         Config.baseUri = baseUri;
114     };
115 })(KISSY);
