132                         v = "css";

142              * Get the fullpath of current module if load dynamically
143              */
144             getFullPath:function () {
145                 var self = this, t, fullpathUri, packageBaseUri;
146                 if (!self.fullpath) {
147                     packageBaseUri = self.getPackageInfo().getBaseUri();
148                     fullpathUri = packageBaseUri.resolve(self.getPath());
149                     if (t = self.getTag()) {
150                         fullpathUri.query.set("t", t);
151                     }
152                     self.fullpath = Loader.Utils.getMappedPath(self.SS, fullpathUri.toString());
153                 }
154                 return self.fullpath;
155             },

209     function defaultComponentJsName(m) {
210         var name = m.name,
211             extname = Path.extname(name) || ".js",
212             min = "-min";
213 
214         name = Path.join(Path.dirname(name), Path.basename(name, extname));
215 
216         if (m.getPackageInfo().isDebug()) {
217             min = "";
218         }
219         return name + min + extname;
220     }

222     function getPackageInfo(self, mod) {
223         var modName = mod.name,
224             Env = self.Env,
225             packages = Env.packages || {},
226             pName = "",
227             p,
228             packageDesc;
229 
230         for (p in packages) {
231             if (packages.hasOwnProperty(p)) {
232                 // longest match
233                 if (S.startsWith(modName, p) &&
234                     p.length > pName.length) {
235                     pName = p;
236                 }
237             }
238         }
239 
240         packageDesc = packages[pName] ||
241             Env.defaultPackage ||
242             (Env.defaultPackage = new Loader.Package({
243                 SS:self,
244                 // need packageName as key
245                 name:''
246             }));
247 
248         return packageDesc;
249     }
