 21             debugUrl:function (url) {
 22                 url = url.replace(/-min\.(js|css)/i, ".$1");
 23                 if (!S["Config"]['debug']) {
 24                     url = url.replace(/\.(js|css)/i, "-min.$1");
 25                 }
 26                 if (url.indexOf("?t") == -1) {
 27                     if (url.indexOf("?") != -1) {
 28                         url += "&";
 29                     } else {
 30                         url += "?";
 31                     }
 32                     url += "t=" + encodeURIComponent("@TIMESTAMP@");
 33                 }
 34                 if (S.startsWith(url, "/")) {
 35                     url = url.substring(1);
 36                 }
 37                 return Editor["Config"].base + url;

 54             /**
 55              * srcDoc 中的位置在 destDoc 的对应位置
 56              * @param x {number}
 57              * @param y {number}
 58              * @param srcDoc {Document}
 59              * @param destDoc {Document}
 60              * @return 在最终文档中的位置
 61              */
 62             getXY:function (x, y, srcDoc, destDoc) {
 63                 var currentWindow = srcDoc.defaultView || srcDoc.parentWindow;
 64 
 65                 //x,y相对于当前iframe文档,防止当前iframe有滚动条
 66                 x -= DOM.scrollLeft(currentWindow);
 67                 y -= DOM.scrollTop(currentWindow);
 68                 if (destDoc) {
 69                     var refWindow = destDoc.defaultView || destDoc.parentWindow;
 70                     if (currentWindow != refWindow && currentWindow['frameElement']) {
 71                         //note:when iframe is static ,still some mistake
 72                         var iframePosition = DOM.offset(currentWindow['frameElement'], undefined, refWindow);
 73                         x += iframePosition.left;
 74                         y += iframePosition.top;
 75                     }
 76                 }
 77                 return {left:x, top:y};
 78             },

 98             /**
 99              * 是否两个数组完全相同
100              * @param arrayA {Array}
101              * @param arrayB {Array}
102              * @return {Boolean}
103              */
104             arrayCompare:function (arrayA, arrayB) {
105                 if (!arrayA && !arrayB)
106                     return TRUE;
107 
108                 if (!arrayA || !arrayB || arrayA.length != arrayB.length)
109                     return FALSE;
110 
111                 for (var i = 0; i < arrayA.length; i++) {
112                     if (arrayA[ i ] !== arrayB[ i ])
113                         return FALSE;
114                 }
115 
116                 return TRUE;
117             },

119             /**
120              * 根据dom路径得到某个节点
121              * @param doc {Document}
122              * @param address {Array.<number>}
123              * @param normalized {Boolean}
124              * @return {NodeList}
125              */
126             getByAddress:function (doc, address, normalized) {
127                 var $ = doc.documentElement;
128 
129                 for (var i = 0; $ && i < address.length; i++) {
130                     var target = address[ i ];
131 
132                     if (!normalized) {
133                         $ = $.childNodes[ target ];
134                         continue;
135                     }
136 
137                     var currentIndex = -1;
138 
139                     for (var j = 0; j < $.childNodes.length; j++) {
140                         var candidate = $.childNodes[ j ];
141 
142                         if (normalized === TRUE &&
143                             candidate.nodeType == 3 &&
144                             candidate.previousSibling &&
145                             candidate.previousSibling.nodeType == 3) {
146                             continue;
147                         }
148 
149                         currentIndex++;
150 
151                         if (currentIndex == target) {
152                             $ = candidate;
153                             break;
154                         }
155                     }
156                 }
157 
158                 return $ ? new Node($) : NULL;
159             },

325             /**
326              *
327              * @param o {Object} 提交 form 配置
328              * @param ps {Object} 动态参数
329              * @param url {string} 目的地 url
330              */
331             doFormUpload:function (o, ps, url) {
332                 var id = S.guid("form-upload-");
333                 var frame = document.createElement('iframe');
334                 frame.id = id;
335                 frame.name = id;
336                 frame.className = 'ke-hidden';
337 
338                 var srcScript = 'document.open();' +
339                     // The document domain must be set any time we
340                     // call document.open().
341                     ( Utils.isCustomDomain() ? ( 'document.domain="' + document.domain + '";' ) : '' ) +
342                     'document.close();';
343                 if (UA['ie']) {
344                     frame.src = UA['ie'] ? 'javascript:void(function(){' + encodeURIComponent(srcScript) + '}())' : '';
345                 }
346                 S.log("doFormUpload : " + frame.src);
347                 document.body.appendChild(frame);
348 
349                 if (UA['ie']) {
350                     document['frames'][id].name = id;
351                 }
352 
353                 var form = o.form,
354                     buf = {
355                         target:DOM.attr(form, "target"),
356                         method:DOM.attr(form, "method"),
357                         encoding:DOM.attr(form, "encoding"),
358                         enctype:DOM.attr(form, "enctype"),
359                         action:DOM.attr(form, "action")
360                     };
361                 DOM.attr(form, {
362                     target:id,
363                     method:"post",
364                     enctype:'multipart/form-data',
365                     encoding:'multipart/form-data'
366                 });
367                 if (url) {
368                     DOM.attr(form, "action", url);
369                 }
370                 var hiddens, hd;
371                 if (ps) { // add dynamic params
372                     hiddens = [];
373                     ps = Editor.Utils.normParams(ps);
374                     for (var k in ps) {
375                         if (ps.hasOwnProperty(k)) {
376                             hd = document.createElement('input');
377                             hd.type = 'hidden';
378                             hd.name = k;
379                             hd.value = ps[k];
380                             form.appendChild(hd);
381                             hiddens.push(hd);
382                         }
383                     }
384                 }
385 
386                 function cb() {
387                     var r = {  // bogus response object
388                         responseText:'',
389                         responseXML:NULL
390                     };
391 
392                     r.argument = o ? o.argument : NULL;
393 
394                     try { //
395                         var doc;
396                         if (UA['ie']) {
397                             doc = frame.contentWindow.document;
398                         } else {
399                             doc = (frame.contentDocument || window.frames[id].document);
400                         }
401 
402                         if (doc && doc.body) {
403                             r.responseText = S.trim(DOM.text(doc.body));
404                         }
405                         if (doc && doc['XMLDocument']) {
406                             r.responseXML = doc['XMLDocument'];
407                         } else {
408                             r.responseXML = doc;
409                         }
410 
411                     }
412                     catch (e) {
413                         // ignore
414                         // 2010-11-15 由于外边设置了document.domain导致读不到数据抛异常
415                         S.log("after data returns error ,maybe domain problem:");
416                         S.log(e, "error");
417                     }
418 
419                     Event.remove(frame, 'load', cb);
420                     o.callback && o.callback(r);
421 
422                     setTimeout(function () {
423                         DOM._4e_remove(frame);
424                     }, 100);
425 
426                 }
427 
428                 Event.on(frame, 'load', cb);
429 
430                 form.submit();
431 
432                 DOM.attr(form, buf);
433 
434                 if (hiddens) { // remove dynamic params
435                     for (var i = 0, len = hiddens.length; i < len; i++) {
436                         DOM._4e_remove(hiddens[i]);
437                     }
438                 }
439                 return frame;
440             },
