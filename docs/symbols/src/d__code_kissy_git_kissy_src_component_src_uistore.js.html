1 /**
  2  * @fileOverview storage for component's css
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("component/uistore", function (S) {
  6     var uis = {
  7         // 不带前缀 prefixCls
  8         /*
  9          "menu" :{
 10          priority:0,
 11          ui:Menu
 12          }
 13          */
 14     };
 15 
 16     function getUIConstructorByCssClass(cls) {
 17         var cs = cls.split(/\s+/), p = -1, ui = null;
 18         for (var i = 0; i < cs.length; i++) {
 19             var uic = uis[cs[i]];
 20             if (uic && uic.priority > p) {
 21                 ui = uic.ui;
 22             }
 23         }
 24         return ui;
 25     }
 26 
 27     function getCssClassByUIConstructor(constructor) {
 28         for (var u in uis) {
 29             var ui = uis[u];
 30             if (ui.ui == constructor) {
 31                 return u;
 32             }
 33         }
 34         return 0;
 35     }
 36 
 37     function setUIConstructorByCssClass(cls, uic) {
 38         uis[cls] = uic;
 39     }
 40 
 41 
 42     function getCssClassWithPrefix(cls) {
 43         var cs = S.trim(cls).split(/\s+/);
 44         for (var i = 0; i < cs.length; i++) {
 45             if (cs[i]) {
 46                 cs[i] = this.get("prefixCls") + cs[i];
 47             }
 48         }
 49         return cs.join(" ");
 50     }
 51 
 52     /**
 53      * @name UIStore
 54      * @namespace
 55      * @memberOf Component
 56      */
 57     var UIStore = /** @lends Component.UIStore */{
 58         getCssClassWithPrefix:getCssClassWithPrefix,
 59         /**
 60          * Get css class name for this component constructor.
 61          * @param {Function} constructor Component's constructor.
 62          * @type {Function}
 63          * @return {String}
 64          * @function
 65          */
 66         getCssClassByUIConstructor:getCssClassByUIConstructor,
 67         /**
 68          * Get component constructor by css class name.
 69          * @param {String} classNames Class names separated by space.
 70          * @type {Function}
 71          * @return {Function}
 72          * @function
 73          */
 74         getUIConstructorByCssClass:getUIConstructorByCssClass,
 75         /**
 76          * Associate css class with component constructor.
 77          * @type {Function}
 78          * @param {String} className Component's class name.
 79          * @param {Function} componentConstructor Component's constructor.
 80          * @function
 81          */
 82         setUIConstructorByCssClass:setUIConstructorByCssClass,
 83 
 84         /**
 85          * Component's constructor's priority enum.
 86          * Used for getCssClassByUIConstructor, when multiple component constructors are found.
 87          * @type Object
 88          */
 89         PRIORITY:{
 90             LEVEL1:10,
 91             LEVEL2:20,
 92             LEVEL3:30,
 93             LEVEL4:40,
 94             LEVEL5:50,
 95             LEVEL6:60
 96         }
 97     };
 98 
 99     UIStore.getCssClassWithPrefix = getCssClassWithPrefix;
100 
101     return UIStore;
102 });
