1 /**
  2  * @fileOverview enhanced base for model with sync
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("mvc/model", function (S, Base, mvcSync) {
  6 
  7     var blacklist = [
  8         "idAttribute",
  9         "clientId",
 10         "urlRoot",
 11         "url",
 12         "parse",
 13         "sync"
 14     ];
 15 
 16     /**
 17      * Model represent a data record.
 18      * @class
 19      * @memberOf MVC
 20      * @extends Base
 21      */
 22     function Model() {
 23         var self = this;
 24         Model.superclass.constructor.apply(self, arguments);
 25         /*
 26          should bubble to its collections
 27          */
 28         self.publish("*Change", {
 29             bubbles:1
 30         });
 31         self.collections = {};
 32     }
 33 
 34     S.extend(Model, Base,
 35         /**
 36          * @lends MVC.Model#
 37          */
 38         {
 39 
 40             /**
 41              * Add current model instance to a specified collection.
 42              * @param {MVC.Collection} c
 43              */
 44             addToCollection:function (c) {
 45                 this.collections[S.stamp(c)] = c;
 46                 this.addTarget(c);
 47             },
 48             /**
 49              * Remove current model instance from a specified collection.
 50              * @param {MVC.Collection} c
 51              */
 52             removeFromCollection:function (c) {
 53                 delete this.collections[S.stamp(c)];
 54                 this.removeTarget(c);
 55             },
 56 
 57             /**
 58              * Get current model 's id.
 59              */
 60             getId:function () {
 61                 return this.get(this.get("idAttribute"));
 62             },
 63 
 64             /**
 65              * Set current model 's id.
 66              * @param id
 67              */
 68             setId:function (id) {
 69                 return this.set(this.get("idAttribute"), id);
 70             },
 71 
 72             __set:function () {
 73                 this.__isModified = 1;
 74                 return Model.superclass.__set.apply(this, arguments);
 75             },
 76 
 77             /**
 78              * whether it is newly created.
 79              * @return {Boolean}
 80              */
 81             isNew:function () {
 82                 return !this.getId();
 83             },
 84 
 85             /**
 86              * whether has been modified since last save.
 87              * @return {Boolean}
 88              */
 89             isModified:function () {
 90                 return !!(this.isNew() || this.__isModified);
 91             },
 92 
 93             /**
 94              * destroy this model and sync with server.
 95              * @param {Object} [opts] destroy config.
 96              * @param {Function} opts.success callback when action is done successfully.
 97              * @param {Function} opts.error callback when error occurs at action.
 98              * @param {Function} opts.complete callback when action is complete.
 99              */
100             destroy:function (opts) {
101                 var self = this;
102                 opts = opts || {};
103                 var success = opts.success;
104                 /**
105                  * @ignore
106                  */
107                 opts.success = function (resp) {
108                     var lists = self.collections;
109                     if (resp) {
110                         var v = self.get("parse").call(self, resp);
111                         if (v) {
112                             self.set(v, opts);
113                         }
114                     }
115                     for (var l in lists) {
116                         lists[l].remove(self, opts);
117                         self.removeFromCollection(lists[l]);
118                     }
119                     self.fire("destroy");
120                     success && success.apply(this, arguments);
121                 };
122                 if (!self.isNew() && opts['delete']) {
123                     self.get("sync").call(self, self, 'delete', opts);
124                 } else {
125                     opts.success();
126                     if (opts.complete) {
127                         opts.complete();
128                     }
129                 }
130 
131                 return self;
132             },
133 
134             /**
135              * Load model data from server.
136              * @param {Object} opts Load config.
137              * @param {Function} opts.success callback when action is done successfully.
138              * @param {Function} opts.error callback when error occurs at action.
139              * @param {Function} opts.complete callback when action is complete.
140              */
141             load:function (opts) {
142                 var self = this;
143                 opts = opts || {};
144                 var success = opts.success;
145                 /**
146                  * @ignore
147                  */
148                 opts.success = function (resp) {
149                     if (resp) {
150                         var v = self.get("parse").call(self, resp);
151                         if (v) {
152                             self.set(v, opts);
153                         }
154                     }
155                     self.__isModified = 0;
156                     success && success.apply(this, arguments);
157                 };
158                 self.get("sync").call(self, self, 'read', opts);
159                 return self;
160             },
161 
162             /**
163              * Save current model 's data to server using sync.
164              * @param {Object} opts Save config.
165              * @param {Function} opts.success callback when action is done successfully.
166              * @param {Function} opts.error callback when error occurs at action.
167              * @param {Function} opts.complete callback when action is complete.
168              */
169             save:function (opts) {
170                 var self = this;
171                 opts = opts || {};
172                 var success = opts.success;
173                 /**
174                  * @ignore
175                  */
176                 opts.success = function (resp) {
177                     if (resp) {
178                         var v = self.get("parse").call(self, resp);
179                         if (v) {
180                             self.set(v, opts);
181                         }
182                     }
183                     self.__isModified = 0;
184                     success && success.apply(this, arguments);
185                 };
186                 self.get("sync").call(self, self, self.isNew() ? 'create' : 'update', opts);
187                 return self;
188             },
189 
190             /**
191              * Get json representation for current model.
192              * @return {Object}
193              */
194             toJSON:function () {
195                 var ret = this.getAttrVals();
196                 S.each(blacklist, function (b) {
197                     delete ret[b];
198                 });
199                 return ret;
200             }
201 
202         }, {
203             ATTRS:/**
204              * @lends MVC.Model#
205              */
206             {
207                 /**
208                  * Attribute name used to store id from server.
209                  * Default: "id".
210                  * @type String
211                  */
212                 idAttribute:{
213                     value:'id'
214                 },
215 
216                 /**
217                  * Generated client id.
218                  * Default call S.guid()
219                  * @type Function
220                  */
221                 clientId:{
222                     valueFn:function () {
223                         return S.guid("mvc-client");
224                     }
225                 },
226                 /**
227                  * Called to get url for delete/edit/new current model.
228                  * Default: collection.url+"/"+mode.id
229                  * @type Function
230                  */
231                 url:{
232                     value:url
233                 },
234                 /**
235                  * If current model does not belong to any collection.
236                  * Use this attribute value as collection.url in {@link MVC.Model#url}
237                  * @type String
238                  */
239                 urlRoot:{
240                     value:""
241                 },
242                 /**
243                  * Sync model data with server.
244                  * Default to call {@link MVC.sync}
245                  * @type Function
246                  */
247                 sync:{
248                     value:function () {
249                         mvcSync.apply(this, arguments);
250                     }
251                 },
252                 /**
253                  * parse json from server to get attr/value pairs.
254                  * Default to return raw data from server.
255                  * @type function
256                  */
257                 parse:{
258                     value:function (resp) {
259                         return resp;
260                     }
261                 }
262             }
263         });
264 
265     function getUrl(o) {
266         var u;
267         if (o && (u = o.get("url"))) {
268             if (S.isString(u)) {
269                 return u;
270             }
271             return u.call(o);
272         }
273         return u;
274     }
275 
276     function url() {
277         var c,
278             cv,
279             collections = this.collections;
280         for (c in collections) {
281             if (collections.hasOwnProperty(c)) {
282                 cv = collections[c];
283                 break;
284             }
285         }
286         var base = getUrl(cv) || this.get("urlRoot");
287 
288         if (this.isNew()) {
289             return base;
290         }
291 
292         base = base + (base.charAt(base.length - 1) == '/' ? '' : '/');
293         return base + encodeURIComponent(this.getId()) + "/";
294     }
295 
296     return Model;
297 
298 }, {
299     requires:['base', './sync']
300 });
