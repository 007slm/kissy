1 /**
  2  * @fileOverview io shortcut
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("ajax", function (S, serializer, io, XhrObject) {
  6     var undef = undefined;
  7     // some shortcut
  8     S.mix(io,
  9 
 10         /**
 11          * @lends io
 12          */
 13         {
 14             XhrObject:XhrObject,
 15             /**
 16              * form 序列化
 17              * @param formElement {HTMLFormElement} 将要序列化的 form 元素
 18              */
 19             serialize:serializer.serialize,
 20 
 21             /**
 22              * get 请求
 23              * @param url
 24              * @param data
 25              * @param callback
 26              * @param [dataType]
 27              * @param [_t]
 28              */
 29             get:function (url, data, callback, dataType, _t) {
 30                 // data 参数可省略
 31                 if (S.isFunction(data)) {
 32                     dataType = callback;
 33                     callback = data;
 34                     data = undef;
 35                 }
 36 
 37                 return io({
 38                     type:_t || "get",
 39                     url:url,
 40                     data:data,
 41                     success:callback,
 42                     dataType:dataType
 43                 });
 44             },
 45 
 46             /**
 47              * post 请求
 48              * @param url
 49              * @param data
 50              * @param callback
 51              * @param [dataType]
 52              */
 53             post:function (url, data, callback, dataType) {
 54                 if (S.isFunction(data)) {
 55                     dataType = callback;
 56                     callback = data;
 57                     data = undef;
 58                 }
 59                 return io.get(url, data, callback, dataType, "post");
 60             },
 61 
 62             /**
 63              * jsonp 请求
 64              * @param url
 65              * @param data
 66              * @param callback
 67              */
 68             jsonp:function (url, data, callback) {
 69                 if (S.isFunction(data)) {
 70                     callback = data;
 71                     data = undef;
 72                 }
 73                 return io.get(url, data, callback, "jsonp");
 74             },
 75 
 76             // 和 S.getScript 保持一致
 77             // 更好的 getScript 可以用
 78             /*
 79              io({
 80              dataType:'script'
 81              });
 82              */
 83             getScript:S.getScript,
 84 
 85             /**
 86              * 获取 json 数据
 87              * @param url
 88              * @param data
 89              * @param callback
 90              */
 91             getJSON:function (url, data, callback) {
 92                 if (S.isFunction(data)) {
 93                     callback = data;
 94                     data = undef;
 95                 }
 96                 return io.get(url, data, callback, "json");
 97             },
 98 
 99             /**
100              * 无刷新上传文件
101              * @param url
102              * @param form
103              * @param data
104              * @param callback
105              * @param [dataType]
106              */
107             upload:function (url, form, data, callback, dataType) {
108                 if (S.isFunction(data)) {
109                     dataType = callback;
110                     callback = data;
111                     data = undef;
112                 }
113                 return io({
114                     url:url,
115                     type:'post',
116                     dataType:dataType,
117                     form:form,
118                     data:data,
119                     success:callback
120                 });
121             }
122         });
123 
124     return io;
125 }, {
126     requires:[
127         "ajax/form-serializer",
128         "ajax/base",
129         "ajax/xhrobject",
130         "ajax/xhr",
131         "ajax/script",
132         "ajax/jsonp",
133         "ajax/form",
134         "ajax/iframe-upload"]
135 });
