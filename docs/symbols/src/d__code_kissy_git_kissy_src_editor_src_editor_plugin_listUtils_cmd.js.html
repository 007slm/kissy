1 /**
  2  * Add ul and ol command identifier for KISSY Editor.
  3  * @author yiminghe@gmail.com

  5 KISSY.add("editor/plugin/listUtils/cmd", function (S, Editor, ListUtils, undefined) {
  6 
  7     var insertUnorderedList = "insertUnorderedList",
  8         insertOrderedList = "insertOrderedList",
  9         listNodeNames = {ol:insertOrderedList, ul:insertUnorderedList},
 10         KER = Editor.RANGE,
 11         ElementPath = Editor.ElementPath,
 12         Walker = Editor.Walker,
 13         KEN = Editor.NODE,
 14         UA = S.UA,
 15         Node = S.Node,
 16         DOM = S.DOM,
 17         headerTagRegex = /^h[1-6]$/;
 18 
 19     function ListCommand(type) {
 20         this.type = type;
 21     }
 22 
 23     ListCommand.prototype = {
 24 
 25         changeListType:function (editor, groupObj, database, listsCreated) {
 26             // This case is easy...
 27             // 1. Convert the whole list into a one-dimensional array.
 28             // 2. Change the list type by modifying the array.
 29             // 3. Recreate the whole list by converting the array to a list.
 30             // 4. Replace the original list with the recreated list.
 31             var listArray = ListUtils.listToArray(groupObj.root, database,
 32                 undefined, undefined, undefined),
 33                 selectedListItems = [];
 34 
 35             for (var i = 0; i < groupObj.contents.length; i++) {
 36                 var itemNode = groupObj.contents[i];
 37                 itemNode = itemNode.closest('li', undefined);
 38                 if ((!itemNode || !itemNode[0]) ||
 39                     itemNode.data('list_item_processed'))
 40                     continue;
 41                 selectedListItems.push(itemNode);
 42                 itemNode._4e_setMarker(database, 'list_item_processed', true, undefined);
 43             }
 44 
 45             var fakeParent = new Node(groupObj.root[0].ownerDocument.createElement(this.type));
 46             for (i = 0; i < selectedListItems.length; i++) {
 47                 var listIndex = selectedListItems[i].data('listarray_index');
 48                 listArray[listIndex].parent = fakeParent;
 49             }
 50             var newList = ListUtils.arrayToList(listArray, database, null, "p");
 51             var child, length = newList.listNode.childNodes.length;
 52             for (i = 0; i < length &&
 53                 ( child = new Node(newList.listNode.childNodes[i]) ); i++) {
 54                 if (child.nodeName() == this.type)
 55                     listsCreated.push(child);
 56             }
 57             newList.listNode.insertBefore(groupObj.root);
 58             groupObj.root.remove();
 59         },
 60 
 61         createList:function (editor, groupObj, listsCreated) {
 62             var contents = groupObj.contents,
 63                 doc = groupObj.root[0].ownerDocument,
 64                 listContents = [];
 65 
 66             // It is possible to have the contents returned by DomRangeIterator to be the same as the root.
 67             // e.g. when we're running into table cells.
 68             // In such a case, enclose the childNodes of contents[0] into a <div>.
 69             if (contents.length == 1
 70                 && contents[0][0] === groupObj.root[0]) {
 71                 var divBlock = new Node(doc.createElement('div'));
 72                 contents[0][0].nodeType != KEN.NODE_TEXT &&
 73                 contents[0]._4e_moveChildren(divBlock, undefined, undefined);
 74                 contents[0][0].appendChild(divBlock[0]);
 75                 contents[0] = divBlock;
 76             }
 77 
 78             // Calculate the common parent node of all content blocks.
 79             var commonParent = groupObj.contents[0].parent();
 80 
 81             for (var i = 0; i < contents.length; i++) {
 82                 commonParent = commonParent._4e_commonAncestor(contents[i].parent(), undefined);
 83             }
 84 
 85             // We want to insert things that are in the same tree level only,
 86             // so calculate the contents again
 87             // by expanding the selected blocks to the same tree level.
 88             for (i = 0; i < contents.length; i++) {
 89                 var contentNode = contents[i],
 90                     parentNode;
 91                 while (( parentNode = contentNode.parent() )) {
 92                     if (parentNode[0] === commonParent[0]) {
 93                         listContents.push(contentNode);
 94                         break;
 95                     }
 96                     contentNode = parentNode;
 97                 }
 98             }
 99 
100             if (listContents.length < 1)
101                 return;
102 
103             // Insert the list to the DOM tree.
104             var insertAnchor = new Node(
105                 listContents[ listContents.length - 1 ][0].nextSibling),
106                 listNode = new Node(doc.createElement(this.type));
107 
108             listsCreated.push(listNode);
109             while (listContents.length) {
110                 var contentBlock = listContents.shift(),
111                     listItem = new Node(doc.createElement('li'));
112 
113                 // Preserve heading structure when converting to list item. (#5271)
114                 if (headerTagRegex.test(contentBlock.nodeName())) {
115                     listItem[0].appendChild(contentBlock[0]);
116                 } else {
117                     contentBlock._4e_copyAttributes(listItem, undefined, undefined);
118                     contentBlock._4e_moveChildren(listItem, undefined, undefined);
119                     contentBlock.remove();
120                 }
121                 listNode[0].appendChild(listItem[0]);
122 
123                 // Append a bogus BR to force the LI to render at full height
124                 if (!UA['ie'])
125                     listItem._4e_appendBogus(undefined);
126             }
127             if (insertAnchor[0]) {
128                 listNode.insertBefore(insertAnchor, undefined);
129             } else {
130                 commonParent.append(listNode);
131             }
132         },
133 
134         removeList:function (editor, groupObj, database) {
135             // This is very much like the change list type operation.
136             // Except that we're changing the selected items' indent to -1 in the list array.
137             var listArray = ListUtils.listToArray(groupObj.root, database,
138                 undefined, undefined, undefined),
139                 selectedListItems = [];
140 
141             for (var i = 0; i < groupObj.contents.length; i++) {
142                 var itemNode = groupObj.contents[i];
143                 itemNode = itemNode.closest('li', undefined);
144                 if (!itemNode || itemNode.data('list_item_processed'))
145                     continue;
146                 selectedListItems.push(itemNode);
147                 itemNode._4e_setMarker(database, 'list_item_processed', true, undefined);
148             }
149 
150             var lastListIndex = null;
151 
152             for (i = 0; i < selectedListItems.length; i++) {
153                 var listIndex = selectedListItems[i].data('listarray_index');
154                 listArray[listIndex].indent = -1;
155                 lastListIndex = listIndex;
156             }
157 
158             // After cutting parts of the list out with indent=-1, we still have to maintain the array list
159             // model's nextItem.indent <= currentItem.indent + 1 invariant. Otherwise the array model of the
160             // list cannot be converted back to a real DOM list.
161             for (i = lastListIndex + 1; i < listArray.length; i++) {
162                 //if (listArray[i].indent > listArray[i - 1].indent + 1) {
163                 //modified by yiminghe
164                 if (listArray[i].indent > Math.max(listArray[i - 1].indent, 0)) {
165                     var indentOffset = listArray[i - 1].indent + 1 -
166                         listArray[i].indent;
167                     var oldIndent = listArray[i].indent;
168                     while (listArray[i]
169                         && listArray[i].indent >= oldIndent) {
170                         listArray[i].indent += indentOffset;
171                         i++;
172                     }
173                     i--;
174                 }
175             }
176 
177             var newList = ListUtils.arrayToList(listArray, database, null, "p");
178 
179             // Compensate <br> before/after the list node if the surrounds are non-blocks.(#3836)
180             var docFragment = newList.listNode, boundaryNode, siblingNode;
181 
182             function compensateBrs(isStart) {
183                 if (( boundaryNode = new Node(docFragment[ isStart ? 'firstChild' : 'lastChild' ]) )
184                     && !( boundaryNode[0].nodeType == KEN.NODE_ELEMENT &&
185                     boundaryNode._4e_isBlockBoundary(undefined, undefined) )
186                     && ( siblingNode = groupObj.root[ isStart ? 'prev' : 'next' ]
187                     (Walker.whitespaces(true)) )
188                     && !( boundaryNode[0].nodeType == KEN.NODE_ELEMENT &&
189                     siblingNode._4e_isBlockBoundary({ br:1 }, undefined) )) {
190                     boundaryNode[ isStart ? 'before' : 'after' ](editor.get("document")[0].createElement('br'));
191                 }
192             }
193 
194             compensateBrs(true);
195             compensateBrs(undefined);
196             groupObj.root.before(docFragment);
197             groupObj.root.remove();
198         },
199 
200         exec:function (editor) {
201             var selection = editor.getSelection(),
202                 ranges = selection && selection.getRanges();
203 
204             // There should be at least one selected range.
205             if (!ranges || ranges.length < 1)
206                 return;
207 
208 
209             var startElement = selection.getStartElement(),
210                 currentPath = new Editor.ElementPath(startElement);
211 
212             var state = queryActive(this.type, currentPath);
213 
214             var bookmarks = selection.createBookmarks(true);
215 
216             // Group the blocks up because there are many cases where multiple lists have to be created,
217             // or multiple lists have to be cancelled.
218             var listGroups = [],
219                 database = {};
220             while (ranges.length > 0) {
221                 var range = ranges.shift();
222 
223                 var boundaryNodes = range.getBoundaryNodes(),
224                     startNode = boundaryNodes.startNode,
225                     endNode = boundaryNodes.endNode;
226 
227                 if (startNode[0].nodeType == KEN.NODE_ELEMENT && startNode.nodeName() == 'td')
228                     range.setStartAt(boundaryNodes.startNode, KER.POSITION_AFTER_START);
229 
230                 if (endNode[0].nodeType == KEN.NODE_ELEMENT && endNode.nodeName() == 'td')
231                     range.setEndAt(boundaryNodes.endNode, KER.POSITION_BEFORE_END);
232 
233                 var iterator = range.createIterator(),
234                     block;
235 
236                 iterator.forceBrBreak = false;
237 
238                 while (( block = iterator.getNextParagraph() )) {
239 
240                     // Avoid duplicate blocks get processed across ranges.
241                     if (block.data('list_block'))
242                         continue;
243                     else
244                         block._4e_setMarker(database, 'list_block', 1, undefined);
245 
246 
247                     var path = new ElementPath(block),
248                         pathElements = path.elements,
249                         pathElementsCount = pathElements.length,
250                         listNode = null,
251                         processedFlag = false,
252                         blockLimit = path.blockLimit,
253                         element;
254 
255                     // First, try to group by a list ancestor.
256                     //2010-11-17 :
257                     //注意从上往下，从body开始找到最早的list祖先，从那里开始重建!!!
258                     for (var i = pathElementsCount - 1; i >= 0 &&
259                         ( element = pathElements[ i ] ); i--) {
260                         if (listNodeNames[ element.nodeName() ]
261                             && blockLimit.contains(element))     // Don't leak outside block limit (#3940).
262                         {
263                             // If we've encountered a list inside a block limit
264                             // The last group object of the block limit element should
265                             // no longer be valid. Since paragraphs after the list
266                             // should belong to a different group of paragraphs before
267                             // the list. (Bug #1309)
268                             blockLimit.removeData('list_group_object');
269 
270                             var groupObj = element.data('list_group_object');
271                             if (groupObj)
272                                 groupObj.contents.push(block);
273                             else {
274                                 groupObj = { root:element, contents:[ block ] };
275                                 listGroups.push(groupObj);
276                                 element._4e_setMarker(database, 'list_group_object', groupObj, undefined);
277                             }
278                             processedFlag = true;
279                             break;
280                         }
281                     }
282 
283                     if (processedFlag) {
284                         continue;
285                     }
286 
287                     // No list ancestor? Group by block limit.
288                     var root = blockLimit || path.block;
289                     if (root.data('list_group_object')) {
290                         root.data('list_group_object').contents.push(block);
291                     } else {
292                         groupObj = { root:root, contents:[ block ] };
293                         root._4e_setMarker(database, 'list_group_object', groupObj, undefined);
294                         listGroups.push(groupObj);
295                     }
296                 }
297             }
298 
299             // Now we have two kinds of list groups, groups rooted at a list, and groups rooted at a block limit element.
300             // We either have to build lists or remove lists, for removing a list does not makes sense when we are looking
301             // at the group that's not rooted at lists. So we have three cases to handle.
302             var listsCreated = [];
303             while (listGroups.length > 0) {
304                 groupObj = listGroups.shift();
305                 if (!state) {
306                     if (listNodeNames[ groupObj.root.nodeName() ]) {
307                         this.changeListType(editor, groupObj, database, listsCreated);
308                     } else {
309                         //2010-11-17
310                         //先将之前原来元素的 expando 去除，
311                         //防止 ie li 复制原来标签属性带来的输出代码多余
312                         Editor.Utils.clearAllMarkers(database);
313                         this.createList(editor, groupObj, listsCreated);
314                     }
315                 } else if (listNodeNames[ groupObj.root.nodeName() ]) {
316                     this.removeList(editor, groupObj, database);
317                 }
318             }
319 
320             var self = this;
321 
322             // For all new lists created, merge adjacent, same type lists.
323             for (i = 0; i < listsCreated.length; i++) {
324                 listNode = listsCreated[i];
325 
326                 // note by yiminghe,why not use merge sibling directly
327                 // listNode._4e_mergeSiblings();
328                 function mergeSibling(rtl, listNode) {
329                     var sibling = listNode[ rtl ?
330                         'prev' : 'next' ](Walker.whitespaces(true));
331                     if (sibling && sibling[0] &&
332                         sibling.nodeName() == self.type) {
333                         sibling.remove();
334                         // Move children order by merge direction.(#3820)
335                         sibling._4e_moveChildren(listNode, rtl ? true : false, undefined);
336                     }
337                 }
338 
339                 mergeSibling(undefined, listNode);
340                 mergeSibling(true, listNode);
341             }
342 
343             // Clean up, restore selection and update toolbar button states.
344             Editor.Utils.clearAllMarkers(database);
345             selection.selectBookmarks(bookmarks);
346         }
347     };

349     var ulCmd = new ListCommand("ul"),
350         olCmd = new ListCommand("ol");
351 
352     function queryActive(type, elementPath) {
353         var element,
354             name,
355             blockLimit = elementPath.blockLimit,
356             elements = elementPath.elements;
357         if (!blockLimit) {
358             return false;
359         }
360         // Grouping should only happen under blockLimit.(#3940).
361         if (elements) {
362             for (var i = 0; i < elements.length &&
363                 ( element = elements[ i ] ) &&
364                 element[0] !== blockLimit[0];
365                  i++) {
366                 if (listNodeNames[name = element.nodeName()]) {
367                     if (name == type) {
368                         return true;
369                     }
370                 }
371             }
372         }
373         return false;
374     }

377     return {
378         ListCommand:ListCommand,
379         queryActive:queryActive
380     };
381 
382     return {
383         init:function (editor) {
384             if (!editor.hasCommand(insertUnorderedList)) {
385                 editor.addCommand(insertUnorderedList, {
386                     exec:function (editor) {
387                         ulCmd.exec(editor);
388                     }
389                 });
390             }
391 
392             if (!editor.hasCommand(insertOrderedList)) {
393                 editor.addCommand(insertOrderedList, {
394                     exec:function (editor) {
395                         olCmd.exec(editor);
396                     }
397                 });
398             }
399 
400             var queryUl = Editor.Utils.getQueryCmd(insertUnorderedList);
401 
402             if (!editor.hasCommand(queryUl)) {
403                 editor.addCommand(queryUl, {
404                     exec:function (editor, elementPath) {
405                         return queryActive("ul", elementPath);
406                     }
407                 });
408             }
409 
410             var queryOl = Editor.Utils.getQueryCmd(insertOrderedList);
411 
412             if (!editor.hasCommand(queryOl)) {
413                 editor.addCommand(queryOl, {
414                     exec:function (editor, elementPath) {
415                         return queryActive("ol", elementPath);
416                     }
417                 });
418             }
419         }
