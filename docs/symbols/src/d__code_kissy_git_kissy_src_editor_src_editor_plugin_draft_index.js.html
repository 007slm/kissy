1 /**
  2  * draft for kissy editor

  4  */
  5 KISSY.add("editor/plugin/draft/index", function (S, Editor, localStorage, Overlay, MenuButton) {
  6     var Node = S.Node,
  7         LIMIT = 5,
  8         Event = S.Event,
  9         INTERVAL = 5,
 10         JSON = S['JSON'],
 11         DRAFT_SAVE = "ks-editor-draft-save20110503";
 12 
 13     function padding(n, l, p) {
 14         n += "";
 15         while (n.length < l) {
 16             n = p + n;
 17         }
 18         return n;
 19     }
 20 
 21     function date(d) {
 22         if (S.isNumber(d)) {
 23             d = new Date(d);
 24         }
 25         if (d instanceof Date)
 26             return [
 27                 d.getFullYear(),
 28                 "-",
 29                 padding(d.getMonth() + 1, 2, "0"),
 30                 "-",
 31                 padding(d.getDate(), 2, "0"),
 32                 " ",
 33                 //" ",
 34                 padding(d.getHours(), 2, "0"),
 35                 ":",
 36                 padding(d.getMinutes(), 2, "0"),
 37                 ":",
 38                 padding(d.getSeconds(), 2, "0")
 39                 //" ",
 40                 //" "
 41             ].join("");
 42         else
 43             return d;
 44     }
 45 
 46     function Draft(editor,config) {
 47         this.editor = editor;
 48         this.config=config;
 49         this._init();
 50     }
 51 
 52     var addRes = Editor.Utils.addRes, destroyRes = Editor.Utils.destroyRes;
 53     S.augment(Draft, {
 54 
 55         _getSaveKey:function () {
 56             var self = this,
 57                 cfg = this.config;
 58             return cfg.draft && cfg.draft['saveKey'] || DRAFT_SAVE;
 59         },
 60 
 61         /**
 62          * parse 历史记录延后，点击 select 时才开始 parse
 63          */
 64         _getDrafts:function () {
 65             var self = this;
 66             if (!self.drafts) {
 67                 var str = localStorage.getItem(self._getSaveKey()),
 68                     drafts = [];
 69                 if (str) {
 70                     /**
 71                      * 原生 localStorage 必须串行化
 72                      */
 73                     drafts = (localStorage == window.localStorage) ?
 74                         JSON.parse(decodeURIComponent(str)) : str;
 75                 }
 76                 self.drafts = drafts;
 77             }
 78             return self.drafts;
 79         },
 80         _init:function () {
 81 
 82             var self = this,
 83                 editor = self.editor,
 84                 statusbar = editor.get("statusBarEl"),
 85                 cfg = this.config;
 86             cfg.draft = cfg.draft || {};
 87             self.draftInterval = cfg.draft.interval
 88                 = cfg.draft.interval || INTERVAL;
 89             self.draftLimit = cfg.draft.limit
 90                 = cfg.draft.limit || LIMIT;
 91             var holder = new Node(
 92                 "<div class='ks-editor-draft'>" +
 93                     "<span class='ks-editor-draft-title'>" +
 94                     "内容正文每" +
 95                     cfg.draft.interval
 96                     + "分钟自动保存一次。" +
 97                     "</span>" +
 98                     "</div>").appendTo(statusbar);
 99             self.timeTip = new Node("<span class='ks-editor-draft-time'/>")
100                 .appendTo(holder);
101 
102             var save = new Node(
103                     "<a href='#' " +
104                         "onclick='return false;' " +
105                         "class='ks-editor-button ks-editor-draft-save-btn ks-inline-block' " +
106                         "style='" +
107                         "vertical-align:middle;" +
108                         "padding:1px 9px;" +
109                         "'>" +
110                         "<span class='ks-editor-draft-save'>" +
111                         "</span>" +
112                         "<span>立即保存</span>" +
113                         "</a>"
114                 ).unselectable().appendTo(holder),
115                 versions = new MenuButton({
116                     render:holder,
117                     width:"100px",
118                     prefixCls:"ks-editor-",
119                     menuCfg:{
120                         width:"225px",
121                         align:{
122                             points:['tr', 'br']
123                         }
124                     },
125                     autoRender:true,
126                     content:"恢复编辑历史"
127                 });
128             self.versions = versions;
129             // 点击才开始 parse
130             versions.on("beforeCollapsedChange", function (e) {
131                 if (!e.newValue) {
132                     versions.detach("beforeCollapsedChange", arguments.callee);
133                     self.sync();
134                 }
135             });
136             save.on("click", function (ev) {
137                 self.save(false);
138                 //如果不阻止，部分页面在ie6下会莫名奇妙把其他input的值丢掉！
139                 ev.halt();
140             });
141 
142             addRes.call(self, save);
143 
144             /*
145              监控form提交，每次提交前保存一次，防止出错
146              */
147             if (editor.get("textarea")[0].form) {
148                 (function () {
149                     var textarea = editor.get("textarea"),
150                         form = textarea[0].form;
151 
152                     function saveF() {
153                         self.save(true);
154                     }
155 
156                     Event.on(form, "submit", saveF);
157                     addRes.call(self, function () {
158                         Event.remove(form, "submit", saveF);
159                     });
160 
161                 })();
162             }
163 
164             var timer = setInterval(function () {
165                 self.save(true);
166             }, self.draftInterval * 60 * 1000);
167 
168             addRes.call(self, function () {
169                 clearInterval(timer);
170             });
171 
172             versions.on("click", self.recover, self);
173             addRes.call(self, versions);
174             self.holder = holder;
175             if (cfg.draft['helpHtml']) {
176 
177                 var help = new Node('<a ' +
178                     'tabindex="0" ' +
179                     'hidefocus="hidefocus" ' +
180                     'class="ks-editor-draft-help" ' +
181                     'title="点击查看帮助" ' +
182                     'href="javascript:void(\'点击查看帮助 \')">点击查看帮助</a>')
183                     .unselectable()
184                     .appendTo(holder);
185 
186                 help.on("click", function () {
187                     help[0].focus();
188                     if (self.helpPopup && self.helpPopup.get("visible")) {
189                         self.helpPopup.hide();
190                     } else {
191                         self._prepareHelp();
192                     }
193                 });
194                 help.on("blur", function () {
195                     self.helpPopup && self.helpPopup.hide();
196                 });
197                 self.helpBtn = help;
198                 addRes.call(self, help);
199                 Editor.Utils.lazyRun(self, "_prepareHelp", "_realHelp");
200             }
201             addRes.call(self, holder);
202         },
203         _prepareHelp:function () {
204             var self = this,
205                 editor = self.editor,
206                 cfg = self.config,
207                 draftCfg = cfg.draft,
208                 help = new Node(draftCfg['helpHtml'] || "");
209             var arrowCss = "height:0;" +
210                 "position:absolute;" +
211                 "fontSize:0;" +
212                 "width:0;" +
213                 "border:8px #000 solid;" +
214                 "border-color:#000 transparent transparent transparent;" +
215                 "border-style:solid dashed dashed dashed;";
216             var arrow = new Node("<div style='" +
217                 arrowCss +
218                 "border-top-color:#CED5E0;" +
219                 "'>" +
220                 "<div style='" +
221                 arrowCss +
222                 "left:-8px;" +
223                 "top:-10px;" +
224                 "border-top-color:white;" +
225                 "'>" +
226                 "</div>" +
227                 "</div>");
228             help.append(arrow);
229             help.css({
230                 border:"1px solid #ACB4BE",
231                 text-align:"left"
232             });
233             self.helpPopup = new Overlay({
234                 content:help,
235                 prefixCls:'ks-editor-',
236                 autoRender:true,
237                 width:help.width() + "px",
238                 zIndex:Editor.baseZIndex(Editor.zIndexManager.OVERLAY),
239                 mask:false
240             });
241             self.helpPopup.get("el")
242                 .css("border", "none");
243             self.helpPopup.arrow = arrow;
244         },
245         _realHelp:function () {
246             var win = this.helpPopup,
247                 helpBtn = this.helpBtn,
248                 arrow = win.arrow;
249             win.show();
250             var off = helpBtn.offset();
251             win.get("el").offset({
252                 left:(off.left - win.get("el").width()) + 17,
253                 top:(off.top - win.get("el").height()) - 7
254             });
255             arrow.offset({
256                 left:off.left - 2,
257                 top:off.top - 8
258             });
259         },
260         disable:function () {
261             this.holder.css("visibility", "hidden");
262         },
263         enable:function () {
264             this.holder.css("visibility", "");
265         },
266         sync:function () {
267             var self = this,
268                 i,
269                 draftLimit = self.draftLimit,
270                 timeTip = self.timeTip,
271                 versions = self.versions,
272                 drafts = self._getDrafts(),
273                 draft, tip
274 
275             if (drafts.length > draftLimit) {
276                 drafts.splice(0, drafts.length - draftLimit);
277             }
278 
279             for (i = 0; i < drafts.length; i++) {
280                 draft = drafts[i];
281                 tip = (draft.auto ? "自动" : "手动") + "保存于 : "
282                     + date(draft.date);
283                 versions.addItem({
284                     xclass:'menuitem',
285                     content:tip,
286                     value:i
287                 });
288             }
289 
290             timeTip.html(tip);
291             localStorage.setItem(self._getSaveKey(),
292                 (localStorage == window.localStorage) ?
293                     encodeURIComponent(JSON.stringify(drafts))
294                     : drafts);
295         },
296 
297         save:function (auto) {
298             var self = this,
299                 drafts = self._getDrafts(),
300                 editor = self.editor,
301             //不使用rawdata
302             //undo 只需获得可视区域内代码
303             //可视区域内代码！= 最终代码
304             //代码模式也要支持草稿功能
305             //统一获得最终代码
306                 data = editor.get("formatData");
307 
308             //如果当前内容为空，不保存版本
309             if (!data) {
310                 return;
311             }
312 
313             if (drafts[drafts.length - 1] &&
314                 data == drafts[drafts.length - 1].content) {
315                 drafts.length -= 1;
316             }
317             self.drafts = drafts.concat({
318                 content:data,
319                 date:new Date().getTime(),
320                 auto:auto
321             });
322             self.sync();
323         },
324 
325         recover:function (ev) {
326             var self = this,
327                 editor = self.editor,
328                 drafts = self._getDrafts(),
329                 v = ev.target.get("value");
330             if (confirm("确认恢复 " + date(drafts[v].date) + " 的编辑历史？")) {
331                 editor.execCommand("save");
332                 editor.set("data", drafts[v].content);
333                 editor.execCommand("save");
334             }
335             self.versions.set("collapsed", true);
336             ev.halt();
337         },
338 
339         destroy:function () {
340             destroyRes.call(this);
341         }
342     });
343 
344     function init(editor,config) {
345         var d = new Draft(editor,config);
346         editor.on("destroy", function () {
347             d.destroy();
348         });
349     }
350 
351     function DraftPlugin(config) {
352 this.config=config||{};
353     }
354 
355     S.augment(DraftPlugin, {
356         renderUI:function (editor) {
357             if (localStorage.ready) {
358                 localStorage.ready(function () {
359                     init(editor,this.config);
360                 });
361             } else {
362                 init(editor,this.config);
363             }
364         }
365     });
366 
367     return DraftPlugin;
368 
369 }, {
370     requires:["editor", "../localStorage/", "overlay", '../menubutton/']
371 });
