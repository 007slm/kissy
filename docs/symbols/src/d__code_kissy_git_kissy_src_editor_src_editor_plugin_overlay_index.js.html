1 /**
  2  * custom overlay  for kissy editor
  3  * @author yiminghe@gmail.com
  4  */
  5 KISSY.add("editor/plugin/overlay/index", function (S, Editor, Component, Overlay, Focus) {
  6     var Overlay4E =Component.define(Overlay, [Focus], {
  7     }, {
  8         ATTRS:{
  9             prefixCls:{
 10                 value:"ke-"
 11             },
 12             zIndex:{
 13                 value:Editor.baseZIndex(Editor.zIndexManager.OVERLAY)
 14             }
 15         }
 16     });
 17 
 18     Overlay4E.Dialog = Component.define(Overlay.Dialog, [Focus], {
 19         show:function () {
 20             var self = this;
 21             //在 show 之前调用
 22             self.center();
 23             var y = self.get("y");
 24             //居中有点偏下
 25             if (y - S.DOM.scrollTop() > 200) {
 26                 y = S.DOM.scrollTop() + 200;
 27                 self.set("y", y);
 28             }
 29             Overlay4E.prototype.show.call(self);
 30         }
 31     }, {
 32         ATTRS:{
 33             prefixCls:{
 34                 value:"ke-"
 35             },
 36             draggable:{
 37                 value:true
 38             },
 39             constrain:{
 40                 value:true
 41             },
 42             aria:{
 43                 value:true
 44             },
 45             zIndex:{
 46                 value:Editor.baseZIndex(Editor.zIndexManager.OVERLAY)
 47             }
 48         }
 49     });
 50 
 51     return Overlay4E
 52 }, {
 53     requires:["editor", 'component', 'overlay', './focus', 'dd']
 54 });
