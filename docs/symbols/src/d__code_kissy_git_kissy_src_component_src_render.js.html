1 /**
  2  * @fileOverview render base class for kissy
  3  * @author yiminghe@gmail.com
  4  * @see http://martinfowler.com/eaaDev/uiArchs.html
  5  */
  6 KISSY.add("component/render", function (S, Component, UIBase, UIStore) {
  7 
  8     /**
  9      * Base Render class for KISSY Component.
 10      * @class
 11      * @memberOf Component
 12      * @name Render
 13      * @extends Component.UIBase
 14      */
 15     return Component.define([UIBase.Box.Render],
 16         /**
 17          * @lends Component.Render#
 18          */
 19         {
 20 
 21             /**
 22              * Get all css class name to be applied to the root element of this component for given state.
 23              * the css class names are prefixed with component name.
 24              * @param {String} [state] This component's state info.
 25              */
 26             getComponentCssClassWithState:function (state) {
 27                 var self = this,
 28                     componentCls = this.__componentClasses;
 29                 state = state || "";
 30                 return self.getCssClassWithPrefix(componentCls.split(/\s+/).join(state + " ") + state);
 31             },
 32 
 33             /**
 34              * Get full class name (with prefix) for current component
 35              * @param classes {String} class names without prefixCls. Separated by space.
 36              * @function
 37              * @return {String} class name with prefixCls
 38              * @private
 39              */
 40             getCssClassWithPrefix:UIStore.getCssClassWithPrefix,
 41 
 42             createDom:function () {
 43                 var self = this;
 44                 self.get("el").addClass(self.getComponentCssClassWithState());
 45             },
 46 
 47             /**
 48              * Returns the dom element which is responsible for listening keyboard events.
 49              * @return {NodeList}
 50              */
 51             getKeyEventTarget:function () {
 52                 return this.get("el");
 53             },
 54 
 55             /**
 56              * Return the dom element into which child component to be rendered.
 57              * @return {NodeList}
 58              */
 59             getContentElement:function () {
 60                 var self = this;
 61                 return self.get("contentEl") || self.get("el");
 62             },
 63 
 64             /**
 65              * @protected
 66              */
 67             _uiSetHighlighted:function (v) {
 68                 var self = this,
 69                     componentCls = self.getComponentCssClassWithState("-hover"),
 70                     el = self.get("el");
 71                 el[v ? 'addClass' : 'removeClass'](componentCls);
 72             },
 73 
 74             /**
 75              * @protected
 76              */
 77             _uiSetDisabled:function (v) {
 78                 var self = this,
 79                     componentCls = self.getComponentCssClassWithState("-disabled"),
 80                     el = self.get("el");
 81                 el[v ? 'addClass' : 'removeClass'](componentCls)
 82                     .attr("aria-disabled", v);
 83                 if (self.get("focusable")) {
 84                     //不能被 tab focus 到
 85                     self.getKeyEventTarget().attr("tabIndex", v ? -1 : 0);
 86                 }
 87             },
 88             /**
 89              * @protected
 90              */
 91             _uiSetActive:function (v) {
 92                 var self = this,
 93                     componentCls = self.getComponentCssClassWithState("-active");
 94                 self.get("el")[v ? 'addClass' : 'removeClass'](componentCls)
 95                     .attr("aria-pressed", !!v);
 96             },
 97             /**
 98              * @protected
 99              */
100             _uiSetFocused:function (v) {
101                 var self = this,
102                     el = self.get("el"),
103                     componentCls = self.getComponentCssClassWithState("-focused");
104                 el[v ? 'addClass' : 'removeClass'](componentCls);
105             }
106 
107         }, {//  screen state
108             ATTRS:/**
109              * @lends Component.Render#
110              */
111             {
112                 /**
113                  * see {@link Component.Controller#prefixCls}
114                  */
115                 prefixCls:{},
116                 /**
117                  * see {@link Component.Controller#focusable}
118                  */
119                 focusable:{},
120                 /**
121                  * see {@link Component.Controller#focused}
122                  */
123                 focused:{},
124                 /**
125                  * see {@link Component.Controller#active}
126                  */
127                 active:{},
128                 /**
129                  * see {@link Component.Controller#disabled}
130                  */
131                 disabled:{},
132                 /**
133                  * see {@link Component.Controller#highlighted}
134                  */
135                 highlighted:{}
136             }
137         });
138 }, {
139     requires:['./base', './uibase', './uistore']
140 });
