  7     function wrapperViewSetter(attrName) {
  8         return function (ev) {
  9             var self = this;
 10             // in case bubbled from sub component
 11             if (self == ev.target) {
 12                 var value = ev.newVal,
 13                     view = self.get("view");
 14                 view.set(attrName, value);
 15             }
 16         };
 17     }
 18 
 19     function wrapperViewGetter(attrName) {
 20         return function (v) {
 21             var self = this,
 22                 view = self.get("view");
 23             return v === undefined ? view.get(attrName) : v;
 24         };
 25     }

 27     function initChild(self, c, renderBefore) {
 28         // 生成父组件的 dom 结构
 29         self.create();
 30         var contentEl = self.getContentElement();
 31         c = Component.create(c, self);
 32         c.__set("parent", self);
 33         // set 通知 view 也更新对应属性
 34         c.set("render", contentEl);
 35         c.set("elBefore", renderBefore);
 36         // 如果 parent 也没渲染，子组件 create 出来和 parent 节点关联
 37         // 子组件和 parent 组件一起渲染
 38         // 之前设好属性，view ，logic 同步还没 bind ,create 不是 render ，还没有 bindUI
 39         c.create(undefined);
 40         return c;
 41     }

 43     /**
 44      * 不使用 valueFn，
 45      * 只有 render 时需要找到默认，其他时候不需要，防止莫名其妙初始化
 46      */
 47     function constructView(self) {
 48         // 逐层找默认渲染器
 49         var attrs,
 50             attrCfg,
 51             attrName,
 52             cfg = {},
 53             v,
 54             Render = self.get('xrender');
 55 
 56         /**
 57          * 将渲染层初始化所需要的属性，直接构造器设置过去
 58          */
 59         attrs = self.getAttrs();
 60 
 61         // 整理属性，对纯属于 view 的属性，添加 getter setter 直接到 view
 62         for (attrName in attrs) {
 63             if (attrs.hasOwnProperty(attrName)) {
 64                 attrCfg = attrs[attrName];
 65                 if (attrCfg.view) {
 66 
 67                     // 先取后 getter
 68                     // 防止死循环
 69                     if (( v = self.get(attrName) ) !== undefined) {
 70                         cfg[attrName] = v;
 71                     }
 72 
 73                     // setter 不应该有实际操作，仅用于正规化比较好
 74                     // attrCfg.setter = wrapperViewSetter(attrName);
 75                     self.on("after" + S.ucfirst(attrName) + "Change",
 76                         wrapperViewSetter(attrName));
 77                     // 逻辑层读值直接从 view 层读
 78                     // 那么如果存在默认值也设置在 view 层
 79                     // 逻辑层不要设置 getter
 80                     attrCfg.getter = wrapperViewGetter(attrName);
 81                 }
 82             }
 83         }
 84         // does not autoRender for view
 85         delete cfg.autoRender;
 86         return new Render(cfg);
 87     }

 89     function setViewCssClassByHierarchy(self, view) {
 90         var constructor = self.constructor,
 91             cls,
 92             re = [];
 93         while (constructor && constructor != Controller) {
 94             cls = Manager.getXClassByConstructor(constructor);
 95             if (cls) {
 96                 re.push(cls);
 97             }
 98             constructor = constructor.superclass && constructor.superclass.constructor;
 99         }
100         return view.__componentClasses = re.join(" ");
101     }

103     function isMouseEventWithinElement(e, elem) {
104         var relatedTarget = e.relatedTarget;
105         // 在里面或等于自身都不算 mouseenter/leave
106         return relatedTarget &&
107             ( relatedTarget === elem[0] ||
108                 elem.contains(relatedTarget) );
109     }

111     function wrapBehavior(self, action) {
112         return self["__ks_wrap_" + action] = function (e) {
113             if (!self.get("disabled")) {
114                 self[action](e);
115             }
116         };
117     }

120         return self["__ks_wrap_" + action];

189                 var self = this,
190                     el = self.getKeyEventTarget();

245             /**
246              * Add the specified component as a child of current component
247              * at the given 0-based index.
248              * @param {Component.Controller|Object} c
249              * Child component instance to be added
250              * or
251              * Object describe child component
252              * @param {String} [c.xclass] When c is a object, specify its child class.
253              * @param {Number} [index]  0-based index at which
254              * the new child component is to be inserted;
255              * If not specified , the new child component will be inserted at last position.
256              */
257             addChild:function (c, index) {
258                 var self = this,
259                     children = self.get("children"),
260                     renderBefore;
261                 if (index === undefined) {
262                     index = children.length;
263                 }
264                 renderBefore = children[index] && children[index].get("el") || null;
265                 c = initChild(self, c, renderBefore);
266                 children.splice(index, 0, c);
267                 // 先 create 占位 再 render
268                 // 防止 render 逻辑里读 parent.get("children") 不同步
269                 // 如果 parent 已经渲染好了子组件也要立即渲染，就 创建 dom ，绑定事件
270                 if (self.get("rendered")) {
271                     c.render();
272                 }
273                 self.fire("addChild", {
274                     child:c
275                 });
276                 return c;
277             },

279             /**
280              * Removed the given child from this component,and returns it.
281              *
282              * If destroy is true, calls {@link Component.UIBase.#destroy} on the removed child component,
283              * and subsequently detaches the child's DOM from the document.
284              * Otherwise it is the caller's responsibility to
285              * clean up the child component's DOM.
286              *
287              * @param {Component.Controller} c The child component to be removed.
288              * @param {Boolean} [destroy=false] If true,
289              * calls {@link Component.UIBase.#destroy} on the removed child component.
290              * @return {Component.Controller} The removed component.
291              */
292             removeChild:function (c, destroy) {
293                 var self = this,
294                     children = self.get("children"),
295                     index = S.indexOf(c, children);
296                 if (index != -1) {
297                     children.splice(index, 1);
298                 }
299                 if (destroy &&
300                     // c is still json
301                     c.destroy) {
302                     c.destroy();
303                 }
304                 self.fire("removeChild", {
305                     child:c
306                 });
307                 return c;
308             },

391             /**
392              * Handles mousedown events. If the component is not disabled,
393              * If the component is activeable, then activate it.
394              * If the component is focusable, then focus it,
395              * else prevent it from receiving keyboard focus.
396              * @protected
397              * @param {Event.Object} ev DOM event to handle.
398              */
399             handleMouseDown:function (ev) {
400                 var self = this,
401                     n,
402                     isMouseActionButton = ev['which'] == 1,
403                     el;
404                 if (isMouseActionButton) {
405                     el = self.getKeyEventTarget();
406                     if (self.get("activeable")) {
407                         self.set("active", true);
408                     }
409                     if (self.get("focusable")) {
410                         el[0].focus();
411                         self.set("focused", true);
412                     }
413 
414                     if (!self.get("allowTextSelection")) {
415                         // firefox /chrome 不会引起焦点转移
416                         n = ev.target.nodeName;
417                         n = n && n.toLowerCase();
418                         // do not prevent focus when click on editable element
419                         if (n != "input" && n != "textarea") {
420                             ev.preventDefault();
421                         }
422                     }
423                 }
424             },

634 /**
635  * 事件冒泡机制
636  *  - child 组件的冒泡源配置为其所属的 parent
637  *  - 性能考虑:不是 child 的所有事件都冒泡到 parent，要具体配置哪些事件需要冒泡
638  *
639  * view 和 controller 的平行关系
640  *  - controller 初始化 -> initializer -> new view()
641  *  - controller createDom -> createDom -> view.createDom()
642  *  - controller renderUI -> renderUI -> view.render()
643  *
644  *
645  * 控制层元属性配置中 view 的作用
646  *   - 如果没有属性变化处理函数，自动生成属性变化处理函数，自动转发给 view 层
647  *   - 如果没有指定 view 层实例，在生成默认 view 实例时，所有用户设置的 view 的属性都转到默认 view 实例中
648  *
649  *
650  * observer synchronization, model 分成两类
651  *  - view 负责监听 view 类 model 变化更新界面
652  *  - control 负责监听 control 类变化改变逻辑
653  *
654  *
655  *
656  * problem: Observer behavior is hard to understand and debug
657  * because it's implicit behavior.
658  *
659  * Keeping screen state and session state synchronized is an important task
660  * Data Binding.
661  *
662  * In general data binding gets tricky
663  * because if you have to avoid cycles where a change to the control,
664  * changes the record set, which updates the control,
665  * which updates the record set....
666  * The flow of usage helps avoid these -
667  * we load from the session state to the screen when the screen is opened,
668  * after that any changes to the screen state propagate back to the session state.
669  * It's unusual for the session state to be updated directly once the screen is up.
670  * As a result data binding might not be entirely bi-directional -
671  * just confined to initial upload and
672  * then propagating changes from the controls to the session state.
673  *
674  *  Refer
675  *    - http://martinfowler.com/eaaDev/uiArchs.html
676  *
677  **/
